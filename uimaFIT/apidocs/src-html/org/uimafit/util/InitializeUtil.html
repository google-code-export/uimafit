058            public static void initialize(Object component, UimaContext context) throws ResourceInitializationException {
059                    initializeParameters(component, context);
060            }
061    
062            /**
063             * This code can be a little confusing because the configuration parameter
064             * annotations are used in two contexts: in describing the component and to
065             * initialize member variables from a {@link UimaContext}. Here we are
066             * performing the latter task. It is important to remember that the
067             * {@link UimaContext} passed in to this method may or may not have been
068             * derived using reflection of the annotations (i.e. using
069             * {@link ConfigurationParameterFactory}). It is just as possible for the
070             * description of the component to come directly from an XML descriptor
071             * file. So, for example, just because the a configuration parameter
072             * specifies a default value, this does not mean that the passed in context
073             * will have a value for that configuration parameter. It should be
074             * possible for a descriptor file to specify its own value or to not
075             * provide one at all. Therefore, we will ignore the defaultValue element of
076             * the {@link ConfigurationParameter} annotation here assuming that if
077             * someone wants to use the defaultValue element for the parameters default
078             * value, then they will use {@link ConfigurationParameterFactory} when
079             * generating the {@link UimaContext} (i.e. the components description will
080             * come from an uimaFIT factory method such as
081             * {@link AnalysisEngineFactory#createPrimitive(Class, org.apache.uima.resource.metadata.TypeSystemDescription, Object...)}
082             * ).
083             * 
084             * @param component
085             * @param context
086             * @throws ResourceInitializationException
087             */
088            public static void initializeParameters(Object component, UimaContext context)
089                            throws ResourceInitializationException {
090                    try {
091                            for (Field field : ReflectionUtil.getFields(component)) { // component.getClass().getDeclaredFields())
092                                    // {
093                                    if (ConfigurationParameterFactory.isConfigurationParameterField(field)) {
094                                            org.uimafit.descriptor.ConfigurationParameter annotation = field
095                                                            .getAnnotation(org.uimafit.descriptor.ConfigurationParameter.class);
096    
097                                            Object parameterValue;
098                                            String configurationParameterName = ConfigurationParameterFactory.getConfigurationParameterName(field);
099                                            
100                                            // Obtain either from the context - or - if the context does
101                                            // not provide the parameter, check if there is a default
102                                            // value. Note there are three possibilities:
103                                            // 1) Parameter present and set
104                                            // 2) Parameter present and set to null (null value)
105                                            // 3) Parameter not present (also provided as null value by UIMA)
106                                            // Unfortunately we cannot make a difference between case 2 and 3 since UIMA
107                                            // does not allow us to actually get a list of the parameters set in the 
108                                            // context. We can only get a list of the declared parameters. Thus we
109                                            // have to rely on the null value. 
110                                            parameterValue = context.getConfigParameterValue(configurationParameterName);
111                                            if (parameterValue == null) {
112                                                    parameterValue = ConfigurationParameterFactory.getDefaultValue(field);
113                                            }
114                                            
115                                            //TODO does this check really belong here?  It seems that this check is already performed by UIMA
116                                            if (annotation.mandatory()) {
117                                                    if (parameterValue == null) {
118                                                            String key = ResourceInitializationException.CONFIG_SETTING_ABSENT;
119                                                            throw new ResourceInitializationException(key, new Object[] { configurationParameterName });
120                                                    }
121                                            }
122                                            else {
123                                                    if (parameterValue == null) {
124                                                            continue;
125                                                    }
126                                            }
127                                            Object fieldValue = convertValue(field, parameterValue);
128                                            setParameterValue(component, field, fieldValue);
129                                    }
130                            }
131                    }
132                    catch (Exception e) {
133                            throw new ResourceInitializationException(e);
134                    }
135    
136            }

138            public static Object convertValue(Field field, Object uimaValue) {
139                    if (ConfigurationParameterFactory.isConfigurationParameterField(field)) {
140    
141                            Object result;
142                            Class<?> fieldType = field.getType();
143                            Class<?> componentType = getComponentType(field);
144                            Converter<?> converter = getConverter(componentType);
145                            
146                            // arrays
147                            if (fieldType.isArray()) {
148                                    Object[] uimaValues = (Object[]) uimaValue;
149                                    result = Array.newInstance(componentType, uimaValues.length);
150                                    for (int index = 0; index < uimaValues.length; ++index) {
151                                            Array.set(result, index, converter.convert(uimaValues[index]));
152                                    }
153                            }
154    
155                            // collections
156                            else if (Collection.class.isAssignableFrom(fieldType)) {
157                                    Collection<Object> collection;
158                                    if (fieldType == List.class) {
159                                            collection = new ArrayList<Object>();
160                                    }
161                                    else if (fieldType == Set.class) {
162                                            collection = new HashSet<Object>();
163                                    }
164                                    else {
165                                            collection = newCollection(fieldType);
166                                    }
167                                    Object[] uimaValues = (Object[]) uimaValue;
168                                    for (Object value : uimaValues) {
169                                            collection.add(converter.convert(value));
170                                    }
171                                    result = collection;
172                            }
173    
174                            // other
175                            else {
176                                    result = converter.convert(uimaValue);
177                            }
178                            return result;
179                    }
180                    else {
181                            throw new IllegalArgumentException("field is not annotated with annotation of type "
182                                            + org.uimafit.descriptor.ConfigurationParameter.class.getName());
183                    }
184            }

186            @SuppressWarnings("unchecked")
187            private static Collection<Object> newCollection(Class<?> cls) {
188                    try {
189                            return cls.asSubclass(Collection.class).newInstance();
190                    } catch (Exception e) {
191                            throw new RuntimeException(e);
192                    }
193            }
194    
195            private static Class<?> getComponentType(Field field) {
196                    Class<?> fieldType = field.getType();
197                    if (fieldType.isArray()) {
198                            return fieldType.getComponentType();
199                    } else if (Collection.class.isAssignableFrom(fieldType)) {
200                            ParameterizedType collectionType = (ParameterizedType)field.getGenericType();
201                            return (Class<?>)collectionType.getActualTypeArguments()[0];
202                    } else {
203                            return fieldType;
204                    }
205            }

209            private static void setParameterValue(Object component, Field field, Object value) throws IllegalArgumentException,
210                            IllegalAccessException, SecurityException, NoSuchMethodException, InvocationTargetException {
211    
212                    boolean accessible = field.isAccessible();
213                    field.setAccessible(true);
214                    try {
215                            field.set(component, value);
216                    } finally {
217                            field.setAccessible(accessible);
218                    }
219            }

224            private static Converter<?> getConverter(Class<?> cls) {
225                    Converter<?> converter = converters.get(cls);
226                    if (converter != null) {
227                            return converter;
228                    }
229                    
230                    // Check if we have an enumeration type
231                    if (Enum.class.isAssignableFrom(cls)) {
232                            @SuppressWarnings("unchecked")
233                            EnumConverter tmp = new EnumConverter(cls);
234                            return tmp;
235                    }
236                    
237                    try {
238                            Constructor<?> constructor = cls.getConstructor(String.class);
239                            return new ConstructorConverter(constructor);
240                    } catch (NoSuchMethodException e) {
241                            throw new IllegalArgumentException("don't know how to convert type " + cls);
242                    }
243            }
244    
245            
246            private static interface Converter<T> {
247                    public T convert(Object o);
248            }
249            private static class BooleanConverter implements Converter<Boolean> {
250                    public Boolean convert(Object o) {
251                            return (Boolean) o;
252                    }
253            }
254            private static class FloatConverter implements Converter<Float> {
255                    public Float convert(Object o) {
256                            return (Float) o;
257                    }
258            }
259            private static class IntegerConverter implements Converter<Integer> {
260                    public Integer convert(Object o) {
261                            return (Integer) o;
262                    }
263            }
264            private static class StringConverter implements Converter<String> {
265                    public String convert(Object o) {
266                            return o.toString();
267                    }
268            }
269    
270            private static class PatternConverter implements Converter<Pattern> {
271                    public Pattern convert(Object o) {
272                            return Pattern.compile(o.toString());
273                    }
274            }
275    
276            private static class ConstructorConverter implements Converter<Object> {
277                    private Constructor<?> constructor;
278                    public ConstructorConverter(Constructor<?> constructor) {
279                            this.constructor = constructor;
280                    }
281                    public Object convert(Object o) {
282                            try {
283                                    return this.constructor.newInstance(o);
284                            } catch (Exception e) {
285                                    throw new RuntimeException(e);
286                            }
287                    }
288                    
289            }
290    
291            private static class EnumConverter<T extends Enum<T>> implements Converter<Object> {
292                    private Class<T> enumClass;
293                    public EnumConverter(Class<T> aClass) {
294                            this.enumClass = aClass;
295                    }
296                    public T convert(Object o) {
297                            try {
298                                    return Enum.valueOf(enumClass, o.toString());
299                            } catch (Exception e) {
300                                    throw new RuntimeException(e);
301                            }
302                    }
303            }
