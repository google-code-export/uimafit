018    package org.uimafit.util;
019    
020    import java.io.IOException;
021    import java.io.OutputStream;
022    import java.io.PrintStream;
023    
024    
025    /**
026     * This class provides a way to hide system output which is sometimes desirable when
027     * running a suite of unit tests which call methods that might contain sysout calls.
028     * This class is a hack and there may be better ways of doing this.  Please advise if you
029     * know better!
030     * 
031     * @author Steven Bethard, Philip Ogren
032     */
033    public class HideOutput extends OutputStream {
034            protected PrintStream out;
035    
036            protected PrintStream err;
037    
038            public HideOutput() {
039                    this.out = System.out;
040                    this.err = System.err;
041                    System.setOut(new PrintStream(this));
042                    System.setErr(new PrintStream(this));
043            }
044    
045            public void restoreOutput() {
046                    System.setOut(this.out);
047                    System.setErr(this.err);
048            }
049    
050            @Override
051            public void write(int b) throws IOException {
052            }
053            
054    }
