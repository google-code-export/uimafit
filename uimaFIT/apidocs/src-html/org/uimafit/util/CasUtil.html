021    package org.uimafit.util;
022    
023    import java.util.ArrayList;
024    import java.util.List;
025    
026    import org.apache.uima.cas.CAS;
027    import org.apache.uima.cas.FSIterator;
028    import org.apache.uima.cas.Type;
029    import org.apache.uima.cas.impl.Subiterator;
030    import org.apache.uima.cas.text.AnnotationFS;
031    
032    /**
033     * Utility methods for convenient access to the {@link CAS}.
034     *
035     * @author Richard Eckart de Castilho
036     */
037    public class CasUtil
038    {
039            /**
040             * Package name of JCas wrapper classes built into UIMA.
041             */
042            public static final String UIMA_BUILTIN_JCAS_PREFIX = "org.apache.uima.jcas.";
043    
044            /**
045             * Get the CAS type for the given JCas wrapper class.
046             *
047             * @param cas the CAS hosting the type system.
048             * @param type the JCas wrapper class.
049             * @return the CAS type.
050             */
051            public static Type getType(CAS cas, Class<?> type)
052            {
053                    String typeName = type.getName();
054                    if (typeName.startsWith(UIMA_BUILTIN_JCAS_PREFIX)) {
055                            typeName = "uima."+typeName.substring(UIMA_BUILTIN_JCAS_PREFIX.length());
056                    }
057                    Type t = cas.getTypeSystem().getType(typeName);
058                    if (t == null) {
059                            throw new IllegalArgumentException("Undeclared type ["+typeName+"]");
060                    }
061                    return t;
062            }
063    
064            /**
065             * Get a list of annotations of the given annotation type constraint by a certain annotation.
066             * Iterates over all annotations of the given type to find the covered annotations. Does not use
067             * subiterators and does not respect type prioritites. Was adapted from {@link Subiterator}.
068             * Uses the same approach except that type priorities are ignored.
069             *
070             * @param <T> the JCas type.
071             * @param cas a JCas.
072             * @param type a UIMA type.
073             * @param coveringAnnotation the covering annotation.
074             * @return a return value.
075             * @see Subiterator
076             */
077            public static <T extends AnnotationFS> List<T> selectCovered(
078                            CAS cas, Type type, AnnotationFS coveringAnnotation)
079            {
080                    int begin = coveringAnnotation.getBegin();
081                    int end = coveringAnnotation.getEnd();
082    
083                    List<T> list = new ArrayList<T>();
084                    FSIterator<AnnotationFS> it = cas.getAnnotationIndex(type).iterator();
085    
086                    // Try to seek the insertion point.
087                    it.moveTo(coveringAnnotation);
088    
089                    // If the insertion point is beyond the index, move back to the last.
090                    if (!it.isValid()) {
091                            it.moveToLast();
092                            if (!it.isValid()) {
093                                    return list;
094                            }
095                    }
096    
097                    // Ignore type priorities by seeking to the first that has the same begin
098                    boolean moved = false;
099                    while (it.isValid() && (it.get()).getBegin() >= begin) {
100                            it.moveToPrevious();
101                            moved = true;
102                    }
103    
104                    // If we moved, then we are now on one starting before the requested begin, so we have to
105                    // move one ahead.
106                    if (moved) {
107                            it.moveToNext();
108                    }
109    
110                    // If we managed to move outside the index, start at first.
111                    if (!it.isValid()) {
112                            it.moveToFirst();
113                    }
114    
115                    // Skip annotations whose start is before the start parameter.
116                    while (it.isValid() && (it.get()).getBegin() < begin) {
117                            it.moveToNext();
118                    }
119    
120                    boolean strict = true;
121                    while (it.isValid()) {
122                            @SuppressWarnings("unchecked")
123                            T a = (T) it.get();
124                            // If the start of the current annotation is past the end parameter, we're done.
125                            if (a.getBegin() > end) {
126                                    break;
127                            }
128                            it.moveToNext();
129                            if (strict && a.getEnd() > end) {
130                                    continue;
131                            }
132    
133                            if (!a.equals(coveringAnnotation)) {
134                                    list.add(a);
135                            }
136                    }
137    
138                    return list;
139            }
140    }
