018    package org.uimafit.util;
019    
020    import org.apache.uima.tools.jcasgen.Jg;
021    import org.apache.uima.tools.jcasgen.LogThrowErrorImpl;
022    
023    /**
024     * This class provides a main method that is identical to the main method in
025     * org.apache.uima.tools.jcasgen.Jg except that it does not call System.exit.
026     * This makes it much more friendly to use with a pom.xml file. See, for
027     * example, the pom.xml file in this project for an example usage.
028     * 
029     * @author Philip Ogren
030     */
031    
032    public class JCasGenPomFriendly {
033    
034            public static void main(String[] args) {
035                    (new Jg()).main0(args, null, null, new LogThrowErrorImpl());
036            }
037    
038    }
