017    package org.uimafit.descriptor;
018    
019    import java.lang.annotation.ElementType;
020    import java.lang.annotation.Retention;
021    import java.lang.annotation.RetentionPolicy;
022    import java.lang.annotation.Target;
023     
024    /**
025     * This annotation marks an analysis component member variable as a
026     * configuration parameter. The types of member variables that can use this
027     * annotation are:
028     * <ul>
029     * <li>String</li>
030     * <li>String[]</li>
031     * <li>Boolean</li>
032     * <li>boolean</li>
033     * <li>Boolean[]</li>
034     * <li>boolean[]</li>
035     * <li>Integer</li>
036     * <li>int</li>
037     * <li>Integer[]</li>
038     * <li>int[]</li>
039     * <li>Float</li>
040     * <li>float</li>
041     * <li>Float[]</li>
042     * <li>float[]</li>
043     * </ul>
044     * 
045     * Fields marked with this annotation should be declared public or have a setter
046     * method. The setter method should be named according to setter naming
047     * convention - e.g. a field named "myConfigurationParameter" should have a
048     * corresponding setter named "setMyConfigurationParameter" with a single
049     * parameter whose type is the same as the type of the field (i.e. you can not
050     * mix and match the object or primitive types here).
051     * 
052     * @author Philip Ogren
053     * 
054     */
055    
056    @Retention(RetentionPolicy.RUNTIME)
057    @Target(ElementType.FIELD)
058    public @interface ConfigurationParameter {
059    
060            public static final String USE_FIELD_NAME = "org.uutuc.descriptor.ConfigurationParameter.USE_FIELD_NAME";
061    
062            /**
063             * If you do not specify a name then the default name will be given by {@link #USE_FIELD_NAME} will be the default name.  This 
064             * tells ConfigurationParameterFactory to use the name of the annotated field as the name of the configuration parameter.
065             */
066            String name() default USE_FIELD_NAME;
067    
068            String description() default "";
069    
070            boolean mandatory() default false;
071    
072            /**
073             * What can be the value should correspond with the type of the field that
074             * is annotated. If for example, the field is a String, then the default
075             * value can be any string - e.g. "asdf". If the field is a boolean, then
076             * the default value can be "true" for true or any other string for false.
077             * If the field is an integer, then the default value can be any string that
078             * Integer.parseInt() can successfully parse. Remember that just because the
079             * default value is a string here that you should give an actual integer
080             * (not an integer parseable string) value when setting the parameter via
081             * e.g. AnalysisEngineFactory.createPrimitiveDescription(). If the field is
082             * a float, then the default value can be any string that
083             * Float.parseFloat() can successfully parse. Remember that just because the
084             * default value is a string here that you should give an actual float value
085             * (not a float parseable string) when setting the parameter via e.g.
086             * AnalysisEngineFactory.createPrimitiveDescription().
087             * <p>
088             * If the field is multiValued, then the value should look something like
089             * this '{"value1", "value2"}' 
090             * <p>
091             * If you want a field to be initialized with a null value, then do not specify a default value or specify the value given by the field {@link #NO_DEFAULT_VALUE}
092             */
093            String[] defaultValue() default NO_DEFAULT_VALUE;
094    
095            public static final String NO_DEFAULT_VALUE = "org.uutuc.descriptor.ConfigurationParameter.NO_DEFAULT_VALUE";
096    
097    }
