018    package org.uimafit.descriptor;
019    
020    import java.lang.annotation.ElementType;
021    import java.lang.annotation.Retention;
022    import java.lang.annotation.RetentionPolicy;
023    import java.lang.annotation.Target;
024    
025    /**
026     * @author Philip Ogren
027     *  A typical use of this annotation might look something like:
028     *  <pre>
029                    @SofaCapability(inputSofas = { GOLD_VIEW, SYSTEM_VIEW })
030                    or
031                    @SofaCapability(inputSofas = CAS.NAME_DEFAULT_SOFA, outputSofas = GOLD_VIEW)
032            </pre>
033    
034     * 
035     * 
036     *         Adding this annotation to your analysis engine description makes your
037     *         component "sofa aware." The base CAS delivered to "sofa aware"
038     *         components has no explicit view associated with it. The logic is that
039     *         it is impossible to know the intent of a sofa aware component and it
040     *         should use getView as needed. You should therefore be aware that if
041     *         you need to work with the "_InitialView" view, then you must
042     *         explicitly request it with a call like:
043     * 
044     *         JCas initialView = jCas.getView(CAS.NAME_DEFAULT_SOFA);
045     * 
046     *         This is because the base CAS that it passed into the process method
047     *         to "sofa aware" components is not the same as the "_InitialView". See
048     *         how the following member variable is used to understand why/how:
049     * 
050     *         org.apache.uima.analysis_engine.impl.PrimitiveAnalysisEngine_impl.
051     *         mSofaAware
052     */
053    
054    @Retention(RetentionPolicy.RUNTIME)
055    @Target(ElementType.TYPE)
056    public @interface SofaCapability {
057            /**
058             * the values should be string vales corresponding to view names such as e.g. CAS.NAME_DEFAULT_SOFA
059             * that this analysis component expects to be present in the CAS.
060             */
061            String[] inputSofas() default NO_DEFAULT_VALUE;
062            /**
063             * the values should be string vales corresponding to view names 
064             * that this analysis component will create.  
065             */
066            String[] outputSofas() default NO_DEFAULT_VALUE;
067    
068            /**
069             * Provides the default value for the inputs and the outputs that tells the CapabilityFactory that no value has been given to the inputs or outputs elements.  
070             */
071    
072            public static final String NO_DEFAULT_VALUE = "org.uimafit.descriptor.SofaCapability.NO_DEFAULT_VALUE";
073    
074    }
