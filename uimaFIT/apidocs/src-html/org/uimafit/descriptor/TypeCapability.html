018    package org.uimafit.descriptor;
019    
020    import java.lang.annotation.ElementType;
021    import java.lang.annotation.Retention;
022    import java.lang.annotation.RetentionPolicy;
023    import java.lang.annotation.Target;
024    
025    /**
026     * @author Philip Ogren
027     * 
028     */
029    
030    @Retention(RetentionPolicy.RUNTIME)
031    @Target(ElementType.TYPE)
032    public @interface TypeCapability {
033    
034            /**
035             * inputs can be type names or feature names. A feature name typically looks
036             * like a type name followed by a colon (':') followed by the feature name.  A valid feature name from
037             * the uutuc test type system is "org.uutuc.type.Token:pos"
038             * 
039             */
040            String[] inputs() default NO_DEFAULT_VALUE;
041    
042            String[] outputs() default NO_DEFAULT_VALUE;
043    
044            public static final String NO_DEFAULT_VALUE = "org.uutuc.descriptor.TypeCapability.NO_DEFAULT_VALUE";
045    
046    }
