018    package org.uimafit.descriptor;
019    
020    import java.lang.annotation.ElementType;
021    import java.lang.annotation.Retention;
022    import java.lang.annotation.RetentionPolicy;
023    import java.lang.annotation.Target;
024    
025    /**
026     * @author Philip Ogren
027     * A typical use of this annotation might look something like:
028     * <pre>
029            @TypeCapability(inputs="org.uimafit.type.Token", outputs="org.uimafit.type.Token:pos")
030            </pre>
031     * or
032     * <pre>
033            @TypeCapability(inputs={"org.uimafit.type.Token","org.uimafit.type.Sentence"}, outputs={"org.uimafit.type.Token:pos", "org.apache.uima.tutorial.RoomNumber"})
034            </pre>    
035     */
036    
037    @Retention(RetentionPolicy.RUNTIME)
038    @Target(ElementType.TYPE)
039    public @interface TypeCapability {
040    
041            /**
042             * inputs can be type names or feature names. A feature name typically looks
043             * like a type name followed by a colon (':') followed by the feature name.  A valid feature name from
044             * the uimaFIT test type system is "org.uimafit.type.Token:pos"
045             * 
046             */
047            String[] inputs() default NO_DEFAULT_VALUE;
048    
049            /**
050             * outputs can be type names or feature names. A feature name typically looks
051             * like a type name followed by a colon (':') followed by the feature name.  A valid feature name from
052             * the uimaFIT test type system is "org.uimafit.type.Token:pos"
053             * @return
054             */
055            String[] outputs() default NO_DEFAULT_VALUE;
056    
057            /**
058             * Provides the default value for the inputs and the outputs that tells the CapabilityFactory that no value has been given to the inputs or outputs elements.  
059             */
060            public static final String NO_DEFAULT_VALUE = "org.uimafit.descriptor.TypeCapability.NO_DEFAULT_VALUE";
061    
062    }
