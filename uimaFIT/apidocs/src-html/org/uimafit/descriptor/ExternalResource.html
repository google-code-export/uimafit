020    package org.uimafit.descriptor;
021    
022    import java.lang.annotation.ElementType;
023    import java.lang.annotation.Retention;
024    import java.lang.annotation.RetentionPolicy;
025    import java.lang.annotation.Target;
026    
027    import org.apache.uima.resource.Resource;
028    
029    /**
030     * Mark a field as external resource.
031     * 
032     * @author Richard Eckart de Castilho
033     */
034    @Retention(RetentionPolicy.RUNTIME)
035    @Target(ElementType.FIELD)
036    public @interface ExternalResource
037    {
038            /**
039             * The key to which external resources bind to. If no key is set, the class
040             * name of the annotated field will be used.
041             * 
042             * @return the key;
043             */
044            String key() default "";
045            
046            /**
047             * The interface that external resources need to implement. Normally this
048             * has to be the type of the field, but if {@link ExternalResourceLocator}s
049             * are used, this should be set to {@link ExternalResourceLocator}
050             * or to a derived interface.
051             * 
052             * @return the required interface.
053             */
054            Class<? extends Resource> api() default Resource.class;
055            
056            /**
057             * determines if this external resource is mandatory
058             * @return
059             */
060            boolean mandatory() default true;
061    }
