025    package org.uimafit.descriptor;
026    
027    import java.lang.annotation.ElementType;
028    import java.lang.annotation.Retention;
029    import java.lang.annotation.RetentionPolicy;
030    import java.lang.annotation.Target;
031    
032    import org.apache.uima.analysis_engine.AnalysisEngine;
033    import org.apache.uima.cas.CAS;
034    
035    
036    /**
037     * Control aspects of the UIMA analysis component.<br/>
038     * 
039     * @author Richard Eckart de Castilho
040     */
041    @Retention(RetentionPolicy.RUNTIME)
042    @Target(ElementType.TYPE)
043    public @interface OperationalProperties
044    {
045            public static final boolean MULTIPLE_DEPLOYMENT_ALLOWED_DEFAULT = true;
046            public static final boolean MODIFIES_CAS_DEFAULT = true;        
047            public static final boolean OUTPUTS_NEW_CASES_DEFAULT = false;
048            
049            /**
050             * Gets whether multiple instances of this component can be run in parallel,
051             * each receiving a subset of the documents from a collection.
052             * 
053             * @return true if multiple instances can be run in parallel, false if not
054             * @see org.apache.uima.resource.metadata.OperationalProperties#isMultipleDeploymentAllowed()
055             */
056            boolean multipleDeploymentAllowed() default MULTIPLE_DEPLOYMENT_ALLOWED_DEFAULT;
057    
058            /**
059             * Gets whether this component will modify the CAS.
060             * 
061             * @return true if this component modifies the CAS, false if it does not.
062             * @see org.apache.uima.resource.metadata.OperationalProperties#getModifiesCas()
063             */
064            boolean modifiesCas() default MODIFIES_CAS_DEFAULT;
065    
066            /**
067             * Gets whether this AnalysisEngine may output new CASes. If this property
068             * is set to true, an application can use the
069             * {@link AnalysisEngine#processAndOutputNewCASes(CAS)} to pass a CAS to
070             * this this AnalysisEngine and then step through all of the output CASes
071             * that it produces. For example, such an AnalysisEngine could segment a CAS
072             * into smaller pieces, emitting each as a separate CAS.
073             * 
074             * @return true if this component may output new CASes, false if it does not
075             * @see org.apache.uima.resource.metadata.OperationalProperties#getOutputsNewCASes()
076             */
077            boolean outputsNewCases() default OUTPUTS_NEW_CASES_DEFAULT;
078    }
