018    package org.uimafit.factory.initializable;
019    
020    import org.apache.uima.UimaContext;
021    import org.apache.uima.resource.ResourceInitializationException;
022    
023    /**
024     * <br>
025     * Copyright (c) 2010, Regents of the University of Colorado <br>
026     * All rights reserved.
027     * 
028     * @author Philip Ogren
029     * 
030     *         This interface provides a way of initializing a class with a
031     *         UimaContext. The standard use case of this interface involves
032     *         situations in which a member variable is declared with an interface
033     *         type and the actual class that is used is decided at runtime. When
034     *         the member variable is instantiated, it is useful to provide it a
035     *         UimaContext so that it can initialize itself based on configuration
036     *         parameters.
037     *         <p>
038     *         As an example, consider the component
039     *         org.uimafit.component.xwriter.XWriter which has a member variable of
040     *         type XWriterFileNamer which is an interface that provides a way for
041     *         XWriter to come up with a name each file that is generated given a
042     *         JCas. The default XWriterFileNamer that is provided,
043     *         IntegerFileNamer, provides a single configuration parameter for
044     *         specifying a prefix to each file name that is generated. This
045     *         parameter is initialized in the IntegerFileNamer.initialize() method.
046     *         This initialize method is called because XWriter instantiates its
047     *         XWriterFileNamer member variable using a InitializableFactory.create
048     *         method. Therefore, when XWriter is instantiated it should be given a
049     *         value for the configuration parameter IntegerFileNamer.PARAM_PREFIX.
050     *         See the unit tests for XWriter for a complete code example. Note that
051     *         the implementation of XWriterFileNamer does not have to implement
052     *         Initializable if it has no need for the initialize method.
053     */
054    
055    public interface Initializable {
056    
057            public void initialize(UimaContext context) throws ResourceInitializationException;
058    
059    }
