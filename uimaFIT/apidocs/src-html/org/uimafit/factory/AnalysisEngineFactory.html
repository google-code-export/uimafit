072         * Get an AnalysisEngine from the name (Java-style, dotted) of an XML descriptor file, and a set of configuration
073         * parameters.
074         * 
075         * @param descriptorName The fully qualified, Java-style, dotted name of the XML descriptor file.
076         * @param configurationData Any additional configuration parameters to be set. These should be supplied as (name,
077         *            value) pairs, so there should always be an even number of parameters.
078         * @return The AnalysisEngine created from the XML descriptor and the configuration parameters.
079         * @throws IOException
080         * @throws UIMAException
081         */
082        public static AnalysisEngine createAnalysisEngine(String descriptorName, Object... configurationData)
083            throws UIMAException, IOException
084        {
085            AnalysisEngineDescription aed = createAnalysisEngineDescription(descriptorName, configurationData);
086            return UIMAFramework.produceAnalysisEngine(aed);
087        }

090            Object... configurationData) throws UIMAException, IOException
091        {
092            Import_impl imprt = new Import_impl();
093            imprt.setName(descriptorName);
094            URL url = imprt.findAbsoluteUrl(UIMAFramework.newDefaultResourceManager());
095            ResourceSpecifier specifier =
096                ResourceCreationSpecifierFactory.createResourceCreationSpecifier(url, configurationData);
097            return (AnalysisEngineDescription) specifier;

102         * view name passed into the method.
103         * 
104         * @param analysisEngineDescription
105         * @param viewName the view name to map the default view to
106         * @return an aggregate analysis engine consisting of a single component whose default view is mapped to the the

187        public static AnalysisEngineDescription createPrimitiveDescription(
188            Class< ? extends AnalysisComponent> componentClass, TypeSystemDescription typeSystem,
189            TypePriorities typePriorities, Capability[] capabilities, ConfigurationParameter[] configurationParameters,
190            Object[] configurationValues) throws ResourceInitializationException
191        {
192    
193            // create the descriptor and set configuration parameters
194            AnalysisEngineDescription desc = new AnalysisEngineDescription_impl();
195            desc.setFrameworkImplementation(Constants.JAVA_FRAMEWORK_NAME);
196            desc.setPrimitive(true);
197            desc.setAnnotatorImplementationName(componentClass.getName());
198            org.uimafit.descriptor.OperationalProperties componentAnno =
199                ReflectionUtil.getInheritableAnnotation(org.uimafit.descriptor.OperationalProperties.class, componentClass);
200            if (componentAnno != null) {
201                OperationalProperties op = desc.getAnalysisEngineMetaData().getOperationalProperties();
202                op.setMultipleDeploymentAllowed(componentAnno.multipleDeploymentAllowed());
203                op.setModifiesCas(componentAnno.modifiesCas());
204                op.setOutputsNewCASes(componentAnno.outputsNewCases());
205            } else {
206                OperationalProperties op = desc.getAnalysisEngineMetaData().getOperationalProperties();
207                op
208                    .setMultipleDeploymentAllowed(org.uimafit.descriptor.OperationalProperties.MULTIPLE_DEPLOYMENT_ALLOWED_DEFAULT);
209                op.setModifiesCas(org.uimafit.descriptor.OperationalProperties.MODIFIES_CAS_DEFAULT);
210                op.setOutputsNewCASes(org.uimafit.descriptor.OperationalProperties.OUTPUTS_NEW_CASES_DEFAULT);
211            }
212    
213            AnalysisEngineMetaData meta = desc.getAnalysisEngineMetaData();
214            meta.setName(componentClass.getName());
215            meta.setVendor(componentClass.getPackage().getName());
216            meta.setDescription("Descriptor automatically generated by uimaFIT");
217            meta.setVersion("unknown");
218    
219            // Extract external resource dependencies
220            Collection<ExternalResourceDependency> deps =
221                ExternalResourceInitializer.getResourceDeclarations(componentClass).values();
222            desc.setExternalResourceDependencies(deps.toArray(new ExternalResourceDependency[deps.size()]));
223    
224            ConfigurationData reflectedConfigurationData =
225                ConfigurationParameterFactory.createConfigurationData(componentClass);
226            ResourceCreationSpecifierFactory.setConfigurationParameters(desc,
227                reflectedConfigurationData.configurationParameters, reflectedConfigurationData.configurationValues);
228            if (configurationParameters != null) {
229                ResourceCreationSpecifierFactory.setConfigurationParameters(desc, configurationParameters,
230                    configurationValues);
231            }
232    
233            // set the type system
234            if (typeSystem != null) {
235                desc.getAnalysisEngineMetaData().setTypeSystem(typeSystem);
236            }
237    
238            if (typePriorities != null) {
239                desc.getAnalysisEngineMetaData().setTypePriorities(typePriorities);
240            }
241    
242            if (capabilities == null) {
243                Capability capability = CapabilityFactory.createCapability(componentClass);
244                if (capability != null) {
245                    capabilities = new Capability[] {capability};
246                }
247            }
248            if (capabilities != null) {
249                desc.getAnalysisEngineMetaData().setCapabilities(capabilities);
250            }
251            return desc;
252        }

275        @SuppressWarnings("unchecked")
276        public static AnalysisEngine createPrimitive(AnalysisEngineDescription desc, Object... configurationData)
277            throws ResourceInitializationException
278        {
279            AnalysisEngine engine = new PrimitiveAnalysisEngine_impl();
280    
281            if (configurationData != null) {
282                ConfigurationData cdata = ConfigurationParameterFactory.createConfigurationData(configurationData);
283                ConfigurationParameter[] configurationParameters = cdata.configurationParameters;
284                Object[] configurationValues = cdata.configurationValues;
285                ResourceCreationSpecifierFactory.setConfigurationParameters(desc, configurationParameters,
286                    configurationValues);
287                ResourceMetaData metaData = desc.getMetaData();
288                ConfigurationParameterSettings paramSettings = metaData.getConfigurationParameterSettings();
289                Map additionalParameters = new HashMap();
290                additionalParameters.put(AnalysisEngine.PARAM_CONFIG_PARAM_SETTINGS, paramSettings);
291                engine.initialize(desc, additionalParameters);
292            } else {
293                engine.initialize(desc, null);
294            }
295            return engine;
296        }

431        public static AnalysisEngineDescription createAggregateDescription(
432            List<AnalysisEngineDescription> analysisEngineDescriptions, List<String> componentNames,
433            TypeSystemDescription typeSystem, TypePriorities typePriorities, SofaMapping[] sofaMappings,
434            FlowControllerDescription flowControllerDescription)
435            throws ResourceInitializationException
436        {
437    
438            // create the descriptor and set configuration parameters
439            AnalysisEngineDescription desc = new AnalysisEngineDescription_impl();
440            desc.setFrameworkImplementation(Constants.JAVA_FRAMEWORK_NAME);
441            desc.setPrimitive(false);
442    
443            // if any of the aggregated analysis engines does not allow multiple
444            // deployment, then the
445            // aggregate engine may also not be multiply deployed
446            boolean allowMultipleDeploy = true;
447            for (AnalysisEngineDescription d : analysisEngineDescriptions) {
448                allowMultipleDeploy &=
449                    d.getAnalysisEngineMetaData().getOperationalProperties().isMultipleDeploymentAllowed();
450            }
451            desc.getAnalysisEngineMetaData().getOperationalProperties().setMultipleDeploymentAllowed(allowMultipleDeploy);
452    
453            List<String> flowNames = new ArrayList<String>();
454    
455            for (int i = 0; i < analysisEngineDescriptions.size(); i++) {
456                AnalysisEngineDescription aed = analysisEngineDescriptions.get(i);
457                String componentName = componentNames.get(i);
458                desc.getDelegateAnalysisEngineSpecifiersWithImports().put(componentName, aed);
459                flowNames.add(componentName);
460            }
461    
462            if (flowControllerDescription != null) {
463               FlowControllerDeclaration flowControllerDeclaration = new FlowControllerDeclaration_impl();
464                flowControllerDeclaration.setSpecifier(flowControllerDescription);
465                desc.setFlowControllerDeclaration(flowControllerDeclaration);
466            }
467    
468            FixedFlow fixedFlow = new FixedFlow_impl();
469            fixedFlow.setFixedFlow(flowNames.toArray(new String[flowNames.size()]));
470            desc.getAnalysisEngineMetaData().setFlowConstraints(fixedFlow);
471    
472            if (typePriorities != null) {
473                desc.getAnalysisEngineMetaData().setTypePriorities(typePriorities);
474            }
475    
476            if (sofaMappings != null) {
477                desc.setSofaMappings(sofaMappings);
478            }
479    
480            return desc;
481        }

484        /**
485         * Creates an AnalysisEngine from the given descriptor, and uses the engine to process the file or text.
486         * 
487         * @param descriptorFileName The fully qualified, Java-style, dotted name of the XML descriptor file.
488         * @param fileNameOrText Either the path of a file to be loaded, or a string to use as the text. If the string given
489         *            is not a valid path in the file system, it will be assumed to be text.
490         * @return A JCas object containing the processed document.
491         * @throws IOException
492         * @throws UIMAException
493         */
494        public static JCas process(String descriptorFileName, String fileNameOrText) throws IOException, UIMAException
495        {
496            AnalysisEngine engine = createAnalysisEngine(descriptorFileName);
497            JCas jCas = process(engine, fileNameOrText);
498            engine.collectionProcessComplete();
499            return jCas;
500        }
501    
502        /**
503         * Processes the file or text with the given AnalysisEngine.
504         * 
505         * @param analysisEngine The AnalysisEngine object to process the text.
506         * @param fileNameOrText Either the path of a file to be loaded, or a string to use as the text. If the string given
507         *            is not a valid path in the file system, it will be assumed to be text.
508         * @return A JCas object containing the processed document.
509         * @throws IOException
510         * @throws UIMAException
511         */
512        public static JCas process(AnalysisEngine analysisEngine, String fileNameOrText) throws IOException, UIMAException
513        {
514            JCas jCas = analysisEngine.newJCas();
515            process(jCas, analysisEngine, fileNameOrText);
516            return jCas;
517        }
518    
519        public static void process(JCas jCas, AnalysisEngine analysisEngine, String fileNameOrText) throws IOException, UIMAException {
520            File textFile = new File(fileNameOrText);
521            String text;
522            if (textFile.exists()) {
523                text = FileUtils.file2String(textFile);
524            } else {
525                text = fileNameOrText;
526            }
527    
528            jCas.setDocumentText(text);
529            analysisEngine.process(jCas);
530         }
