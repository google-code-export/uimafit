080         * @param configurationData Any additional configuration parameters to be set. These should be supplied as (name,

082         * @return The AnalysisEngine created from the XML descriptor and the configuration parameters.

090            return UIMAFramework.produceAnalysisEngine(aed);

095         * @param descriptorName
096         * @param configurationData should consist of name value pairs. Will override configuration parameter settings in the descriptor file
097         * @return
098         * @throws UIMAException
099         * @throws IOException
100         */
101        public static AnalysisEngineDescription createAnalysisEngineDescription(String descriptorName,
102            Object... configurationData) throws UIMAException, IOException
103        {
104            Import_impl imprt = new Import_impl();
105            imprt.setName(descriptorName);
106            URL url = imprt.findAbsoluteUrl(UIMAFramework.newDefaultResourceManager());
107            ResourceSpecifier specifier =
108                ResourceCreationSpecifierFactory.createResourceCreationSpecifier(url, configurationData);
109            return (AnalysisEngineDescription) specifier;

233        /**
234         * @param componentClass
235         * @param typeSystem
236         * @param typePriorities
237         * @param capabilities
238         * @param configurationParameters
239         * @param configurationValues
240         * @return
241         * @throws ResourceInitializationException
242         */
243        public static AnalysisEngineDescription createPrimitiveDescription(
244            Class< ? extends AnalysisComponent> componentClass, TypeSystemDescription typeSystem,
245            TypePriorities typePriorities, Capability[] capabilities, ConfigurationParameter[] configurationParameters,
246            Object[] configurationValues) throws ResourceInitializationException
247        {
248    
249            // create the descriptor and set configuration parameters
250            AnalysisEngineDescription desc = new AnalysisEngineDescription_impl();
251            desc.setFrameworkImplementation(Constants.JAVA_FRAMEWORK_NAME);
252            desc.setPrimitive(true);
253            desc.setAnnotatorImplementationName(componentClass.getName());
254            org.uimafit.descriptor.OperationalProperties componentAnno =
255                ReflectionUtil.getInheritableAnnotation(org.uimafit.descriptor.OperationalProperties.class, componentClass);
256            if (componentAnno != null) {
257                OperationalProperties op = desc.getAnalysisEngineMetaData().getOperationalProperties();
258                op.setMultipleDeploymentAllowed(componentAnno.multipleDeploymentAllowed());
259                op.setModifiesCas(componentAnno.modifiesCas());
260                op.setOutputsNewCASes(componentAnno.outputsNewCases());
261            } else {
262                OperationalProperties op = desc.getAnalysisEngineMetaData().getOperationalProperties();
263                op
264                    .setMultipleDeploymentAllowed(org.uimafit.descriptor.OperationalProperties.MULTIPLE_DEPLOYMENT_ALLOWED_DEFAULT);
265                op.setModifiesCas(org.uimafit.descriptor.OperationalProperties.MODIFIES_CAS_DEFAULT);
266                op.setOutputsNewCASes(org.uimafit.descriptor.OperationalProperties.OUTPUTS_NEW_CASES_DEFAULT);
267            }
268    
269            AnalysisEngineMetaData meta = desc.getAnalysisEngineMetaData();
270            meta.setName(componentClass.getName());
271            meta.setVendor(componentClass.getPackage().getName());
272            meta.setDescription("Descriptor automatically generated by uimaFIT");
273            meta.setVersion("unknown");
274    
275            // Extract external resource dependencies
276            Collection<ExternalResourceDependency> deps =
277                ExternalResourceInitializer.getResourceDeclarations(componentClass).values();
278            desc.setExternalResourceDependencies(deps.toArray(new ExternalResourceDependency[deps.size()]));
279    
280            ConfigurationData reflectedConfigurationData =
281                ConfigurationParameterFactory.createConfigurationData(componentClass);
282            ResourceCreationSpecifierFactory.setConfigurationParameters(desc,
283                reflectedConfigurationData.configurationParameters, reflectedConfigurationData.configurationValues);
284            if (configurationParameters != null) {
285                ResourceCreationSpecifierFactory.setConfigurationParameters(desc, configurationParameters,
286                    configurationValues);
287            }
288    
289            // set the type system
290            if (typeSystem != null) {
291                desc.getAnalysisEngineMetaData().setTypeSystem(typeSystem);
292            }
293    
294            if (typePriorities != null) {
295                desc.getAnalysisEngineMetaData().setTypePriorities(typePriorities);
296            }
297    
298            if (capabilities == null) {
299                Capability capability = CapabilityFactory.createCapability(componentClass);
300                if (capability != null) {
301                    capabilities = new Capability[] {capability};
302                }
303            }
304            if (capabilities != null) {
305                desc.getAnalysisEngineMetaData().setCapabilities(capabilities);
306            }
307            return desc;
308        }

345        /**
346         * @param desc
347         * @param configurationData
348         * @return
349         * @throws ResourceInitializationException
350         */
351        public static AnalysisEngine createPrimitive(AnalysisEngineDescription desc, Object... configurationData)
352            throws ResourceInitializationException
353        {
354            AnalysisEngine engine = new PrimitiveAnalysisEngine_impl();
355    
356            if (configurationData != null) {
357                ConfigurationData cdata = ConfigurationParameterFactory.createConfigurationData(configurationData);
358                ConfigurationParameter[] configurationParameters = cdata.configurationParameters;
359                Object[] configurationValues = cdata.configurationValues;
360                ResourceCreationSpecifierFactory.setConfigurationParameters(desc, configurationParameters,
361                    configurationValues);
362                ResourceMetaData metaData = desc.getMetaData();
363                ConfigurationParameterSettings paramSettings = metaData.getConfigurationParameterSettings();
364                Map<String, Object> additionalParameters = new HashMap<String, Object>();
365                additionalParameters.put(AnalysisEngine.PARAM_CONFIG_PARAM_SETTINGS, paramSettings);
366                engine.initialize(desc, additionalParameters);
367            } else {
368                engine.initialize(desc, null);
369            }
370            return engine;
371        }

580        /**
581         * A factory method for creating an aggregate description.
582         * @param analysisEngineDescriptions
583         * @param componentNames
584         * @param typeSystem
585         * @param typePriorities
586         * @param sofaMappings
587         * @param flowControllerDescription
588         * @return
589         * @throws ResourceInitializationException
590         */
591        public static AnalysisEngineDescription createAggregateDescription(
592            List<AnalysisEngineDescription> analysisEngineDescriptions, List<String> componentNames,
593            TypeSystemDescription typeSystem, TypePriorities typePriorities, SofaMapping[] sofaMappings,
594            FlowControllerDescription flowControllerDescription)
595            throws ResourceInitializationException
596        {
597    
598            // create the descriptor and set configuration parameters
599            AnalysisEngineDescription desc = new AnalysisEngineDescription_impl();
600            desc.setFrameworkImplementation(Constants.JAVA_FRAMEWORK_NAME);
601            desc.setPrimitive(false);
602    
603            // if any of the aggregated analysis engines does not allow multiple
604            // deployment, then the
605            // aggregate engine may also not be multiply deployed
606            boolean allowMultipleDeploy = true;
607            for (AnalysisEngineDescription d : analysisEngineDescriptions) {
608                allowMultipleDeploy &=
609                    d.getAnalysisEngineMetaData().getOperationalProperties().isMultipleDeploymentAllowed();
610            }
611            desc.getAnalysisEngineMetaData().getOperationalProperties().setMultipleDeploymentAllowed(allowMultipleDeploy);
612    
613            List<String> flowNames = new ArrayList<String>();
614    
615            for (int i = 0; i < analysisEngineDescriptions.size(); i++) {
616                AnalysisEngineDescription aed = analysisEngineDescriptions.get(i);
617                String componentName = componentNames.get(i);
618                desc.getDelegateAnalysisEngineSpecifiersWithImports().put(componentName, aed);
619                flowNames.add(componentName);
620            }
621    
622            if (flowControllerDescription != null) {
623               FlowControllerDeclaration flowControllerDeclaration = new FlowControllerDeclaration_impl();
624                flowControllerDeclaration.setSpecifier(flowControllerDescription);
625                desc.setFlowControllerDeclaration(flowControllerDeclaration);
626            }
627    
628            FixedFlow fixedFlow = new FixedFlow_impl();
629            fixedFlow.setFixedFlow(flowNames.toArray(new String[flowNames.size()]));
630            desc.getAnalysisEngineMetaData().setFlowConstraints(fixedFlow);
631    
632            if (typePriorities != null) {
633                desc.getAnalysisEngineMetaData().setTypePriorities(typePriorities);
634            }
635    
636            if (sofaMappings != null) {
637                desc.setSofaMappings(sofaMappings);
638            }
639    
640            return desc;
641        }

644        /**
645         * Creates an AnalysisEngine from the given descriptor, and uses the engine to process the file or text.
646         *
647         * @param descriptorFileName The fully qualified, Java-style, dotted name of the XML descriptor file.
648         * @param fileNameOrText Either the path of a file to be loaded, or a string to use as the text. If the string given
649         *            is not a valid path in the file system, it will be assumed to be text.
650         * @return A JCas object containing the processed document.
651         * @throws IOException
652         * @throws UIMAException
653         */
654        public static JCas process(String descriptorFileName, String fileNameOrText) throws IOException, UIMAException
655        {
656            AnalysisEngine engine = createAnalysisEngine(descriptorFileName);
657            JCas jCas = process(engine, fileNameOrText);
658            engine.collectionProcessComplete();
659            return jCas;
660        }

665         * @param analysisEngine The AnalysisEngine object to process the text.
666         * @param fileNameOrText Either the path of a file to be loaded, or a string to use as the text. If the string given
667         *            is not a valid path in the file system, it will be assumed to be text.
668         * @return A JCas object containing the processed document.
669         * @throws IOException
670         * @throws UIMAException
671         */
672        public static JCas process(AnalysisEngine analysisEngine, String fileNameOrText) throws IOException, UIMAException
673        {
674            JCas jCas = analysisEngine.newJCas();
675            process(jCas, analysisEngine, fileNameOrText);
676            return jCas;
677        }
678    
679        /**
680         * Provides a convenience method for running  an AnalysisEngine over some text with a given JCas.
681         * @param jCas
682         * @param analysisEngine
683         * @param fileNameOrText
684         * @throws IOException
685         * @throws UIMAException
686         */
687        public static void process(JCas jCas, AnalysisEngine analysisEngine, String fileNameOrText) throws IOException, UIMAException {
688            File textFile = new File(fileNameOrText);
689            String text;
690            if (textFile.exists()) {
691                text = FileUtils.file2String(textFile);
692            } else {
693                text = fileNameOrText;
694            }
695    
696            jCas.setDocumentText(text);
697            analysisEngine.process(jCas);
698         }
