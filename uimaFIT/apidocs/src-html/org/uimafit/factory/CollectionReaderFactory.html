052            /**
053             * Create a CollectionReader from an XML descriptor file and a set of
054             * configuration parameters.
055             * 
056             * @param descriptorPath
057             *            The path to the XML descriptor file.
058             * @param configurationData
059             *            Any additional configuration parameters to be set. These
060             *            should be supplied as (name, value) pairs, so there should
061             *            always be an even number of parameters.
062             * @return The CollectionReader created from the XML descriptor and the
063             *         configuration parameters.
064             * @throws UIMAException
065             * @throws IOException
066             */
067            public static CollectionReader createCollectionReaderFromPath(String descriptorPath, Object... configurationData)
068                            throws UIMAException, IOException {
069                    ResourceCreationSpecifier specifier = ResourceCreationSpecifierFactory.createResourceCreationSpecifier(
070                                    descriptorPath, configurationData);
071                    return UIMAFramework.produceCollectionReader(specifier);
072            }
073    
074            /**
075             * Get a CollectionReader from the name (Java-style, dotted) of an XML
076             * descriptor file, and a set of configuration parameters.
077             * 
078             * @param descriptorName
079             *            The fully qualified, Java-style, dotted name of the XML
080             *            descriptor file.
081             * @param configurationData
082             *            Any additional configuration parameters to be set. These
083             *            should be supplied as (name, value) pairs, so there should
084             *            always be an even number of parameters.
085             * @return The AnalysisEngine created from the XML descriptor and the
086             *         configuration parameters.
087             * @throws UIMAException
088             * @throws IOException
089             */
090    
091            public static CollectionReader createCollectionReader(String descriptorName, Object... configurationData)
092                            throws UIMAException, IOException {
093                    Import_impl imp = new Import_impl();
094                    imp.setName(descriptorName);
095                    URL url = imp.findAbsoluteUrl(UIMAFramework.newDefaultResourceManager());
096                    ResourceSpecifier specifier = ResourceCreationSpecifierFactory.createResourceCreationSpecifier(url, configurationData);
097                    return UIMAFramework.produceCollectionReader(specifier);
098            }

137            @SuppressWarnings("unchecked")
138            public static CollectionReader createCollectionReader(CollectionReaderDescription desc, Object... configurationData) throws ResourceInitializationException {
139                    // create the CollectionReader
140                    CollectionReader reader;
141                    try {
142                            reader = (CollectionReader)  (Class.forName(desc.getImplementationName()).newInstance());
143                    }
144                    catch (Exception e) {
145                            throw new ResourceInitializationException(e);
146                    }
147                    
148                    if (configurationData != null) {
149                            ConfigurationData cdata = ConfigurationParameterFactory.createConfigurationData(configurationData);
150                            ConfigurationParameter[] configurationParameters = cdata.configurationParameters;
151                            Object[] configurationValues = cdata.configurationValues;
152                            ResourceCreationSpecifierFactory.setConfigurationParameters(desc, configurationParameters, configurationValues);
153                            ResourceMetaData metaData = desc.getMetaData();
154                            ConfigurationParameterSettings paramSettings = metaData.getConfigurationParameterSettings();
155                            Map additionalParameters = new HashMap();
156                            additionalParameters.put(AnalysisEngine.PARAM_CONFIG_PARAM_SETTINGS, paramSettings);
157                            reader.initialize(desc, additionalParameters);
158                    }
159                    else {
160                            reader.initialize(desc, null);
161                    }
162                    return reader;
163                    
164                    
165                    
166    
167            }

197            public static CollectionReaderDescription createDescription(Class<? extends CollectionReader> readerClass,
198                            TypeSystemDescription typeSystem, TypePriorities typePriorities, Capability[] capabilities, ConfigurationParameter[] configurationParameters, Object[] configurationValues) throws ResourceInitializationException {
199                    // create the descriptor and set configuration parameters
200                    CollectionReaderDescription desc = new CollectionReaderDescription_impl();
201                    desc.setFrameworkImplementation(Constants.JAVA_FRAMEWORK_NAME);
202                    desc.setImplementationName(readerClass.getName());
203                    
204                    // Extract external resource dependencies
205                    Collection<ExternalResourceDependency> deps = ExternalResourceInitializer.getResourceDeclarations(
206                                    readerClass).values();
207                    desc.setExternalResourceDependencies(deps.toArray(new ExternalResourceDependency[deps.size()]));
208                    
209                    ConfigurationData reflectedConfigurationData = ConfigurationParameterFactory
210                                    .createConfigurationData(readerClass);
211                    ResourceCreationSpecifierFactory.setConfigurationParameters(desc,
212                                    reflectedConfigurationData.configurationParameters, reflectedConfigurationData.configurationValues);
213                    if (configurationParameters != null) {
214                            ResourceCreationSpecifierFactory.setConfigurationParameters(desc,
215                                            configurationParameters, configurationValues);
216                    }
217    
218                    // set the type system
219                    if (typeSystem != null) {
220                            desc.getCollectionReaderMetaData().setTypeSystem(typeSystem);
221                    }
222                    
223                    if(typePriorities != null)
224                            desc.getCollectionReaderMetaData().setTypePriorities(typePriorities);
225    
226                    if(capabilities == null) {
227                            Capability capability = CapabilityFactory.createCapability(readerClass);
228                            if(capability != null) {
229                                    capabilities = new Capability[] {capability};
230                            }
231                    }
232                    if(capabilities != null)
233                            desc.getCollectionReaderMetaData().setCapabilities(capabilities);
234                    
235                    return desc;
236            }
