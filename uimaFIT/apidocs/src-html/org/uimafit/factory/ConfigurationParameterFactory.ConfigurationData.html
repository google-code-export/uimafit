045            /**
046             * A mapping from Java class names to UIMA configuration parameter type names. Used by
047             * setConfigurationParameters().
048             */
049            public static final Map<String, String> javaUimaTypeMap = new HashMap<String, String>();
050            static {
051                    javaUimaTypeMap.put(Boolean.class.getName(), ConfigurationParameter.TYPE_BOOLEAN);
052                    javaUimaTypeMap.put(Float.class.getName(), ConfigurationParameter.TYPE_FLOAT);
053                    javaUimaTypeMap.put(Double.class.getName(), ConfigurationParameter.TYPE_FLOAT);
054                    javaUimaTypeMap.put(Integer.class.getName(), ConfigurationParameter.TYPE_INTEGER);
055                    javaUimaTypeMap.put(String.class.getName(), ConfigurationParameter.TYPE_STRING);
056                    javaUimaTypeMap.put("boolean", ConfigurationParameter.TYPE_BOOLEAN);
057                    javaUimaTypeMap.put("float", ConfigurationParameter.TYPE_FLOAT);
058                    javaUimaTypeMap.put("double", ConfigurationParameter.TYPE_FLOAT);
059                    javaUimaTypeMap.put("int", ConfigurationParameter.TYPE_INTEGER);

063            /**
064             * This method determines if the field is annotated with {@link org.uimafit.descriptor.ConfigurationParameter}.
065             * @param field
066             * @return
067             */
068            public static boolean isConfigurationParameterField(Field field)
069            {
070                    return field.isAnnotationPresent(org.uimafit.descriptor.ConfigurationParameter.class);
071            }

073            /**
074             * Determines the default value of an annotated configuration parameter. The returned value is
075             * not necessarily the value that the annotated member variable will be instantiated with in
076             * ConfigurationParameterInitializer which does extra work to convert the UIMA configuration
077             * parameter value to comply with the type of the member variable.
078             * 
079             * @param field
080             * @return
081             */
082            public static Object getDefaultValue(Field field)
083            {
084                    if (isConfigurationParameterField(field)) {
085                            org.uimafit.descriptor.ConfigurationParameter annotation = field
086                                            .getAnnotation(org.uimafit.descriptor.ConfigurationParameter.class);
087    
088                            String[] stringValue = annotation.defaultValue();
089                            if (stringValue.length == 1
090                                            && stringValue[0]
091                                                            .equals(org.uimafit.descriptor.ConfigurationParameter.NO_DEFAULT_VALUE)) {
092                                    return null;
093                            }
094    
095                            String valueType = getConfigurationParameterType(field);
096                            boolean isMultiValued = isMultiValued(field);
097    
098                            if (!isMultiValued) {
099                                    if (ConfigurationParameter.TYPE_BOOLEAN.equals(valueType)) {
100                                            return Boolean.parseBoolean(stringValue[0]);
101                                    }
102                                    else if (ConfigurationParameter.TYPE_FLOAT.equals(valueType)) {
103                                            return Float.parseFloat(stringValue[0]);
104                                    }
105                                    else if (ConfigurationParameter.TYPE_INTEGER.equals(valueType)) {
106                                            return Integer.parseInt(stringValue[0]);
107                                    }
108                                    else if (ConfigurationParameter.TYPE_STRING.equals(valueType)) {
109                                            return stringValue[0];
110                                    }
111                                    throw new UIMA_IllegalArgumentException(
112                                                    UIMA_IllegalArgumentException.METADATA_ATTRIBUTE_TYPE_MISMATCH,
113                                                    new Object[] { valueType, "type" });
114                            }
115                            else {
116                                    if (ConfigurationParameter.TYPE_BOOLEAN.equals(valueType)) {
117                                            Boolean[] returnValues = new Boolean[stringValue.length];
118                                            for (int i = 0; i < stringValue.length; i++) {
119                                                    returnValues[i] = Boolean.parseBoolean(stringValue[i]);
120                                            }
121                                            return returnValues;
122                                    }
123                                    else if (ConfigurationParameter.TYPE_FLOAT.equals(valueType)) {
124                                            Float[] returnValues = new Float[stringValue.length];
125                                            for (int i = 0; i < stringValue.length; i++) {
126                                                    returnValues[i] = Float.parseFloat(stringValue[i]);
127                                            }
128                                            return returnValues;
129                                    }
130                                    else if (ConfigurationParameter.TYPE_INTEGER.equals(valueType)) {
131                                            Integer[] returnValues = new Integer[stringValue.length];
132                                            for (int i = 0; i < stringValue.length; i++) {
133                                                    returnValues[i] = Integer.parseInt(stringValue[i]);
134                                            }
135                                            return returnValues;
136                                    }
137                                    else if (ConfigurationParameter.TYPE_STRING.equals(valueType)) {
138                                            return stringValue;
139                                    }
140                                    throw new UIMA_IllegalArgumentException(
141                                                    UIMA_IllegalArgumentException.METADATA_ATTRIBUTE_TYPE_MISMATCH,
142                                                    new Object[] { valueType, "type" });
143    
144                            }
145    
146                    }
147                    else {
148                            throw new IllegalArgumentException("field is not annotated with annotation of type "
149                                            + org.uimafit.descriptor.ConfigurationParameter.class.getName());
150                    }
151            }

153            private static String getConfigurationParameterType(Field field)
154            {
155                    Class<?> parameterClass = field.getType();
156                    String parameterClassName;
157                    if (parameterClass.isArray()) {
158                            parameterClassName = parameterClass.getComponentType().getName();
159                    }
160                    else if (Collection.class.isAssignableFrom(parameterClass)) {
161                            ParameterizedType collectionType = (ParameterizedType) field.getGenericType();
162                            parameterClassName = ((Class<?>) (collectionType.getActualTypeArguments()[0]))
163                                            .getName();
164                    }
165                    else {
166                            parameterClassName = parameterClass.getName();
167                    }
168                    String parameterType = javaUimaTypeMap.get(parameterClassName);
169                    if (parameterType == null) {
170                            return ConfigurationParameter.TYPE_STRING;
171                    }
172                    return parameterType;
173            }
174    
175            private static boolean isMultiValued(Field field)
176            {
177                    Class<?> parameterClass = field.getType();
178                    if (parameterClass.isArray()) {
179                            return true;
180                    }
181                    else if (Collection.class.isAssignableFrom(parameterClass)) {
182                            return true;
183                    }
184                    return false;
185            }

187            /**
188             * This method generates the default name of a configuration parameter that is defined by an
189             * {@link org.uimafit.descriptor.ConfigurationParameter} annotation when no name is given
190             * 
191             * @param field
192             * @return
193             */
194            public static String getConfigurationParameterName(Field field)
195            {
196                    if (isConfigurationParameterField(field)) {
197                            org.uimafit.descriptor.ConfigurationParameter annotation = field
198                                            .getAnnotation(org.uimafit.descriptor.ConfigurationParameter.class);
199                            String name = annotation.name();
200                            if (name.equals(org.uimafit.descriptor.ConfigurationParameter.USE_FIELD_NAME)) {
201                                    name = field.getDeclaringClass().getName() + "." + field.getName();
202                            }
203                            return name;
204                    }
205                    return null;
206            }

208            /**
209             * This method provides a convenient way to generate a configuration parameter name for a member
210             * variable that is annotated with {@link org.uimafit.descriptor.ConfigurationParameter} and no
211             * name is provided in the annotation.
212             * 
213             * @param clazz
214             * @param fieldName
215             * @return
216             * @throws RuntimeException
217             */
218            public static String createConfigurationParameterName(Class<?> clazz, String fieldName)
219                    throws RuntimeException
220            {
221                    try {
222                            return ConfigurationParameterFactory.getConfigurationParameterName(clazz
223                                            .getDeclaredField(fieldName));
224                    }
225                    catch (Exception e) {
226                            throw new RuntimeException(e);
227                    }
228            }

230            /**
231             * A factory method for creating a ConfigurationParameter from a given field definition
232             * 
233             * @param field
234             * @return
235             */
236            public static ConfigurationParameter createPrimitiveParameter(Field field)
237            {
238                    if (isConfigurationParameterField(field)) {
239                            org.uimafit.descriptor.ConfigurationParameter annotation = field
240                                            .getAnnotation(org.uimafit.descriptor.ConfigurationParameter.class);
241                            String name = getConfigurationParameterName(field);
242                            boolean multiValued = isMultiValued(field);
243                            String parameterType = getConfigurationParameterType(field);
244                            return createPrimitiveParameter(name, parameterType, annotation.description(),
245                                            multiValued, annotation.mandatory());
246                    }
247                    else {
248                            throw new IllegalArgumentException("field is not annotated with annotation of type "
249                                            + org.uimafit.descriptor.ConfigurationParameter.class.getName());
250                    }
251            }

257            /**
258             * A factory method for creating a ConfigurationParameter object.
259             * 
260             * @param name
261             * @param parameterClass
262             * @param parameterDescription
263             * @param isMandatory
264             * @return
265             */
266            public static ConfigurationParameter createPrimitiveParameter(String name,
267                            Class<?> parameterClass, String parameterDescription, boolean isMandatory)
268            {
269                    String parameterClassName;
270                    if (parameterClass.isArray()) {
271                            parameterClassName = parameterClass.getComponentType().getName();
272                    }
273                    else {
274                            parameterClassName = parameterClass.getName();
275                    }
276    
277                    String parameterType = javaUimaTypeMap.get(parameterClassName);
278                    if (parameterType == null) {
279                            throw new UIMA_IllegalArgumentException(
280                                            UIMA_IllegalArgumentException.METADATA_ATTRIBUTE_TYPE_MISMATCH, new Object[] {
281                                                            parameterClassName, "type" });
282                    }
283                    return createPrimitiveParameter(name, parameterType, parameterDescription,
284                                    parameterClass.isArray(), isMandatory);
285            }

287            /**
288             * A factory method for creating a ConfigurationParameter object.
289             * 
290             * @param name
291             * @param parameterType
292             * @param parameterDescription
293             * @param isMultiValued
294             * @param isMandatory
295             * @return
296             */
297            public static ConfigurationParameter createPrimitiveParameter(String name,
298                            String parameterType, String parameterDescription, boolean isMultiValued,
299                            boolean isMandatory)
300            {
301                    ConfigurationParameter param = new ConfigurationParameter_impl();
302                    param.setName(name);
303                    param.setType(parameterType);
304                    param.setDescription(parameterDescription);
305                    param.setMultiValued(isMultiValued);
306                    param.setMandatory(isMandatory);
307                    return param;
308            }

310            /**
311             * This method converts configuration data provided as an array of objects and returns a
312             * {@link ConfigurationData} object
313             * 
314             * @param configurationData
315             * @return
316             */
317            public static ConfigurationData createConfigurationData(Object... configurationData)
318            {
319                    if (configurationData == null) {
320                            return new ConfigurationData(new ConfigurationParameter[0], new Object[0]);
321                    }
322                    if (configurationData.length % 2 != 0) {
323                            String message = "a value must be specified for each parameter name: an odd number of values passed in ("
324                                            + configurationData.length + ")";
325                            throw new IllegalArgumentException(message);
326                    }
327    
328                    int numberOfParameters = configurationData.length / 2;
329                    List<ConfigurationParameter> configurationParameters = new ArrayList<ConfigurationParameter>();
330                    List<Object> configurationValues = new ArrayList<Object>();
331    
332                    for (int i = 0; i < numberOfParameters; i++) {
333                            String name = (String) configurationData[i * 2];
334                            Object value = configurationData[i * 2 + 1];
335    
336                            if (value == null) {
337                                    continue;
338                            }
339    
340                            if (value.getClass().isArray()
341                                            && value.getClass().getComponentType().getName().equals("boolean")) {
342                                    value = ArrayUtils.toObject((boolean[]) value);
343                            }
344                            else if (value.getClass().isArray()
345                                            && value.getClass().getComponentType().getName().equals("int")) {
346                                    value = ArrayUtils.toObject((int[]) value);
347                            }
348                            else if (value.getClass().isArray()
349                                            && value.getClass().getComponentType().getName().equals("float")) {
350                                    value = ArrayUtils.toObject((float[]) value);
351                            }
352    
353                            ConfigurationParameter param = ConfigurationParameterFactory.createPrimitiveParameter(
354                                            name, value.getClass(), null, false);
355                            configurationParameters.add(param);
356                            configurationValues.add(value);
357                    }
358                    return new ConfigurationData(
359                                    configurationParameters.toArray(new ConfigurationParameter[configurationParameters
360                                                    .size()]), configurationValues.toArray());
361            }

363            /**
364             * This method creates configuration data for a given class definition using reflection and the
365             * configuration parameter annotation
366             * 
367             * @param componentClass
368             * @return
369             */
370            public static ConfigurationData createConfigurationData(Class<?> componentClass)
371            {
372                    List<ConfigurationParameter> configurationParameters = new ArrayList<ConfigurationParameter>();
373                    List<Object> configurationValues = new ArrayList<Object>();
374    
375                    for (Field field : ReflectionUtil.getFields(componentClass)) {
376                            if (ConfigurationParameterFactory.isConfigurationParameterField(field)) {
377                                    configurationParameters.add(ConfigurationParameterFactory
378                                                    .createPrimitiveParameter(field));
379                                    configurationValues.add(ConfigurationParameterFactory.getDefaultValue(field));
380                            }
381                    }
382    
383                    return new ConfigurationData(
384                                    configurationParameters.toArray(new ConfigurationParameter[configurationParameters
385                                                    .size()]), configurationValues.toArray(new Object[configurationValues
386                                                    .size()]));
387            }

389            /**
390             * A simple class for storing an array of configuration parameters along with an array of the
391             * values that will fill in those configuration parameters
392             * 
393             */
394            public static class ConfigurationData
395            {
396                    /**
397                     * 
398                     */
399                    public ConfigurationParameter[] configurationParameters;
400    
401                    /**
402                     * 
403                     */
404                    public Object[] configurationValues;
405    
406                    /**
407                     * @param configurationParameters
408                     * @param configurationValues
409                     */
410                    public ConfigurationData(ConfigurationParameter[] configurationParameters,
411                                    Object[] configurationValues)
412                    {
413                            this.configurationParameters = configurationParameters;
414                            this.configurationValues = configurationValues;
415                    }

419            /**
420             * This method adds configuration parameter information to the specifier given the provided
421             * configuration data
422             * 
423             * @param specifier
424             * @param configurationData
425             *            should consist of name value pairs.
426             */
427            public static void addConfigurationParameters(ResourceCreationSpecifier specifier,
428                            Object... configurationData)
429            {
430                    ConfigurationData cdata = ConfigurationParameterFactory
431                                    .createConfigurationData(configurationData);
432                    ResourceCreationSpecifierFactory.setConfigurationParameters(specifier,
433                                    cdata.configurationParameters, cdata.configurationValues);
434            }

436            /**
437             * Provides a mechanism to add configuration parameter information to a specifier for the given
438             * classes. this method may be useful in situations where a class definition has annotated
439             * configuration parameters that you want to include in the given specifier
440             * 
441             * @param specifier
442             * @param dynamicallyLoadedClasses
443             */
444            public static void addConfigurationParameters(ResourceCreationSpecifier specifier,
445                            List<Class<?>> dynamicallyLoadedClasses)
446            {
447                    for (Class<?> dynamicallyLoadedClass : dynamicallyLoadedClasses) {
448                            ConfigurationData reflectedConfigurationData = ConfigurationParameterFactory
449                                            .createConfigurationData(dynamicallyLoadedClass);
450                            ResourceCreationSpecifierFactory.setConfigurationParameters(specifier,
451                                            reflectedConfigurationData.configurationParameters,
452                                            reflectedConfigurationData.configurationValues);
453                    }
454    
455            }
