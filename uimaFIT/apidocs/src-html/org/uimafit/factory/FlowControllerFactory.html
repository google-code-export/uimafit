018    package org.uimafit.factory;
019    
020    import org.apache.uima.Constants;
021    import org.apache.uima.flow.FlowController;
022    import org.apache.uima.flow.FlowControllerDescription;
023    import org.apache.uima.flow.impl.FlowControllerDescription_impl;
024    import org.apache.uima.resource.ResourceInitializationException;
025    import org.apache.uima.resource.metadata.ConfigurationParameter;
026    import org.apache.uima.resource.metadata.ResourceMetaData;
027    import org.uimafit.factory.ConfigurationParameterFactory.ConfigurationData;
028    
029    /**
030     * @author Fabio Mancinelli, Philip Ogren
031     */
032    
033    public class FlowControllerFactory {
034    
035            public static FlowControllerDescription createFlowControllerDescription(Class<? extends FlowController> flowControllerClass,
036                            Object... configurationData) throws ResourceInitializationException {
037                    ConfigurationData cdata = ConfigurationParameterFactory.createConfigurationData(configurationData);
038                    return createFlowControllerDescription(flowControllerClass, cdata.configurationParameters, cdata.configurationValues);
039            }
040    
041            public static FlowControllerDescription createFlowControllerDescription(Class<? extends FlowController> flowControllerClass,
042                            ConfigurationParameter[] configurationParameters, Object[] configurationValues) throws ResourceInitializationException {
043    
044                    FlowControllerDescription desc = new FlowControllerDescription_impl();
045                    desc.setFrameworkImplementation(Constants.JAVA_FRAMEWORK_NAME);
046                    desc.setImplementationName(flowControllerClass.getName());
047    
048            ConfigurationData reflectedConfigurationData =
049                ConfigurationParameterFactory.createConfigurationData(flowControllerClass);
050            ResourceCreationSpecifierFactory.setConfigurationParameters(desc,
051                reflectedConfigurationData.configurationParameters, reflectedConfigurationData.configurationValues);
052            if (configurationParameters != null) {
053                ResourceCreationSpecifierFactory.setConfigurationParameters(desc, configurationParameters,
054                    configurationValues);
055            }
056    
057                    
058                    
059                    ResourceMetaData meta = desc.getMetaData();
060                    meta.setName(flowControllerClass.getName());
061                    meta.setVendor(flowControllerClass.getPackage().toString());
062                    meta.setDescription("Descriptor automatically generated by uimaFIT");
063                    meta.setVersion("unknown");
064    
065                    return desc;
066            }
067    
068    }
