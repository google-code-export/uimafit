018    package org.uimafit.factory;
019    
020    import java.util.ArrayList;
021    import java.util.List;
022    
023    import org.apache.uima.UIMAException;
024    import org.apache.uima.cas.Feature;
025    import org.apache.uima.jcas.JCas;
026    import org.apache.uima.jcas.tcas.Annotation;
027    
028    /**
029     * This class was written as a simple way to populate a JCas with tokens along
030     * with part-of-speech and stem information and assumes you already have this
031     * information at your disposal. This should not be confused with an actual
032     * tokenizer (or part-of-speech tagger) which tokenizes and tags plain text.  
033     * 
034     * @author Steven Bethard, Philip Ogren
035     */
036    public class TokenFactory {
037    
038            public static <TOKEN_TYPE extends Annotation, SENTENCE_TYPE extends Annotation> void createTokens(JCas jCas,
039                            String text, Class<TOKEN_TYPE> tokenClass, Class<SENTENCE_TYPE> sentenceClass) throws UIMAException {
040                    createTokens(jCas, text, tokenClass, sentenceClass, null, null, null, null, null);
041            }
042    
043            public static <TOKEN_TYPE extends Annotation, SENTENCE_TYPE extends Annotation> void createTokens(JCas jCas,
044                            String text, Class<TOKEN_TYPE> tokenClass, Class<SENTENCE_TYPE> sentenceClass, String tokensString)
045                            throws UIMAException {
046                    createTokens(jCas, text, tokenClass, sentenceClass, tokensString, null, null, null, null);
047            }
048    
049            /**
050             * Add Token and Sentence annotations to the JCas.
051             * 
052             * @param jCas
053             *            The JCas where the annotation should be added.
054             * @param text
055             *            The text of the document.
056             * @param tokensString
057             *            The text of the document with spaces separating words and
058             *            newlines separating sentences.
059             * @param posTagsString
060             *            The part of speech tags for each word, separated by spaces.
061             * @param stemsString
062             *            The stems for each word, separated by spaces.
063             * @throws UIMAException
064             */
065            public static <TOKEN_TYPE extends Annotation, SENTENCE_TYPE extends Annotation> void createTokens(JCas jCas,
066                            String text, Class<TOKEN_TYPE> tokenClass, Class<SENTENCE_TYPE> sentenceClass, String tokensString,
067                            String posTagsString, String stemsString, String posFeatureName, String stemFeatureName)
068                            throws UIMAException {
069                    // set the document text and add Token annotations as indicated
070                    jCas.setDocumentText(text);
071                    int offset = 0;
072                    int tokenIndex = 0;
073    
074                    Feature posFeature = null;
075                    if (posFeatureName != null) posFeature = jCas.getTypeSystem().getFeatureByFullName(posFeatureName);
076                    Feature stemFeature = null;
077                    if (stemFeatureName != null) stemFeature = jCas.getTypeSystem().getFeatureByFullName(stemFeatureName);
078    
079                    String[] sentenceStrings = (tokensString != null ? tokensString : text).split("\n");
080                    String[] posTags = posTagsString != null ? posTagsString.split(" ") : null;
081                    String[] stems = stemsString != null ? stemsString.split(" ") : null;
082                    for (String sentenceString : sentenceStrings) {
083                            String[] tokenStrings = sentenceString.trim().split(" ");
084                            List<Annotation> tokenAnnotations = new ArrayList<Annotation>();
085                            for (String tokenString : tokenStrings) {
086                                    // move the offset up to the beginning of the token
087                                    while (!text.startsWith(tokenString, offset)) {
088                                            offset++;
089                                            if (offset > text.length()) {
090                                                    throw new IllegalArgumentException(String.format("unable to find string %s", tokenString));
091                                            }
092                                    }
093    
094                                    // add the Token
095                                    int start = offset;
096                                    offset = offset + tokenString.length();
097                                    Annotation token = AnnotationFactory.createAnnotation(jCas, start, offset, tokenClass);
098                                    tokenAnnotations.add(token);
099    
100                                    // set the stem and part of speech if present
101                                    if (posFeatureName != null) {
102                                            token.setStringValue(posFeature, posTags[tokenIndex]);
103                                    }
104                                    if (stemFeatureName != null) {
105                                            token.setStringValue(stemFeature, stems[tokenIndex]);
106                                    }
107                                    tokenIndex++;
108                            }
109                            if (tokenAnnotations.size() > 0) {
110                                    int begin = tokenAnnotations.get(0).getBegin();
111                                    int end = tokenAnnotations.get(tokenAnnotations.size() - 1).getEnd();
112                                    AnnotationFactory.createAnnotation(jCas, begin, end, sentenceClass);
113                            }
114                    }
115            }
116    }
