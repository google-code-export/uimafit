045            public static FileFilter createPatternFilter(final String[] patternStrings) {
046                    int length = patternStrings == null ? 0 : patternStrings.length;
047                    final Pattern[] patterns = new Pattern[length];
048                    for (int i = 0; i < length; ++i) {
049                            patterns[i] = Pattern.compile(patternStrings[i]);
050                    }
051                    return new FileFilter() {
052                            public boolean accept(File file) {
053                                    if (patterns.length == 0) {
054                                            return true;
055                                    }
056                                    else {
057                                            String path = file.getName();
058                                            for (Pattern pattern : patterns) {
059                                                    if (pattern.matcher(path).find()) {
060                                                            return true;
061                                                    }
062                                            }
063                                            return false;
064                                    }
065                            }
066                    };
067            }
068    
069            public static FileFilter createSuffixFilter(final String[] suffixes, final String[] exclusionSuffixes) {
070                    return new FileFilter() {
071    
072                            public boolean accept(File file) {
073                                    if (suffixes == null) return true;
074                                    if (suffixes.length == 0) return true;
075                                    else {
076                                            if (exclusionSuffixes != null) {
077                                                    for (String exclusionSuffix : exclusionSuffixes) {
078                                                            if (file.getName().endsWith(exclusionSuffix)) return false;
079                                                    }
080                                            }
081                                            if (suffixes != null) {
082                                                    for (String suffix : suffixes) {
083                                                            if (file.getName().endsWith(suffix)) return true;
084                                                    }
085                                            }
086                                    }
087                                    return false;
088                            }
089                    };
090            }

092            public static Iterable<File> getFiles(final File folder, final Set<String> fileNames) {
093                    return new Iterable<File>() {
094    
095                            public Iterator<File> iterator() {
096                                    return new Iterator<File>() {
097                                            private LinkedList<File> queue = new LinkedList<File>();
098                                            {
099                                                    queue.offer(folder);
100                                                    processDirectories();
101                                            }
102    
103                                            public boolean hasNext() {
104                                                    this.processDirectories();
105                                                    return !queue.isEmpty();
106                                            }
107    
108                                            public File next() {
109                                                    this.processDirectories();
110                                                    if (queue.isEmpty()) throw new NoSuchElementException();
111                                                    return queue.poll();
112                                            }
113    
114                                            private void processDirectories() {
115                                                    while (!queue.isEmpty()) {
116                                                            if (!queue.peek().isDirectory()) break;
117                                                            File next = queue.poll();
118                                                            for (File file : next.listFiles()) {
119                                                                    String fileName = file.getName();
120                                                                    if (file.getName().startsWith(".")) continue;
121                                                                    if (file.isDirectory() || fileNames.contains(fileName)) {
122                                                                            queue.offer(file);
123                                                                    }
124                                                            }
125                                                    }
126                                            }
127    
128                                            public void remove() {
129                                                    throw new UnsupportedOperationException();
130                                            }
131    
132                                    };
133                            }
134                    };
135    
136            }
137    
138            public static Iterable<File> getFiles(final File folder, final FileFilter filter) {
139                    return new Iterable<File>() {
140    
141                            public Iterator<File> iterator() {
142                                    return new Iterator<File>() {
143                                            private LinkedList<File> queue = new LinkedList<File>();
144                                            {
145                                                    queue.offer(folder);
146                                                    processDirectories();
147                                            }
148    
149                                            public boolean hasNext() {
150                                                    this.processDirectories();
151                                                    return !queue.isEmpty();
152                                            }
153    
154                                            public File next() {
155                                                    this.processDirectories();
156                                                    if (queue.isEmpty()) throw new NoSuchElementException();
157                                                    return queue.poll();
158                                            }
159    
160                                            private void processDirectories() {
161                                                    while (!queue.isEmpty()) {
162                                                            if (!queue.peek().isDirectory()) break;
163                                                            File next = queue.poll();
164                                                            for (File each : next.listFiles()) {
165                                                                    if (each.getName().startsWith(".")) continue;
166                                                                    if (each.isDirectory() || filter == null || filter.accept(each)) {
167                                                                            queue.offer(each);
168                                                                    }
169                                                            }
170                                                    }
171                                            }
172    
173                                            public void remove() {
174                                                    throw new UnsupportedOperationException();
175                                            }
176    
177                                    };
178                            }
179                    };
180            }

183                    return getFiles(new File(filename));

199                    return getFiles(new File(filename), suffixes, exclusionSuffixes);

202            public static Iterable<File> getFiles(File file, String[] suffixes) {
203                    return getFiles(file, suffixes, null);
204            }
205    
206            public static Iterable<File> getFiles(File file, String[] suffixes, String[] exclusionSuffixes) {
207                    return getFiles(file, createSuffixFilter(suffixes, exclusionSuffixes));
208            }
