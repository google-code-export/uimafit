148                    String typeName = aTypename;
149                    if (typeName.startsWith(UIMA_BUILTIN_JCAS_PREFIX)) {
150                            typeName = "uima." + typeName.substring(UIMA_BUILTIN_JCAS_PREFIX.length());
151                    }
152                    final Type type = aCas.getTypeSystem().getType(typeName);
153                    if (type == null) {
154                            throw new IllegalArgumentException("Undeclared type [" + aTypename + "]");
155                    }
156                    return type;

303            /**
304             * Get a list of annotations of the given annotation type located between two annotations.
305             * Does not use subiterators and does not respect type priorities. Zero-width annotations
306             * what lie on the borders are included in the result, e.g. if the boundary annotations are
307             * [1..2] and [2..3] then an annotation [2..2] is returned. If there is a non-zero overlap
308             * between the boundary annotations, the result is empty. The method properly handles cases
309             * where the second boundary annotations occurs before the first boundary annotation by
310             * switching their roles.
311             *
312             * @param cas
313             *            a CAS.
314             * @param type
315             *            a UIMA type.
316             * @param ann1
317             *            the first boundary annotation.
318             * @param ann2
319             *            the second boundary annotation.
320             * @return a return value.
321             * @see Subiterator
322             */
323            public static List<AnnotationFS> selectBetween(final CAS cas, final Type type, 
324                            final AnnotationFS ann1, final AnnotationFS ann2) {
325                    AnnotationFS left;
326                    AnnotationFS right;
327                    if (ann1.getEnd() > ann2.getBegin()) {
328                            left = ann2;
329                            right = ann1;
330                    }
331                    else {
332                            left = ann1;
333                            right = ann2;
334                    }
335                    
336                    int begin = left.getEnd();
337                    int end = right.getBegin();
338    
339                    List<AnnotationFS> list = new ArrayList<AnnotationFS>();
340                    FSIterator<AnnotationFS> it = cas.getAnnotationIndex(type).iterator();
341    
342                    // Try to seek the insertion point.
343                    it.moveTo(left);
344    
345                    // If the insertion point is beyond the index, move back to the last.
346                    if (!it.isValid()) {
347                            it.moveToLast();
348                            if (!it.isValid()) {
349                                    return list;
350                            }
351                    }
352    
353                    // Ignore type priorities by seeking to the first that has the same begin
354                    boolean moved = false;
355                    while (it.isValid() && (it.get()).getBegin() >= begin) {
356                            it.moveToPrevious();
357                            moved = true;
358                    }
359    
360                    // If we moved, then we are now on one starting before the requested begin, so we have to
361                    // move one ahead.
362                    if (moved) {
363                            it.moveToNext();
364                    }
365    
366                    // If we managed to move outside the index, start at first.
367                    if (!it.isValid()) {
368                            it.moveToFirst();
369                    }
370    
371                    // Skip annotations whose start is before the start parameter.
372                    while (it.isValid() && (it.get()).getBegin() < begin) {
373                            it.moveToNext();
374                    }
375    
376                    boolean strict = true;
377                    while (it.isValid()) {
378                            AnnotationFS a = it.get();
379                            // If the start of the current annotation is past the end parameter, we're done.
380                            if (a.getBegin() > end) {
381                                    break;
382                            }
383                            it.moveToNext();
384                            if (strict && a.getEnd() > end) {
385                                    continue;
386                            }
387    
388                            assert (a.getBegin() >= left.getEnd()) : "Illegal begin "
389                                            + a.getBegin() + " in [" + begin + ".." + end + "]";
390    
391                            assert (a.getEnd() <= right.getBegin()) : "Illegal end "
392                                            + a.getBegin() + " in [" + begin + ".." + end + "]";
393    
394    
395                            if (!a.equals(left) && !a.equals(right)) {
396                                    list.add(a);
397                            }
398                    }
399    
400                    return unmodifiableList(list);  }

504            /**
505             * Get a list of annotations of the given annotation type constraint by a certain annotation.
506             * Iterates over all annotations of the given type to find the covered annotations. Does not use
507             * subiterators and does not respect type prioritites. Was adapted from {@link Subiterator}.
508             * Uses the same approach except that type priorities are ignored.
509             * <p>
510             * <b>Note:</b> this is significantly slower than using
511             * {@link #selectCovered(CAS, Type, AnnotationFS)}. It is possible to use {@code 
512             * selectCovered(cas, type, new Annotation(jCas, int, int))}, but that will allocate memory
513             * in the jCas for the new annotation. If you do that repeatedly many times, memory may fill up. 
514             *
515             * @param cas
516             *            a CAS.
517             * @param type
518             *            a UIMA type.
519             * @param begin
520             *            begin offset.
521             * @param end
522             *            end offset.
523             * @return a return value.
524             * @see Subiterator
525             */
526            public static List<AnnotationFS> selectCovered(CAS cas, Type type, int begin, int end) {
527    
528                    List<AnnotationFS> list = new ArrayList<AnnotationFS>();
529                    FSIterator<AnnotationFS> it = cas.getAnnotationIndex(type).iterator();
530    
531                    // Skip annotations whose start is before the start parameter.
532                    while (it.isValid() && (it.get()).getBegin() < begin) {
533                            it.moveToNext();
534                    }
535    
536                    boolean strict = true;
537                    while (it.isValid()) {
538                            AnnotationFS a = it.get();
539                            // If the start of the current annotation is past the end parameter, we're done.
540                            if (a.getBegin() > end) {
541                                    break;
542                            }
543                            it.moveToNext();
544                            if (strict && a.getEnd() > end) {
545                                    continue;
546                            }
547    
548                            assert (a.getBegin() >= begin) : "Illegal begin " + a.getBegin() + " in [" + begin
549                                            + ".." + end + "]";
550    
551                            assert (a.getEnd() <= end) : "Illegal end " + a.getEnd() + " in [" + begin + ".."
552                                            + end + "]";
553    
554                            list.add(a);
555                    }
556    
557                    return list;
558            }

560            /**
561             * Get a list of annotations of the given annotation type constraint by a certain annotation.
562             * Iterates over all annotations to find the covering annotations.
563             *
564             * <p>
565             * <b>Note:</b> this is <b>REALLY SLOW!</b> You don't want to use this. Instead, consider
566             * using {@link #indexCovering(CAS, Type, Type)} or a {@link ContainmentIndex}.
567             *
568             * @param cas
569             *            a CAS.
570             * @param type
571             *            a UIMA type.
572             * @param begin
573             *            begin offset.
574             * @param end
575             *            end offset.
576             * @return a return value.
577             */
578            public static List<AnnotationFS> selectCovering(CAS cas, Type type, int begin, int end) {
579    
580                    TypeSystem ts = cas.getTypeSystem();
581                    List<AnnotationFS> list = new ArrayList<AnnotationFS>();
582                    FSIterator<AnnotationFS> iter = cas.getAnnotationIndex().iterator();
583                    while (iter.hasNext()) {
584                            AnnotationFS a = iter.next();
585                            if ((a.getBegin() <= begin) && (a.getEnd() >= end)
586                                            && ((type == null) || (ts.subsumes(type, a.getType())))) {
587                                    list.add(a);
588                            }
589                    }
590                    return list;
591            }

593            /**
594             * Create an index for quickly lookup up the annotations covering a particular annotation. This
595             * is preferable to using {@link #selectCovering(CAS, Type, int, int)} because the overhead of
596             * scanning the CAS occurs only when the index is build. Subsequent lookups to the index are
597             * fast.
598             *
599             * @param cas
600             *            a CAS.
601             * @param type
602             *            type to create the index for - this is used in lookups.
603             * @param coveringType
604             *            type of covering annotations.
605             * @return the index.
606             */
607            public static Map<AnnotationFS, Collection<AnnotationFS>> indexCovering(CAS cas, Type type,
608                            Type coveringType) {
609                    Map<AnnotationFS, Collection<AnnotationFS>> index = new HashMap<AnnotationFS, Collection<AnnotationFS>>() {
610                            private static final long serialVersionUID = 1L;
611    
612                            @Override
613                            public Collection<AnnotationFS> get(Object paramObject) {
614                                    Collection<AnnotationFS> res = super.get(paramObject);
615                                    if (res == null) {
616                                            return emptyList();
617                                    }
618                                    else {
619                                            return res;
620                                    }
621                            }
622                    };
623                    for (AnnotationFS s : select(cas, coveringType)) {
624                            for (AnnotationFS u : selectCovered(cas, type, s)) {
625                                    Collection<AnnotationFS> c = index.get(u);
626                                    if (c == EMPTY_LIST) {
627                                            c = new LinkedList<AnnotationFS>();
628                                            index.put(u, c);
629                                    }
630                                    c.add(s);
631                            }
632                    }
633                    return unmodifiableMap(index);
634            }

640             * fast.
641             *
642             * @param cas
643             *            a CAS.
644             * @param type
645             *            type to create the index for - this is used in lookups.
646             * @param coveredType
647             *            type of covering annotations.
648             * @return the index.
649             */
650            public static Map<AnnotationFS, Collection<AnnotationFS>> indexCovered(CAS cas, Type type,
651                            Type coveredType) {

653                            private static final long serialVersionUID = 1L;

660                                    }
661                                    else {
662                                            return res;
663                                    }
664                            }
665                    };
666                    for (AnnotationFS s : select(cas, type)) {
667                            for (AnnotationFS u : selectCovered(cas, coveredType, s)) {
668                                    Collection<AnnotationFS> c = index.get(s);
669                                    if (c == EMPTY_LIST) {

678            /**
679             * This method exists simply as a convenience method for unit testing. It is not very efficient
680             * and should not, in general be used outside the context of unit testing.
681             *
682             * @param cas
683             *            a CAS containing the feature structure.
684             * @param type
685             *            a UIMA type.
686             * @param index
687             *            this can be either positive (0 corresponds to the first annotation of a type) or
688             *            negative (-1 corresponds to the last annotation of a type.)
689             * @return an annotation of the given type
690             */
691            public static FeatureStructure selectFSByIndex(CAS cas, Type type, int index) {
692                    FSIterator<FeatureStructure> i = cas.getIndexRepository().getAllIndexedFS(type);
693                    int n = index;
694                    i.moveToFirst();
695                    if (n > 0) {
696                            while (n > 0 && i.isValid()) {
697                                    i.moveToNext();
698                                    n--;
699                            }
700                    }
701                    if (n < 0) {
702                            i.moveToLast();
703                            while (n < -1 && i.isValid()) {
704                                    i.moveToPrevious();
705                                    n++;
706                            }
707                    }
708    
709                    return i.isValid() ? i.get() : null;
710            }

794            /**
795             * Return an annotation preceding or following of a given reference annotation.
796             * 
797             * @param cas
798             *            a CAS containing the annotation.
799             * @param type
800             *            a type.
801             * @param annotation
802             *            anchor annotation
803             * @param index
804             *            relative position to access. A negative value selects a preceding annotation
805             *            while a positive number selects a following annotation.
806             * @return the addressed annotation.
807             * @throws IndexOutOfBoundsException if the relative index points beyond the type index bounds.
808             */
809            public static AnnotationFS selectSingleRelative(CAS cas, Type type,
810                            AnnotationFS annotation, int index) {
811                    if (!cas.getTypeSystem().subsumes(cas.getAnnotationType(), type)) {
812                            throw new IllegalArgumentException("Type [" + type.getName()
813                                            + "] is not an annotation type");
814                    }
815    
816                    // move to first previous annotation
817                    FSIterator<AnnotationFS> itr = cas.getAnnotationIndex(type).iterator();
818                    itr.moveTo(annotation);
819    
820                    if (index < 0) {
821                            // make sure we're past the beginning of the reference annotation
822                            while (itr.isValid() && itr.get().getEnd() > annotation.getBegin()) {
823                                    itr.moveToPrevious();
824                            }
825                            
826                            for (int i = 0; i < (-index - 1) && itr.isValid(); ++i, itr.moveToPrevious()) {
827                                    // Seeking
828                            }
829                            
830                            if (!itr.isValid()) {
831                                    throw new IndexOutOfBoundsException("Reached end of index while seeking.");
832                            }
833                            else {
834                                    return itr.get();
835                            }
836                    }
837                    else if (index > 0) {
838                            // make sure we're past the end of the reference annotation
839                            while (itr.isValid() && itr.get().getBegin() < annotation.getEnd()) {
840                                    itr.moveToNext();
841                            }
842                            
843                            for (int i = 0; i < (index - 1) && itr.isValid(); ++i, itr.moveToPrevious()) {
844                                    // Seeking
845                            }
846                            
847                            if (!itr.isValid()) {
848                                    throw new IndexOutOfBoundsException("Reached end of index while seeking.");
849                            }
850                            else {
851                                    return itr.get();
852                            }
853                    }
854                    else {
855                            return annotation;
856                    }
857            }

859            /**
860             * Returns the n annotations preceding the given annotation
861             *
862             * @param cas
863             *            a CAS.
864             * @param type
865             *            a UIMA type.
866             * @param annotation
867             *            anchor annotation
868             * @param count
869             *            number of annotations to collect
870             * @return List of aType annotations preceding anchor annotation
871             */
872            public static List<AnnotationFS> selectPreceding(CAS cas, Type type, AnnotationFS annotation,
873                            int count) {
874                    if (!cas.getTypeSystem().subsumes(cas.getAnnotationType(), type)) {
875                            throw new IllegalArgumentException("Type [" + type.getName()
876                                            + "] is not an annotation type");
877                    }
878    
879                    // move to first previous annotation
880                    FSIterator<AnnotationFS> itr = cas.getAnnotationIndex(type).iterator();
881                    itr.moveTo(annotation);
882    
883                    // make sure we're past the beginning of the reference annotation
884                    while (itr.isValid() && itr.get().getEnd() > annotation.getBegin()) {
885                            itr.moveToPrevious();
886                    }
887    
888                    // add annotations from the iterator into the result list
889                    List<AnnotationFS> precedingAnnotations = new LinkedList<AnnotationFS>();
890                    for (int i = 0; i < count && itr.isValid(); ++i, itr.moveToPrevious()) {
891                            precedingAnnotations.add(itr.get());
892                    }
893    
894                    // return in correct order
895                    Collections.reverse(precedingAnnotations);
896                    return precedingAnnotations;
897            }
