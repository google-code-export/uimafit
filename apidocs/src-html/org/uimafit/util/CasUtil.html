257            /**
258             * Get a list of annotations of the given annotation type constraint by a certain annotation.
259             * Iterates over all annotations of the given type to find the covered annotations. Does not use
260             * subiterators and does not respect type prioritites. Was adapted from {@link Subiterator}.
261             * Uses the same approach except that type priorities are ignored.
262             *
263             * @param cas
264             *            a CAS.
265             * @param type
266             *            a UIMA type.
267             * @param coveringAnnotation
268             *            the covering annotation.
269             * @return a return value.
270             * @see Subiterator
271             */
272            public static List<AnnotationFS> selectCovered(CAS cas, Type type,
273                            AnnotationFS coveringAnnotation) {
274                    int begin = coveringAnnotation.getBegin();
275                    int end = coveringAnnotation.getEnd();
276    
277                    List<AnnotationFS> list = new ArrayList<AnnotationFS>();
278                    FSIterator<AnnotationFS> it = cas.getAnnotationIndex(type).iterator();
279    
280                    // Try to seek the insertion point.
281                    it.moveTo(coveringAnnotation);
282    
283                    // If the insertion point is beyond the index, move back to the last.
284                    if (!it.isValid()) {
285                            it.moveToLast();
286                            if (!it.isValid()) {
287                                    return list;
288                            }
289                    }
290    
291                    // Ignore type priorities by seeking to the first that has the same begin
292                    boolean moved = false;
293                    while (it.isValid() && (it.get()).getBegin() >= begin) {
294                            it.moveToPrevious();
295                            moved = true;
296                    }
297    
298                    // If we moved, then we are now on one starting before the requested begin, so we have to
299                    // move one ahead.
300                    if (moved) {
301                            it.moveToNext();
302                    }
303    
304                    // If we managed to move outside the index, start at first.
305                    if (!it.isValid()) {
306                            it.moveToFirst();
307                    }
308    
309                    // Skip annotations whose start is before the start parameter.
310                    while (it.isValid() && (it.get()).getBegin() < begin) {
311                            it.moveToNext();
312                    }
313    
314                    boolean strict = true;
315                    while (it.isValid()) {
316                            AnnotationFS a = it.get();
317                            // If the start of the current annotation is past the end parameter, we're done.
318                            if (a.getBegin() > end) {
319                                    break;
320                            }
321                            it.moveToNext();
322                            if (strict && a.getEnd() > end) {
323                                    continue;
324                            }
325    
326                            assert !(a.getBegin() < coveringAnnotation.getBegin()) : "Illegal begin "
327                                            + a.getBegin() + " in [" + coveringAnnotation.getBegin() + ".."
328                                            + coveringAnnotation.getEnd() + "]";
329    
330                            assert !(a.getEnd() < coveringAnnotation.getBegin()) : "Illegal end " + a.getEnd()
331                                            + " in [" + coveringAnnotation.getBegin() + ".." + coveringAnnotation.getEnd()
332                                            + "]";
333    
334                            if (!a.equals(coveringAnnotation)) {
335                                    list.add(a);
336                            }
337                    }
338    
339                    return unmodifiableList(list);
340            }

342            /**
343             * Get a list of annotations of the given annotation type constraint by a certain annotation.
344             * Iterates over all annotations of the given type to find the covered annotations. Does not use
345             * subiterators and does not respect type prioritites. Was adapted from {@link Subiterator}.
346             * Uses the same approach except that type priorities are ignored.
347             * <p>
348             * <b>Note:</b> this is significantly slower than using
349             * {@link #selectCovered(CAS, Type, AnnotationFS)}
350             *
351             * @param cas
352             *            a CAS.
353             * @param type
354             *            a UIMA type.
355             * @param begin
356             *            begin offset.
357             * @param end
358             *            end offset.
359             * @return a return value.
360             * @see Subiterator
361             */
362            public static List<AnnotationFS> selectCovered(CAS cas, Type type, int begin, int end) {
363    
364                    List<AnnotationFS> list = new ArrayList<AnnotationFS>();
365                    FSIterator<AnnotationFS> it = cas.getAnnotationIndex(type).iterator();
366    
367                    // Skip annotations whose start is before the start parameter.
368                    while (it.isValid() && (it.get()).getBegin() < begin) {
369                            it.moveToNext();
370                    }
371    
372                    boolean strict = true;
373                    while (it.isValid()) {
374                            AnnotationFS a = it.get();
375                            // If the start of the current annotation is past the end parameter, we're done.
376                            if (a.getBegin() > end) {
377                                    break;
378                            }
379                            it.moveToNext();
380                            if (strict && a.getEnd() > end) {
381                                    continue;
382                            }
383    
384                            assert !(a.getBegin() < begin) : "Illegal begin " + a.getBegin() + " in [" + begin
385                                            + ".." + end + "]";
386    
387                            assert !(a.getEnd() < begin) : "Illegal end " + a.getEnd() + " in [" + begin + ".."
388                                            + end + "]";
389    
390                            list.add(a);
391                    }
392    
393                    return list;
394            }

396            /**
397             * Get a list of annotations of the given annotation type constraint by a certain annotation.
398             * Iterates over all annotations to find the covering annotations.
399             *
400             * <p>
401             * <b>Note:</b> this is <b>REALLY SLOW!</b> You don't want to use this.
402             *
403             * @param cas
404             *            a CAS.
405             * @param type
406             *            a UIMA type.
407             * @param begin
408             *            begin offset.
409             * @param end
410             *            end offset.
411             * @return a return value.
412             */
413            public static List<AnnotationFS> selectCovering(CAS cas, Type type, int begin, int end) {
414    
415                    TypeSystem ts = cas.getTypeSystem();
416                    List<AnnotationFS> list = new ArrayList<AnnotationFS>();
417                    FSIterator<AnnotationFS> iter = cas.getAnnotationIndex().iterator();
418                    while (iter.hasNext()) {
419                            AnnotationFS a = iter.next();
420                            if (a.getBegin() <= begin && a.getEnd() >= end) {
421                                    if (type == null || ts.subsumes(type, a.getType())) {
422                                            list.add(a);
423                                    }
424                            }
425                    }
426                    return list;
427            }

429            /**
430             * Create an index for quickly lookup up the annotations covering a particular annotation. This
431             * is preferable to using {@link #selectCovering(CAS, Type, int, int)} because the overhead of
432             * scanning the CAS occurs only when the index is build. Subsequent lookups to the index are
433             * fast.
434             *
435             * @param cas
436             *            a CAS.
437             * @param type
438             *            type to create the index for - this is used in lookups.
439             * @param coveringType
440             *            type of covering annotations.
441             * @return the index.
442             */
443            public static Map<AnnotationFS, Collection<AnnotationFS>> indexCovering(CAS cas, Type type,
444                            Type coveringType) {
445                    Map<AnnotationFS, Collection<AnnotationFS>> index = new HashMap<AnnotationFS, Collection<AnnotationFS>>() {
446                            private static final long serialVersionUID = 1L;
447    
448                            @Override
449                            public Collection<AnnotationFS> get(Object paramObject) {
450                                    Collection<AnnotationFS> res = super.get(paramObject);
451                                    if (res == null) {
452                                            return emptyList();
453                                    }
454                                    else {
455                                            return res;
456                                    }
457                            }
458                    };
459                    for (AnnotationFS s : select(cas, coveringType)) {
460                            for (AnnotationFS u : selectCovered(cas, type, s)) {
461                                    Collection<AnnotationFS> c = index.get(u);
462                                    if (c == EMPTY_LIST) {
463                                            c = new LinkedList<AnnotationFS>();
464                                            index.put(u, c);
465                                    }
466                                    c.add(s);
467                            }
468                    }
469                    return unmodifiableMap(index);
470            }

514            /**
515             * This method exists simply as a convenience method for unit testing. It is not very efficient
516             * and should not, in general be used outside the context of unit testing.
517             *
518             * @param cas
519             *            a CAS containing the feature structure.
520             * @param type
521             *            a UIMA type.
522             * @param index
523             *            this can be either positive (0 corresponds to the first annotation of a type) or
524             *            negative (-1 corresponds to the last annotation of a type.)
525             * @return an annotation of the given type
526             */
527            public static FeatureStructure selectFSByIndex(CAS cas, Type type, int index) {
528                    FSIterator<FeatureStructure> i = cas.getIndexRepository().getAllIndexedFS(type);
529                    int n = index;
530                    i.moveToFirst();
531                    if (n > 0) {
532                            while (n > 0 && i.isValid()) {
533                                    i.moveToNext();
534                                    n--;
535                            }
536                    }
537                    if (n < 0) {
538                            i.moveToLast();
539                            while (n < -1 && i.isValid()) {
540                                    i.moveToPrevious();
541                                    n++;
542                            }
543                    }
544    
545                    return i.isValid() ? i.get() : null;
546            }

592             *            a UIMA type.

596            public static FeatureStructure selectSingle(CAS cas, Type type) {
597                    FSIterator<FeatureStructure> iterator = cas.getIndexRepository().getAllIndexedFS(type);

599                    if (!iterator.hasNext()) {
600                            throw new IllegalArgumentException("CAS does not contain any [" + type.getName() + "]");

605                    if (iterator.hasNext()) {

607                                            + "]");
608                    }
609    
610                    return result;
611            }

613            /**
614             * Returns the n annotations preceding the given annotation
615             *
616             * @param cas
617             *            a CAS.
618             * @param type
619             *            a UIMA type.
620             * @param annotation
621             *            anchor annotation
622             * @param count
623             *            number of annotations to collect
624             * @return List of aType annotations preceding anchor annotation
625             */
626            public static List<AnnotationFS> selectPreceding(CAS cas, Type type, Annotation annotation,
627                            int count) {
628                    if (!cas.getTypeSystem().subsumes(cas.getAnnotationType(), type)) {
629                            throw new IllegalArgumentException("Type [" + type.getName()
630                                            + "] is not an annotation type");
631                    }
632    
633                    // move to first previous annotation
634                    FSIterator<AnnotationFS> itr = cas.getAnnotationIndex(type).iterator();
635                    itr.moveTo(annotation);
636    
637                    // make sure we're past the beginning of the reference annotation
638                    while (itr.isValid() && itr.get().getEnd() > annotation.getBegin()) {
639                            itr.moveToPrevious();
640                    }
641    
642                    // add annotations from the iterator into the result list
643                    List<AnnotationFS> precedingAnnotations = new LinkedList<AnnotationFS>();
644                    for (int i = 0; i < count && itr.isValid(); ++i, itr.moveToPrevious()) {
645                            precedingAnnotations.add(itr.get());
646                    }
647    
648                    // return in correct order
649                    Collections.reverse(precedingAnnotations);
650                    return precedingAnnotations;
651            }

715             * Convenience method to get the specified view or create a new view if the requested view does
716             * not exist.
717             *
718             * @param cas
719             *            a CAS
720             * @param viewName
721             *            the requested view.
722             * @param create
723             *            the view is created if it does not exist.
724             * @return the requested view
725             * @throws IllegalArgumentException
726             *             if the view does not exist and is not to be created.
727             */
728            public static CAS getView(CAS cas, String viewName, boolean create) {
729                    CAS view = null;
730                    try {
731                            view = cas.getView(viewName);
732                    }
733                    catch (CASRuntimeException e) {
734                            // View does not exist
735                    }
736    
737                    if (view == null && create) {
738                            view = cas.createView(viewName);
739                    }
740    
741                    if (view == null) {
742                            throw new IllegalArgumentException("No view with name [" + viewName + "]");
743                    }
744    
745                    return view;
746            }

749             * Fetch the text covered by the specified annotations and return it as a list of strings.

753             * @param iterable
754             *            annotation container.
755             * @return list of covered strings.

761            /**
762             * Fetch the text covered by the specified annotations and return it as a list of strings.
763             *
764             * @param <T>
765             *            UIMA JCas type.
766             * @param iterator
767             *            annotation iterator.
768             * @return list of covered strings.
769             */
770            public static <T extends AnnotationFS> List<String> toText(Iterator<T> iterator) {
771                    List<String> text = new ArrayList<String>();
772                    while (iterator.hasNext()) {
773                            AnnotationFS a = iterator.next();
774                            text.add(a.getCoveredText());
775                    }
776                    return text;
777            }
