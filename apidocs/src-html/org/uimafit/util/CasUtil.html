148                    String typeName = aTypename;
149                    if (typeName.startsWith(UIMA_BUILTIN_JCAS_PREFIX)) {
150                            typeName = "uima." + typeName.substring(UIMA_BUILTIN_JCAS_PREFIX.length());

154                            throw new IllegalArgumentException("Undeclared type [" + aTypename + "]");

281            /**
282             * Get a list of annotations of the given annotation type located between two annotations.
283             * Does not use subiterators and does not respect type priorities. Zero-width annotations
284             * what lie on the borders are included in the result, e.g. if the boundary annotations are
285             * [1..2] and [2..3] then an annotation [2..2] is returned. If there is a non-zero overlap
286             * between the boundary annotations, the result is empty. The method properly handles cases
287             * where the second boundary annotations occurs before the first boundary annotation by
288             * switching their roles.
289             *
290             * @param cas
291             *            a CAS.
292             * @param type
293             *            a UIMA type.
294             * @param ann1
295             *            the first boundary annotation.
296             * @param ann2
297             *            the second boundary annotation.
298             * @return a return value.
299             * @see Subiterator
300             */
301            public static List<AnnotationFS> selectBetween(final CAS cas, final Type type, 
302                            final AnnotationFS ann1, final AnnotationFS ann2) {
303                    AnnotationFS left;
304                    AnnotationFS right;
305                    if (ann1.getEnd() > ann2.getBegin()) {
306                            left = ann2;
307                            right = ann1;
308                    }
309                    else {
310                            left = ann1;
311                            right = ann2;
312                    }
313                    
314                    int begin = left.getEnd();
315                    int end = right.getBegin();
316    
317                    List<AnnotationFS> list = new ArrayList<AnnotationFS>();
318                    FSIterator<AnnotationFS> it = cas.getAnnotationIndex(type).iterator();
319    
320                    // Try to seek the insertion point.
321                    it.moveTo(left);
322    
323                    // If the insertion point is beyond the index, move back to the last.
324                    if (!it.isValid()) {
325                            it.moveToLast();
326                            if (!it.isValid()) {
327                                    return list;
328                            }
329                    }
330    
331                    // Ignore type priorities by seeking to the first that has the same begin
332                    boolean moved = false;
333                    while (it.isValid() && (it.get()).getBegin() >= begin) {
334                            it.moveToPrevious();
335                            moved = true;
336                    }
337    
338                    // If we moved, then we are now on one starting before the requested begin, so we have to
339                    // move one ahead.
340                    if (moved) {
341                            it.moveToNext();
342                    }
343    
344                    // If we managed to move outside the index, start at first.
345                    if (!it.isValid()) {
346                            it.moveToFirst();
347                    }
348    
349                    // Skip annotations whose start is before the start parameter.
350                    while (it.isValid() && (it.get()).getBegin() < begin) {
351                            it.moveToNext();
352                    }
353    
354                    boolean strict = true;
355                    while (it.isValid()) {
356                            AnnotationFS a = it.get();
357                            // If the start of the current annotation is past the end parameter, we're done.
358                            if (a.getBegin() > end) {
359                                    break;
360                            }
361                            it.moveToNext();
362                            if (strict && a.getEnd() > end) {
363                                    continue;
364                            }
365    
366                            assert (a.getBegin() >= left.getEnd()) : "Illegal begin "
367                                            + a.getBegin() + " in [" + begin + ".." + end + "]";
368    
369                            assert (a.getEnd() <= right.getBegin()) : "Illegal end "
370                                            + a.getBegin() + " in [" + begin + ".." + end + "]";
371    
372    
373                            if (!a.equals(left) && !a.equals(right)) {
374                                    list.add(a);
375                            }
376                    }
377    
378                    return unmodifiableList(list);  }

536            /**
537             * Get a list of annotations of the given annotation type constraint by a certain annotation.
538             * Iterates over all annotations to find the covering annotations.
539             *
540             * <p>
541             * <b>Note:</b> this is <b>REALLY SLOW!</b> You don't want to use this.
542             *
543             * @param cas
544             *            a CAS.
545             * @param type
546             *            a UIMA type.
547             * @param begin
548             *            begin offset.
549             * @param end
550             *            end offset.
551             * @return a return value.
552             */
553            public static List<AnnotationFS> selectCovering(CAS cas, Type type, int begin, int end) {
554    
555                    TypeSystem ts = cas.getTypeSystem();
556                    List<AnnotationFS> list = new ArrayList<AnnotationFS>();
557                    FSIterator<AnnotationFS> iter = cas.getAnnotationIndex().iterator();
558                    while (iter.hasNext()) {
559                            AnnotationFS a = iter.next();
560                            if ((a.getBegin() <= begin) && (a.getEnd() >= end)
561                                            && ((type == null) || (ts.subsumes(type, a.getType())))) {
562                                    list.add(a);
563                            }
564                    }
565                    return list;
566            }

568            /**
569             * Create an index for quickly lookup up the annotations covering a particular annotation. This
570             * is preferable to using {@link #selectCovering(CAS, Type, int, int)} because the overhead of
571             * scanning the CAS occurs only when the index is build. Subsequent lookups to the index are
572             * fast.
573             *
574             * @param cas
575             *            a CAS.
576             * @param type
577             *            type to create the index for - this is used in lookups.
578             * @param coveringType
579             *            type of covering annotations.
580             * @return the index.
581             */
582            public static Map<AnnotationFS, Collection<AnnotationFS>> indexCovering(CAS cas, Type type,
583                            Type coveringType) {
584                    Map<AnnotationFS, Collection<AnnotationFS>> index = new HashMap<AnnotationFS, Collection<AnnotationFS>>() {
585                            private static final long serialVersionUID = 1L;
586    
587                            @Override
588                            public Collection<AnnotationFS> get(Object paramObject) {
589                                    Collection<AnnotationFS> res = super.get(paramObject);
590                                    if (res == null) {
591                                            return emptyList();
592                                    }
593                                    else {
594                                            return res;
595                                    }
596                            }
597                    };
598                    for (AnnotationFS s : select(cas, coveringType)) {
599                            for (AnnotationFS u : selectCovered(cas, type, s)) {
600                                    Collection<AnnotationFS> c = index.get(u);
601                                    if (c == EMPTY_LIST) {
602                                            c = new LinkedList<AnnotationFS>();
603                                            index.put(u, c);
604                                    }
605                                    c.add(s);
606                            }
607                    }
608                    return unmodifiableMap(index);
609            }

653            /**
654             * This method exists simply as a convenience method for unit testing. It is not very efficient
655             * and should not, in general be used outside the context of unit testing.
656             *
657             * @param cas
658             *            a CAS containing the feature structure.
659             * @param type
660             *            a UIMA type.
661             * @param index
662             *            this can be either positive (0 corresponds to the first annotation of a type) or
663             *            negative (-1 corresponds to the last annotation of a type.)
664             * @return an annotation of the given type
665             */
666            public static FeatureStructure selectFSByIndex(CAS cas, Type type, int index) {
667                    FSIterator<FeatureStructure> i = cas.getIndexRepository().getAllIndexedFS(type);
668                    int n = index;
669                    i.moveToFirst();
670                    if (n > 0) {
671                            while (n > 0 && i.isValid()) {
672                                    i.moveToNext();
673                                    n--;
674                            }
675                    }
676                    if (n < 0) {
677                            i.moveToLast();
678                            while (n < -1 && i.isValid()) {
679                                    i.moveToPrevious();
680                                    n++;
681                            }
682                    }
683    
684                    return i.isValid() ? i.get() : null;
685            }

699             */
700            public static AnnotationFS selectByIndex(CAS cas, Type type, int index) {
701                    if (!cas.getTypeSystem().subsumes(cas.getAnnotationType(), type)) {
702                            throw new IllegalArgumentException("Type [" + type.getName()
703                                            + "] is not an annotation type");
704                    }
705                    FSIterator<AnnotationFS> i = cas.getAnnotationIndex(type).iterator();
706                    int n = index;
707                    i.moveToFirst();
708                    if (n > 0) {
709                            while (n > 0 && i.isValid()) {
710                                    i.moveToNext();
711                                    n--;
712                            }
713                    }
714                    if (n < 0) {
715                            i.moveToLast();
716                            while (n < -1 && i.isValid()) {
717                                    i.moveToPrevious();
718                                    n++;
719                            }
720                    }
721    
722                    return i.isValid() ? i.get() : null;
723            }

728             * @param cas

730             * @param type

733             *         one instance if the given type is present.

746                                            + "]");
747                    }
748    
749                    return result;

752            /**
753             * Returns the n annotations preceding the given annotation
754             *
755             * @param cas
756             *            a CAS.
757             * @param type
758             *            a UIMA type.
759             * @param annotation
760             *            anchor annotation
761             * @param count
762             *            number of annotations to collect
763             * @return List of aType annotations preceding anchor annotation
764             */
765            public static List<AnnotationFS> selectPreceding(CAS cas, Type type, Annotation annotation,
766                            int count) {
767                    if (!cas.getTypeSystem().subsumes(cas.getAnnotationType(), type)) {
768                            throw new IllegalArgumentException("Type [" + type.getName()
769                                            + "] is not an annotation type");
770                    }
771    
772                    // move to first previous annotation
773                    FSIterator<AnnotationFS> itr = cas.getAnnotationIndex(type).iterator();
774                    itr.moveTo(annotation);
775    
776                    // make sure we're past the beginning of the reference annotation
777                    while (itr.isValid() && itr.get().getEnd() > annotation.getBegin()) {
778                            itr.moveToPrevious();
779                    }
780    
781                    // add annotations from the iterator into the result list
782                    List<AnnotationFS> precedingAnnotations = new LinkedList<AnnotationFS>();
783                    for (int i = 0; i < count && itr.isValid(); ++i, itr.moveToPrevious()) {
784                            precedingAnnotations.add(itr.get());
785                    }
786    
787                    // return in correct order
788                    Collections.reverse(precedingAnnotations);
789                    return precedingAnnotations;
790            }

900             *
901             * @param <T>
902             *            UIMA JCas type.
903             * @param iterator
904             *            annotation iterator.
905             * @return list of covered strings.
906             */
907            public static <T extends AnnotationFS> List<String> toText(Iterator<T> iterator) {
