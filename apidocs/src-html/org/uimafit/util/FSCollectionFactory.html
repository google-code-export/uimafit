023    import static java.util.Arrays.asList;
024

026    import java.util.ArrayList;
027    import java.util.Collection;
028    import java.util.Iterator;
029    import java.util.List;

031    import org.apache.uima.cas.ArrayFS;
032    import org.apache.uima.cas.CAS;
033    import org.apache.uima.cas.FSIterator;
034    import org.apache.uima.cas.FeatureStructure;
035    import org.apache.uima.cas.Type;
036    import org.apache.uima.cas.TypeSystem;
037    import org.apache.uima.cas.text.AnnotationFS;
038    import org.apache.uima.cas.text.AnnotationIndex;

043    /**
044     * Bridge between Java {@link Collection Collections} from different representations of collections
045     * of UIMA {@link FeatureStructure FeatureStructures}.
046     *
047     * @author Richard Eckart de Castilho
048     *
049     * @param <T> data type.
050     */
051    public abstract class FSCollectionFactory<T extends FeatureStructure> extends AbstractCollection<T> {
052    
053            @SuppressWarnings({ "unchecked", "rawtypes" })
054            public static Collection<FeatureStructure> create(CAS cas, Type type)
055            {
056                    // If the type is an annotation type, we can use the annotation index, which directly
057                    // provides us with its size. If not, we have to use getAllIndexedFS() which we have to
058                    // scan from beginning to end in order to determine its size.
059                    TypeSystem ts = cas.getTypeSystem();
060                    if (ts.subsumes(cas.getAnnotationType(), type)) {
061                            return (Collection) create(cas.getAnnotationIndex(type));
062                    }
063                    else {
064                            return create(cas.getIndexRepository().getAllIndexedFS(type));
065                    }
066            }
067    
068            public static <T extends FeatureStructure> Collection<T> create(FSIterator<T> aIterator)
069            {
070                    return new FSIteratorAdapter<T>(aIterator);
071            }
072    
073            public static <T extends AnnotationFS> Collection<T> create(AnnotationIndex<T> aIndex)
074            {
075                    return new AnnotationIndexAdapter<T>(aIndex);
076            }
077    
078            public static Collection<FeatureStructure> create(ArrayFS aArray)
079            {
080                    return create(aArray, null);
081            }
082    
083            public static Collection<FeatureStructure> create(ArrayFS aArray, Type type)
084            {
085                    TypeSystem ts = aArray.getCAS().getTypeSystem();
086                    List<FeatureStructure> data = new ArrayList<FeatureStructure>(aArray.size());
087                    for (int i = 0; i < aArray.size(); i++) {
088                            FeatureStructure value = aArray.get(i);
089                            if (value != null && (type == null || ts.subsumes(type, value.getType()))) {
090                                    data.add(value);
091                            }
092                    }
093                    return asList(data.toArray(new FeatureStructure[data.size()]));
094            }
095    
096            // Using TOP here because FSList is only available in the JCas.
097            public static Collection<TOP> create(FSList aList)
098            {
099                    return create(aList, null);
100            }
101    
102            // Using TOP here because FSList is only available in the JCas.
103            public static Collection<TOP> create(FSList aList, Type type)
104            {
105                    TypeSystem ts = aList.getCAS().getTypeSystem();
106                    List<FeatureStructure> data = new ArrayList<FeatureStructure>();
107                    FSList i = aList;
108                    while (i instanceof NonEmptyFSList) {
109                            NonEmptyFSList l = (NonEmptyFSList) i;
110                            TOP value = l.getHead();
111                            if (value != null && (type == null || ts.subsumes(type, value.getType()))) {
112                                    data.add(l.getHead());
113                            }
114                            i = l.getTail();
115                    }
116    
117                    return asList(data.toArray(new TOP[data.size()]));
118            }
119    
120            private static class FSIteratorAdapter<T extends FeatureStructure> extends FSCollectionFactory<T>
121            {
122                    private volatile int sizeCache = -1;
123                    private final FSIterator<T> index;
124    
125                    public FSIteratorAdapter(FSIterator<T> aIterator)
126                    {
127                            index = aIterator.copy();
128                            index.moveToFirst();
129                    }
130    
131                    @Override
132                    public Iterator<T> iterator() {
133                            return index.copy();
134                    }
135    
136                    @Override
137                    public int size() {
138                            // Unfortunately FSIterator does not expose the sizes of its internal collection,
139                            // neither the current position although FSIteratorAggregate has a private field
140                            // with that information.
141                            if (sizeCache == -1) {
142                                    synchronized (this) {
143                                            if (sizeCache == -1) {
144                                                    FSIterator<T> clone = index.copy();
145                                                    clone.moveToFirst();
146                                                    sizeCache = 0;
147                                                    while (clone.isValid()) {
148                                                            sizeCache++;
149                                                            clone.moveToNext();
150                                                    }
151                                            }
152                                    }
153                            }
154    
155                            return sizeCache;
156                    }
157            }
158    
159            private static class AnnotationIndexAdapter<T extends AnnotationFS> extends FSCollectionFactory<T>
160            {
161                    private final AnnotationIndex<T> index;
162    
163                    public AnnotationIndexAdapter(AnnotationIndex<T> aIndex)
164                    {
165                            index = aIndex;
166                    }
167    
168                    @Override
169                    public Iterator<T> iterator() {
170                            return index.iterator();
171                    }
172    
173                    @Override
174                    public int size() {
175                            return index.size();
176                    }
177            }
178    }
