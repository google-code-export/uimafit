106            /**
107             * Convenience method to iterator over all annotations of a given type occurring within the
108             * scope of a provided annotation (sub-iteration).
109             *
110             * @param <T>
111             *            the iteration type.
112             * @param jCas
113             *            a JCas.
114             * @param container
115             *            the containing annotation.
116             * @param type
117             *            the type.
118             * @param ambiguous
119             *            If set to <code>false</code>, resulting iterator will be unambiguous.
120             * @param strict
121             *            Controls if annotations that overlap to the right are considered in or out.
122             * @return A sub-iterator iterable.
123             * @see AnnotationIndex#subiterator(AnnotationFS, boolean, boolean)
124             */
125            public static <T extends Annotation> Iterable<T> subiterate(final JCas jCas,
126                            final Class<T> type, final Annotation container, final boolean ambiguous,
127                            final boolean strict) {
128                    return new Iterable<T>() {
129                            public Iterator<T> iterator() {
130                                    return JCasUtil.iterator(container, type, ambiguous, strict);
131                            }
132                    };
133            }

136             * Get an iterator over the given feature structure type.
137             *
138             * @param <T>
139             *            the JCas type.
140             * @param jCas
141             *            a JCas.
142             * @param type
143             *            a type.
144             * @return a return value.

146            @SuppressWarnings({ "unchecked", "rawtypes" })
147            public static <T extends TOP> Iterator<T> iterator(JCas jCas, Class<T> type) {
148                    return (FSIterator) jCas.getIndexRepository().getAllIndexedFS(getType(jCas, type));
149            }

151            /**
152             * Convenience method to get a sub-iterator for the specified type.
153             *
154             * @param <T>
155             *            the iteration type.
156             * @param container
157             *            the containing annotation.
158             * @param type
159             *            the type.
160             * @param ambiguous
161             *            If set to <code>false</code>, resulting iterator will be unambiguous.
162             * @param strict
163             *            Controls if annotations that overlap to the right are considered in or out.
164             * @return A sub-iterator.
165             */
166            @SuppressWarnings("unchecked")
167            public static <T extends AnnotationFS> Iterator<T> iterator(Annotation container,
168                            Class<T> type, boolean ambiguous, boolean strict) {
169                    CAS cas = container.getCAS();
170                    return ((AnnotationIndex<T>) cas.getAnnotationIndex(CasUtil.getType(cas, type)))
171                                    .subiterator(container, ambiguous, strict);
172            }

342             * Create an index for quickly lookup up the annotations covering a particular annotation. This
343             * is preferable to using {@link #selectCovering(JCas, Class, int, int)} because the overhead of
344             * scanning the CAS occurs only when the index is build. Subsequent lookups to the index are
345             * fast.
346             *
347             * @param <T>
348             *            the JCas type.
349             * @param jCas
350             *            a JCas.
351             * @param type
352             *            type to create the index for - this is used in lookups.
353             * @param coveringType
354             *            type of covering annotations.
355             * @return the index.
356             */
357            public static <T extends Annotation, S extends Annotation> Map<T, Collection<S>> indexCovering(
358                            JCas jCas, Class<T> type, Class<S> coveringType) {
359                    return cast(CasUtil.indexCovering(jCas.getCas(), getType(jCas, type),
360                                    getType(jCas, coveringType)));
361            }

495            /**
496             * Convenience method to get the specified view or a default view if the requested view does not
497             * exist. The default can also be {@code null}.
498             *
499             * @param jcas
500             *            a JCas
501             * @param viewName
502             *            the requested view.
503             * @param fallback
504             *            the default view if the requested view does not exist.
505             * @return the requested view or the default if the requested view does not exist.
506             * @throws IllegalStateException
507             *             if the JCas wrapper cannot be obtained.
508             */
509            public static JCas getView(JCas jcas, String viewName, JCas fallback) {
510                    try {
511                            CAS fallbackCas = fallback != null ? fallback.getCas() : null;
512                            CAS view = CasUtil.getView(jcas.getCas(), viewName, fallbackCas);
513                            return view != null ? view.getJCas() : null;
514                    }
515                    catch (CASException e) {
516                            throw new IllegalStateException(e);
517                    }
518            }
519    
520            /**
521             * Convenience method to get the specified view or create a new view if the requested view does
522             * not exist.
523             *
524             * @param jcas
525             *            a JCas
526             * @param viewName
527             *            the requested view.
528             * @param create
529             *            the view is created if it does not exist.
530             * @return the requested view
531             * @throws IllegalStateException
532             *             if the JCas wrapper cannot be obtained.
533             */
534            public static JCas getView(JCas jcas, String viewName, boolean create) {
535                    try {
536                            CAS view = CasUtil.getView(jcas.getCas(), viewName, create);
537                            return view != null ? view.getJCas() : null;
538                    }
539                    catch (CASException e) {
540                            throw new IllegalStateException(e);
541                    }
542            }
