107             * Convenience method to iterator over all annotations of a given type occurring within the
108             * scope of a provided annotation (sub-iteration).
109             *
110             * @param <T>
111             *            the iteration type.
112             * @param jCas
113             *            a JCas.
114             * @param container
115             *            the containing annotation.
116             * @param type
117             *            the type.
118             * @param ambiguous
119             *            If set to <code>false</code>, resulting iterator will be unambiguous.
120             * @param strict
121             *            Controls if annotations that overlap to the right are considered in or out.
122             * @return A sub-iterator iterable.
123             * @see AnnotationIndex#subiterator(AnnotationFS, boolean, boolean)
124             */
125            public static <T extends Annotation> Iterable<T> subiterate(final JCas jCas,
126                            final Class<T> type, final Annotation container, final boolean ambiguous,
127                            final boolean strict) {
128                    return new Iterable<T>() {
129                            public Iterator<T> iterator() {
130                                    return JCasUtil.iterator(container, type, ambiguous, strict);
131                            }
132                    };
133            }

152             * Convenience method to get a sub-iterator for the specified type.
153             *
154             * @param <T>
155             *            the iteration type.
156             * @param container
157             *            the containing annotation.
158             * @param type
159             *            the type.
160             * @param ambiguous
161             *            If set to <code>false</code>, resulting iterator will be unambiguous.
162             * @param strict
163             *            Controls if annotations that overlap to the right are considered in or out.
164             * @return A sub-iterator.
165             */
166            @SuppressWarnings("unchecked")
167            public static <T extends AnnotationFS> Iterator<T> iterator(Annotation container,
168                            Class<T> type, boolean ambiguous, boolean strict) {
169                    CAS cas = container.getCAS();
170                    return ((AnnotationIndex<T>) cas.getAnnotationIndex(CasUtil.getType(cas, type)))
171                                    .subiterator(container, ambiguous, strict);
172            }

250            /**
251             * Get a list of annotations of the given annotation type located between two annotations.
252             * Does not use subiterators and does not respect type priorities. Zero-width annotations
253             * what lie on the borders are included in the result, e.g. if the boundary annotations are
254             * [1..2] and [2..3] then an annotation [2..2] is returned. If there is a non-zero overlap
255             * between the boundary annotations, the result is empty. The method properly handles cases
256             * where the second boundary annotations occurs before the first boundary annotation by
257             * switching their roles.
258             *
259             * @param type
260             *            a UIMA type.
261             * @param ann1
262             *            the first boundary annotation.
263             * @param ann2
264             *            the second boundary annotation.
265             * @return a return value.
266             * @see Subiterator
267             */
268            public static <T extends Annotation> List<T> selectBetween(final Class<T> type,
269                             AnnotationFS ann1, AnnotationFS ann2) {
270                    return cast(CasUtil.selectBetween(CasUtil.getType(ann1.getCAS(), type), ann1, ann2));
271            }

543            /**
544             * Convenience method to get the specified view or a default view if the requested view does not
545             * exist. The default can also be {@code null}.
546             *
547             * @param jcas
548             *            a JCas
549             * @param viewName
550             *            the requested view.
551             * @param fallback
552             *            the default view if the requested view does not exist.
553             * @return the requested view or the default if the requested view does not exist.
554             * @throws IllegalStateException
555             *             if the JCas wrapper cannot be obtained.
556             */
557            public static JCas getView(JCas jcas, String viewName, JCas fallback) {
558                    try {
559                            CAS fallbackCas = fallback != null ? fallback.getCas() : null;
560                            CAS view = CasUtil.getView(jcas.getCas(), viewName, fallbackCas);
561                            return view != null ? view.getJCas() : null;
562                    }
563                    catch (CASException e) {
564                            throw new IllegalStateException(e);
565                    }
566            }

570             * not exist.

576             * @param create
577             *            the view is created if it does not exist.
578             * @return the requested view

582            public static JCas getView(JCas jcas, String viewName, boolean create) {
