111             *            the iteration type.
112             * @param jCas
113             *            a JCas.
114             * @param container
115             *            the containing annotation.
116             * @param type
117             *            the type.
118             * @param ambiguous
119             *            If set to <code>false</code>, resulting iterator will be unambiguous.
120             * @param strict
121             *            Controls if annotations that overlap to the right are considered in or out.
122             * @return A sub-iterator iterable.
123             * @see AnnotationIndex#subiterator(AnnotationFS, boolean, boolean)

126                            final Class<T> type, final AnnotationFS container, final boolean ambiguous,
127                            final boolean strict) {
128                    return new Iterable<T>() {
129                            public Iterator<T> iterator() {
130                                    return JCasUtil.iterator(container, type, ambiguous, strict);
131                            }
132                    };
133            }

262             * Get a list of annotations of the given annotation type located between two annotations.
263             * Does not use subiterators and does not respect type priorities. Zero-width annotations
264             * what lie on the borders are included in the result, e.g. if the boundary annotations are
265             * [1..2] and [2..3] then an annotation [2..2] is returned. If there is a non-zero overlap
266             * between the boundary annotations, the result is empty. The method properly handles cases
267             * where the second boundary annotations occurs before the first boundary annotation by
268             * switching their roles.
269             *
270             * @param type
271             *            a UIMA type.
272             * @param ann1
273             *            the first boundary annotation.
274             * @param ann2
275             *            the second boundary annotation.
276             * @return a return value.
277             * @see Subiterator
278             */
279            public static <T extends Annotation> List<T> selectBetween(final Class<T> type,
280                             AnnotationFS ann1, AnnotationFS ann2) {
281                    return cast(CasUtil.selectBetween(CasUtil.getType(ann1.getCAS(), type), ann1, ann2));
282            }

623             * Convenience method to get the specified view or a default view if the requested view does not
624             * exist. The default can also be {@code null}.
625             *
626             * @param jcas
627             *            a JCas
628             * @param viewName
629             *            the requested view.
630             * @param fallback
631             *            the default view if the requested view does not exist.
632             * @return the requested view or the default if the requested view does not exist.
633             * @throws IllegalStateException
634             *             if the JCas wrapper cannot be obtained.
635             */
636            public static JCas getView(JCas jcas, String viewName, JCas fallback) {
637                    try {
638                            CAS fallbackCas = fallback != null ? fallback.getCas() : null;
639                            CAS view = CasUtil.getView(jcas.getCas(), viewName, fallbackCas);
640                            return view != null ? view.getJCas() : null;
641                    }
642                    catch (CASException e) {
643                            throw new IllegalStateException(e);
644                    }
645            }
