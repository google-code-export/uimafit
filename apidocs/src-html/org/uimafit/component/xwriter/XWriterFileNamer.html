018    package org.uimafit.component.xwriter;
019    
020    import org.apache.uima.jcas.JCas;
021    
022    /**
023     * This interface provides {@link XWriter} with a way of getting a file name for a given jCas. If
024     * there is information in {@link JCas} that you can use to name the resulting ".xmi" or ".xcas"
025     * files, then this interface will allow you to use it. Implementation of this interface should come
026     * up with a file name to be used for the file that it is written by {@link XWriter}. Do not specify
027     * the full path and do not specify the suffix (.xmi or .xcas will be automatically appended as
028     * appropriate)
029     * 
030     * @author Philip Ogren
031     */
032    public interface XWriterFileNamer {
033            /**
034             * Get file name to be used for the file that it is written by {@link XWriter}.
035             * 
036             * @param jCas
037             *            a file namer should be able to generate a name from a given JCas to be used for
038             *            creating a file name
039             * @return a name for the file to be used for creating a file name Do not specify the full path
040             *         and do not specify the suffix (.xmi or .xcas will be automatically appended as
041             *         appropriate)
042             */
043            public String nameFile(JCas jCas);
044    }
