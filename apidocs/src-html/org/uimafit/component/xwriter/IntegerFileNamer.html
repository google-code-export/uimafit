018    package org.uimafit.component.xwriter;
019    
020    import org.apache.uima.UimaContext;
021    import org.apache.uima.jcas.JCas;
022    import org.apache.uima.resource.ResourceInitializationException;
023    import org.uimafit.component.initialize.ConfigurationParameterInitializer;
024    import org.uimafit.descriptor.ConfigurationParameter;
025    import org.uimafit.factory.ConfigurationParameterFactory;
026    import org.uimafit.factory.initializable.Initializable;
027    
028    /**
029     * This is a very simple implementation of XWriterFileNamer that generates file names based on a
030     * prefix string and a incrementing counter.
031     * 
032     * @author Philip Ogren
033     */
034    
035    public class IntegerFileNamer implements XWriterFileNamer, Initializable {
036    
037            /**
038             * The parameter name for the configuration parameter that specifies a fixed prefix for all
039             * returned file names.
040             */
041            public static final String PARAM_PREFIX = ConfigurationParameterFactory
042                            .createConfigurationParameterName(IntegerFileNamer.class, "prefix");
043            @ConfigurationParameter(description = "specify a prefix that is prepended to all returned file names")
044            private String prefix = "";
045    
046            int i = 1;
047    
048            public String nameFile(JCas jCas) {
049                    return prefix + i++;
050            }
051    
052            public void initialize(UimaContext context) throws ResourceInitializationException {
053                    ConfigurationParameterInitializer.initialize(this, context);
054            }
055    }
