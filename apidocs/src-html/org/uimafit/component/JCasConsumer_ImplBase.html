019    package org.uimafit.component;
020    
021    import org.apache.uima.UimaContext;
022    import org.apache.uima.resource.ResourceInitializationException;
023    import org.uimafit.component.initialize.ConfigurationParameterInitializer;
024    import org.uimafit.component.initialize.ExternalResourceInitializer;
025    import org.uimafit.descriptor.OperationalProperties;
026    import org.uimafit.util.ExtendedLogger;
027    
028    /**
029     * Base class for JCas consumers (actually a
030     * {@link org.apache.uima.analysis_component.JCasAnnotator_ImplBase}) which initializes itself based
031     * on annotations.
032     *
033     * @author Richard Eckart de Castilho
034     */
035    @OperationalProperties(multipleDeploymentAllowed = false)
036    public abstract class JCasConsumer_ImplBase extends
037                    org.apache.uima.analysis_component.JCasAnnotator_ImplBase {
038            private ExtendedLogger logger;
039            
040            public ExtendedLogger getLogger() {
041                    if (logger == null) {
042                            logger = new ExtendedLogger(getContext());
043                    }
044                    return logger;
045            }
046            
047            @Override
048            public void initialize(final UimaContext context) throws ResourceInitializationException {
049                    super.initialize(context);
050                    ConfigurationParameterInitializer.initialize(this, context);
051                    ExternalResourceInitializer.initialize(context, this);
052            }
053    }
