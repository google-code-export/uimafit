019    package org.uimafit.descriptor;
020    
021    import java.lang.annotation.ElementType;
022    import java.lang.annotation.Retention;
023    import java.lang.annotation.RetentionPolicy;
024    import java.lang.annotation.Target;
025    
026    import org.apache.uima.resource.metadata.FsIndexKeyDescription;
027    import org.uimafit.factory.FsIndexFactory;
028    
029    /**
030     * @author Richard Eckart de Castilho
031     * @see FsIndexKeyDescription
032     * @see FsIndexFactory
033     */
034    @Retention(RetentionPolicy.RUNTIME)
035    @Target(ElementType.TYPE)
036    public @interface FsIndexKey {
037            /**
038             * @see FsIndexKeyDescription#STANDARD_COMPARE
039             */
040            public static final int STANDARD_COMPARE = FsIndexKeyDescription.STANDARD_COMPARE;
041    
042            /**
043             * @see FsIndexKeyDescription#REVERSE_STANDARD_COMPARE
044             */
045            public static final int REVERSE_STANDARD_COMPARE = FsIndexKeyDescription.REVERSE_STANDARD_COMPARE;
046    
047            /**
048             * @see FsIndexKeyDescription#getFeatureName()
049             */
050            String featureName();
051    
052            /**
053             * @see FsIndexKeyDescription#getComparator()
054             */
055            int comparator() default STANDARD_COMPARE;
056    }
