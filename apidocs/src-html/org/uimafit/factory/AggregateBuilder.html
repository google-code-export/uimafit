018    package org.uimafit.factory;
019    
020    import java.util.ArrayList;
021    import java.util.Arrays;
022    import java.util.List;
023    
024    import org.apache.uima.analysis_engine.AnalysisEngine;
025    import org.apache.uima.analysis_engine.AnalysisEngineDescription;
026    import org.apache.uima.analysis_engine.metadata.SofaMapping;
027    import org.apache.uima.flow.FlowControllerDescription;
028    import org.apache.uima.resource.ResourceInitializationException;
029    import org.apache.uima.resource.metadata.TypePriorities;
030    import org.apache.uima.resource.metadata.TypeSystemDescription;
031    
032    /**
033     * This builder makes it easier to create an aggregate analysis engine. A typical use-case would
034     * involve initializing this builder with your preferred type system and type priorities (the latter
035     * may be null). This is followed by adding analysis engine descriptions one at a time until done.
036     * This makes it easy to have runtime decisions determine how the aggregate engine should be built.
037     * Finally, one of the create methods are called and an AnalysisEngine or AnalysisEngineDescription
038     * is returned.
039     *
040     * <p>This is an example taken from our test cases:</p>
041     *
042     * <p><blockquote><pre>
043     * import static org.uimafit.factory.AnalysisEngineFactory.createPrimitiveDescription;
044     *
045     * AggregateBuilder builder = new AggregateBuilder();
046     * builder.add(createPrimitiveDescription(Annotator1.class, typeSystemDescription),
047     *     ViewNames.PARENTHESES_VIEW, "A");
048     * builder.add(createPrimitiveDescription(Annotator2.class, typeSystemDescription),
049     *     ViewNames.SORTED_VIEW, "B",
050     *     ViewNames.SORTED_PARENTHESES_VIEW, "C",
051     *     ViewNames.PARENTHESES_VIEW, "A");
052     * builder.add(createPrimitiveDescription(Annotator3.class, typeSystemDescription),
053     *     ViewNames.INITIAL_VIEW, "B");
054     * AnalysisEngine aggregateEngine = builder.createAggregate();
055     * </pre></blockquote></p>
056     *
057     * @author Philip Ogren
058     */
059    public class AggregateBuilder {
060    
061            List<String> componentNames = new ArrayList<String>();
062    
063            List<SofaMapping> sofaMappings = new ArrayList<SofaMapping>();
064    
065            List<AnalysisEngineDescription> analysisEngineDescriptions = new ArrayList<AnalysisEngineDescription>();
066    
067            TypeSystemDescription typeSystemDescription;
068    
069            TypePriorities typePriorities;
070    
071            FlowControllerDescription flowControllerDescription;
072    
073            /**
074             * The default no-args constructor calls
075             * {@link AggregateBuilder#AggregateBuilder(TypeSystemDescription, TypePriorities, FlowControllerDescription)}
076             * with null-valued args.
077             */
078            public AggregateBuilder() {
079                    this(null, null, null);
080            }
081    
082            /**
083             * Instantiate an AggregateBuilder with a given type system, type priorities, and flow
084             * controller. Generally, speaking it suffices to use the no arguments constructor
085             *
086             * @param typeSystemDescription
087             *            this can be instantiated using {@link TypeSystemDescriptionFactory}
088             * @param typePriorities
089             *            this can be instantiated using {@link TypePrioritiesFactory}
090             * @param flowControllerDescription
091             *            this can be instantiated using {@link FlowControllerFactory}
092             */
093            public AggregateBuilder(TypeSystemDescription typeSystemDescription,
094                            TypePriorities typePriorities, FlowControllerDescription flowControllerDescription) {
095                    this.typeSystemDescription = typeSystemDescription;
096                    this.typePriorities = typePriorities;
097                    this.flowControllerDescription = flowControllerDescription;
098            }
099    
100            /**
101             * This method simply calls {@link #add(String, AnalysisEngineDescription, String...)} using the
102             * result of {@link AnalysisEngineDescription#getAnnotatorImplementationName()} for the
103             * component name
104             *
105             * @param aed
106             * @param viewNames
107             * @return the name of the component generated for the {@link AnalysisEngineDescription}
108             */
109            public String add(AnalysisEngineDescription aed, String... viewNames) {
110                    String componentName = aed.getAnalysisEngineMetaData().getName();
111                    if (componentName == null || componentName.equals("")) {
112                            if (aed.isPrimitive()) {
113                                    componentName = aed.getAnnotatorImplementationName();
114                            }
115                            else {
116                                    componentName = "aggregate";
117                            }
118                    }
119                    if (componentNames.contains(componentName)) {
120                            componentName = componentName + "." + (componentNames.size() + 1);
121                    }
122                    add(componentName, aed, viewNames);
123                    return componentName;
124            }
125    
126            /**
127             * @param componentName
128             *            the name of the component to add
129             * @param aed
130             *            an analysis engine description to add to the aggregate analysis engine
131             * @param viewNames
132             *            pairs of view names corresponding to a componentSofaName followed by the
133             *            aggregateSofaName that it is mapped to. An even number of names must be passed in
134             *            or else an IllegalArgumentException will be thrown. See
135             *            {@link SofaMappingFactory#createSofaMapping(String, String, String)}
136             */
137            public void add(String componentName, AnalysisEngineDescription aed, String... viewNames) {
138                    if (componentNames.contains(componentName)) {
139                            throw new IllegalArgumentException("the component name '" + componentName
140                                            + "' has already been used for another added analysis engine description.");
141                    }
142                    if (viewNames != null && viewNames.length % 2 != 0) {
143                            throw new IllegalArgumentException(
144                                            "an even number of view names is required (as "
145                                                            + "component view name, aggregate view name pairs) for the AggregateBuilder.add "
146                                                            + "method. " + viewNames.length + " view names passed: "
147                                                            + Arrays.asList(viewNames));
148                    }
149    
150                    analysisEngineDescriptions.add(aed);
151                    componentNames.add(componentName);
152    
153                    if (viewNames != null) {
154                            for (int i = 0; i < viewNames.length; i += 2) {
155                                    sofaMappings.add(SofaMappingFactory.createSofaMapping(componentName, viewNames[i],
156                                                    viewNames[i + 1]));
157                            }
158                    }
159            }
160    
161            /**
162             * Provide a sofa mapping for a component from the component's view to the aggregate view.
163             *
164             * @param componentName
165             *            the name of the component
166             * @param componentViewName
167             *            the name of the component view
168             * @param aggregateViewName
169             *            the name of the aggregate view to map the component view to.
170             */
171            public void addSofaMapping(String componentName, String componentViewName,
172                            String aggregateViewName) {
173                    if (componentNames.contains(componentName)) {
174                            sofaMappings.add(SofaMappingFactory.createSofaMapping(componentName, componentViewName,
175                                            aggregateViewName));
176                    }
177                    else {
178                            throw new IllegalArgumentException("No component with the name '" + componentName
179                                            + "' has been added to this builder.  Sofa mappings may only be added for "
180                                            + "components that have been added to this builder. ");
181                    }
182            }
183    
184            /**
185             * Set the flow controller description of the aggregate engine created by this builder.
186             *
187             * @param flowControllerDescription
188             *            see {@link FlowControllerFactory}
189             */
190            public void setFlowControllerDescription(FlowControllerDescription flowControllerDescription) {
191                    this.flowControllerDescription = flowControllerDescription;
192            }
193    
194            /**
195             * This method simply delegates to
196             * {@link AnalysisEngineFactory#createAggregate(List, TypeSystemDescription, TypePriorities, SofaMapping[], Object...)}
197             * with the data collected by this builder.
198             *
199             * @return an aggregate analysis engine
200             * @throws ResourceInitializationException
201             */
202            public AnalysisEngine createAggregate() throws ResourceInitializationException {
203                    return AnalysisEngineFactory.createAggregate(analysisEngineDescriptions, componentNames,
204                                    typeSystemDescription, typePriorities,
205                                    sofaMappings.toArray(new SofaMapping[sofaMappings.size()]),
206                                    flowControllerDescription);
207            }
208    
209            /**
210             * This method simply delegates to
211             * {@link AnalysisEngineFactory#createAggregateDescription(List, TypeSystemDescription, TypePriorities, SofaMapping[], Object...)}
212             * with the data collected by this builder.
213             *
214             * @return a description of an aggregate analysis engine
215             * @throws ResourceInitializationException
216             */
217            public AnalysisEngineDescription createAggregateDescription()
218                            throws ResourceInitializationException {
219                    return AnalysisEngineFactory.createAggregateDescription(analysisEngineDescriptions,
220                                    componentNames, typeSystemDescription, typePriorities,
221                                    sofaMappings.toArray(new SofaMapping[sofaMappings.size()]),
222                                    flowControllerDescription);
223            }
224    }
