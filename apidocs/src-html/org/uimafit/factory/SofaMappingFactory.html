017    package org.uimafit.factory;
018    
019    import org.apache.uima.analysis_component.AnalysisComponent;
020    import org.apache.uima.analysis_engine.metadata.SofaMapping;
021    import org.apache.uima.analysis_engine.metadata.impl.SofaMapping_impl;
022    
023    /**
024     * @author Philip Ogren
025     */
026    public final class SofaMappingFactory {
027            private SofaMappingFactory() {
028                    // This class is not meant to be instantiated
029            }
030    
031            /**
032             * This method creates a sofa mapping which is useful for mapping view names in individual
033             * components used in aggregate analysis engines to the view names used by the aggregate.
034             * 
035             * WARNING: in version 0.9.12 the ordering of the parameters was changed! The order used to be
036             * aggregateSofaName, componentKey, componentSofaName. This was changed because it seemed an
037             * unnatural ordering.
038             * 
039             * @param componentKey
040             *            the key/name used by the aggregate analysis engine for the component whose view is
041             *            being mapped.
042             * @param componentSofaName
043             *            the sofa name used by the the component
044             * @param aggregateSofaName
045             *            the view name that the component name is mapped to and used by the aggregate
046             *            analysis engine
047             * @return a sofa mapping with the componentSofaName mapped to the aggregateSofaName
048             */
049            public static SofaMapping createSofaMapping(String componentKey, String componentSofaName,
050                            String aggregateSofaName) {
051    
052                    SofaMapping sofaMapping = new SofaMapping_impl();
053                    sofaMapping.setComponentKey(componentKey);
054                    sofaMapping.setComponentSofaName(componentSofaName);
055                    sofaMapping.setAggregateSofaName(aggregateSofaName);
056                    return sofaMapping;
057            }
058    
059            /**
060             * create a sofa mapping using the component class rather than the component name
061             * 
062             * @param componentClass
063             * @param componentSofaName
064             * @param aggregateSofaName
065             * @return
066             */
067            public static SofaMapping createSofaMapping(Class<? extends AnalysisComponent> componentClass,
068                            String componentSofaName, String aggregateSofaName) {
069    
070                    return createSofaMapping(componentClass.getName(), componentSofaName, aggregateSofaName);
071            }
072    
073    }
