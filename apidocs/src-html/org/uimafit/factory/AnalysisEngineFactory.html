076             * of configuration parameters.

078             * @param descriptorName
079             *            The fully qualified, Java-style, dotted name of the XML descriptor file.
080             * @param configurationData
081             *            Any additional configuration parameters to be set. These should be supplied as
082             *            (name, value) pairs, so there should always be an even number of parameters.
083             * @return The AnalysisEngine created from the XML descriptor and the configuration parameters.
084             * @throws IOException
085             * @throws UIMAException
086             */

088                            Object... configurationData) throws UIMAException, IOException {
089                    AnalysisEngineDescription aed = createAnalysisEngineDescription(descriptorName,
090                                    configurationData);
091                    return UIMAFramework.produceAnalysisEngine(aed);
092            }

095             * Provides a way to create an AnalysisEngineDescription using a descriptor file referenced by
096             * name
097             *
098             * @param descriptorName
099             * @param configurationData
100             *            should consist of name value pairs. Will override configuration parameter settings
101             *            in the descriptor file
102             * @return
103             * @throws UIMAException
104             * @throws IOException
105             */
106            public static AnalysisEngineDescription createAnalysisEngineDescription(String descriptorName,
107                            Object... configurationData) throws UIMAException, IOException {
108                    Import_impl imprt = new Import_impl();
109                    imprt.setName(descriptorName);
110                    URL url = imprt.findAbsoluteUrl(UIMAFramework.newDefaultResourceManager());
111                    ResourceSpecifier specifier = ResourceCreationSpecifierFactory
112                                    .createResourceCreationSpecifier(url, configurationData);
113                    return (AnalysisEngineDescription) specifier;
114            }

289            /**
290             * @param componentClass
291             * @param typeSystem
292             * @param typePriorities
293             * @param capabilities
294             * @param configurationParameters
295             * @param configurationValues
296             * @return
297             * @throws ResourceInitializationException
298             */
299            public static AnalysisEngineDescription createPrimitiveDescription(
300                            Class<? extends AnalysisComponent> componentClass, TypeSystemDescription typeSystem,
301                            TypePriorities typePriorities, FsIndexCollection indexes, Capability[] capabilities,
302                            ConfigurationParameter[] configurationParameters, Object[] configurationValues)
303                            throws ResourceInitializationException {
304    
305                    // create the descriptor and set configuration parameters
306                    AnalysisEngineDescription desc = new AnalysisEngineDescription_impl();
307                    desc.setFrameworkImplementation(Constants.JAVA_FRAMEWORK_NAME);
308                    desc.setPrimitive(true);
309                    desc.setAnnotatorImplementationName(componentClass.getName());
310                    org.uimafit.descriptor.OperationalProperties componentAnno = ReflectionUtil
311                                    .getInheritableAnnotation(org.uimafit.descriptor.OperationalProperties.class,
312                                                    componentClass);
313                    if (componentAnno != null) {
314                            OperationalProperties op = desc.getAnalysisEngineMetaData().getOperationalProperties();
315                            op.setMultipleDeploymentAllowed(componentAnno.multipleDeploymentAllowed());
316                            op.setModifiesCas(componentAnno.modifiesCas());
317                            op.setOutputsNewCASes(componentAnno.outputsNewCases());
318                    }
319                    else {
320                            OperationalProperties op = desc.getAnalysisEngineMetaData().getOperationalProperties();
321                            op.setMultipleDeploymentAllowed(org.uimafit.descriptor.OperationalProperties.MULTIPLE_DEPLOYMENT_ALLOWED_DEFAULT);
322                            op.setModifiesCas(org.uimafit.descriptor.OperationalProperties.MODIFIES_CAS_DEFAULT);
323                            op.setOutputsNewCASes(org.uimafit.descriptor.OperationalProperties.OUTPUTS_NEW_CASES_DEFAULT);
324                    }
325    
326                    AnalysisEngineMetaData meta = desc.getAnalysisEngineMetaData();
327                    meta.setName(componentClass.getName());
328    
329                    if(componentClass.getPackage() != null){
330                            meta.setVendor(componentClass.getPackage().getName());
331                    }
332                    meta.setDescription("Descriptor automatically generated by uimaFIT");
333                    meta.setVersion("unknown");
334    
335                    // Extract external resource dependencies
336                    Collection<ExternalResourceDependency> deps = ExternalResourceInitializer
337                                    .getResourceDeclarations(componentClass).values();
338                    desc.setExternalResourceDependencies(deps.toArray(new ExternalResourceDependency[deps
339                                    .size()]));
340    
341                    ConfigurationData reflectedConfigurationData = ConfigurationParameterFactory
342                                    .createConfigurationData(componentClass);
343                    ResourceCreationSpecifierFactory.setConfigurationParameters(desc,
344                                    reflectedConfigurationData.configurationParameters,
345                                    reflectedConfigurationData.configurationValues);
346                    if (configurationParameters != null) {
347                            ResourceCreationSpecifierFactory.setConfigurationParameters(desc,
348                                            configurationParameters, configurationValues);
349                    }
350    
351                    // set the type system
352                    if (typeSystem != null) {
353                            desc.getAnalysisEngineMetaData().setTypeSystem(typeSystem);
354                    }
355    
356                    if (typePriorities != null) {
357                            desc.getAnalysisEngineMetaData().setTypePriorities(typePriorities);
358                    }
359    
360                    // set indexes from the argument to this call or from the annotation present in the
361                    // component if the argument is null
362                    if (indexes != null) {
363                            desc.getAnalysisEngineMetaData().setFsIndexCollection(indexes);
364                    }
365                    else {
366                            desc.getAnalysisEngineMetaData().setFsIndexCollection(
367                                            FsIndexFactory.createFsIndexCollection(componentClass));
368                    }
369    
370                    // set capabilities from the argument to this call or from the annotation present in the
371                    // component if the argument is null
372                    if (capabilities != null) {
373                            desc.getAnalysisEngineMetaData().setCapabilities(capabilities);
374                    }
375                    else {
376                            Capability capability = CapabilityFactory.createCapability(componentClass);
377                            if (capability != null) {
378                                    desc.getAnalysisEngineMetaData().setCapabilities(new Capability[] { capability });
379                            }
380                    }
381    
382                    return desc;
383            }

422            /**
423             * @param desc
424             * @param configurationData
425             * @return
426             * @throws ResourceInitializationException
427             */
428            public static AnalysisEngine createPrimitive(AnalysisEngineDescription desc,
429                            Object... configurationData) throws ResourceInitializationException {
430                    if (configurationData != null) {
431                            ConfigurationData cdata = ConfigurationParameterFactory
432                                            .createConfigurationData(configurationData);
433                            ConfigurationParameter[] configurationParameters = cdata.configurationParameters;
434                            Object[] configurationValues = cdata.configurationValues;
435                            ResourceCreationSpecifierFactory.setConfigurationParameters(desc,
436                                            configurationParameters, configurationValues);
437                            ResourceMetaData metaData = desc.getMetaData();
438                            ConfigurationParameterSettings paramSettings = metaData
439                                            .getConfigurationParameterSettings();
440                            Map<String, Object> additionalParameters = new HashMap<String, Object>();
441                            additionalParameters.put(AnalysisEngine.PARAM_CONFIG_PARAM_SETTINGS, paramSettings);
442    
443                            return UIMAFramework.produceAnalysisEngine(desc, null, additionalParameters);
444                    }
445                    else {
446                            return UIMAFramework.produceAnalysisEngine(desc, null, null);
447                    }
448            }

669            public static AnalysisEngineDescription createAggregateDescription(
670                            List<AnalysisEngineDescription> analysisEngineDescriptions,
671                            List<String> componentNames, TypeSystemDescription typeSystem,
672                            TypePriorities typePriorities, SofaMapping[] sofaMappings,
673                            FlowControllerDescription flowControllerDescription)
674                            throws ResourceInitializationException {
675    
676                    // create the descriptor and set configuration parameters
677                    AnalysisEngineDescription desc = new AnalysisEngineDescription_impl();
678                    desc.setFrameworkImplementation(Constants.JAVA_FRAMEWORK_NAME);
679                    desc.setPrimitive(false);
680    
681                    // if any of the aggregated analysis engines does not allow multiple
682                    // deployment, then the
683                    // aggregate engine may also not be multiply deployed
684                    boolean allowMultipleDeploy = true;
685                    for (AnalysisEngineDescription d : analysisEngineDescriptions) {
686                            allowMultipleDeploy &= d.getAnalysisEngineMetaData().getOperationalProperties()
687                                            .isMultipleDeploymentAllowed();
688                    }
689                    desc.getAnalysisEngineMetaData().getOperationalProperties()
690                                    .setMultipleDeploymentAllowed(allowMultipleDeploy);
691    
692                    List<String> flowNames = new ArrayList<String>();
693    
694                    for (int i = 0; i < analysisEngineDescriptions.size(); i++) {
695                            AnalysisEngineDescription aed = analysisEngineDescriptions.get(i);
696                            String componentName = componentNames.get(i);
697                            desc.getDelegateAnalysisEngineSpecifiersWithImports().put(componentName, aed);
698                            flowNames.add(componentName);
699                    }
700    
701                    if (flowControllerDescription != null) {
702                            FlowControllerDeclaration flowControllerDeclaration = new FlowControllerDeclaration_impl();
703                            flowControllerDeclaration.setSpecifier(flowControllerDescription);
704                            desc.setFlowControllerDeclaration(flowControllerDeclaration);
705                    }
706    
707                    FixedFlow fixedFlow = new FixedFlow_impl();
708                    fixedFlow.setFixedFlow(flowNames.toArray(new String[flowNames.size()]));
709                    desc.getAnalysisEngineMetaData().setFlowConstraints(fixedFlow);
710    
711                    if (typePriorities != null) {
712                            desc.getAnalysisEngineMetaData().setTypePriorities(typePriorities);
713                    }
714    
715                    if (sofaMappings != null) {
716                            desc.setSofaMappings(sofaMappings);
717                    }
718    
719                    return desc;
720            }

722            /**
723             * Creates an AnalysisEngine from the given descriptor, and uses the engine to process the file
724             * or text.
725             *
726             * @param descriptorFileName
727             *            The fully qualified, Java-style, dotted name of the XML descriptor file.
728             * @param fileNameOrText
729             *            Either the path of a file to be loaded, or a string to use as the text. If the
730             *            string given is not a valid path in the file system, it will be assumed to be
731             *            text.
732             * @return A JCas object containing the processed document.
733             * @throws IOException
734             * @throws UIMAException
735             */
736            public static JCas process(String descriptorFileName, String fileNameOrText)
737                            throws IOException, UIMAException {
738                    AnalysisEngine engine = createAnalysisEngine(descriptorFileName);
739                    JCas jCas = process(engine, fileNameOrText);
740                    engine.collectionProcessComplete();
741                    return jCas;
742            }

744            /**
745             * Processes the file or text with the given AnalysisEngine.
746             *
747             * @param analysisEngine
748             *            The AnalysisEngine object to process the text.
749             * @param fileNameOrText
750             *            Either the path of a file to be loaded, or a string to use as the text. If the
751             *            string given is not a valid path in the file system, it will be assumed to be
752             *            text.
753             * @return A JCas object containing the processed document.
754             * @throws IOException
755             * @throws UIMAException
756             */
757            public static JCas process(AnalysisEngine analysisEngine, String fileNameOrText)
758                            throws IOException, UIMAException {
759                    JCas jCas = analysisEngine.newJCas();
760                    process(jCas, analysisEngine, fileNameOrText);
761                    return jCas;
762            }

764            /**
765             * Provides a convenience method for running an AnalysisEngine over some text with a given JCas.
766             *
767             * @param jCas
768             * @param analysisEngine
769             * @param fileNameOrText
770             * @throws IOException
771             * @throws UIMAException
772             */
773            public static void process(JCas jCas, AnalysisEngine analysisEngine, String fileNameOrText)
774                            throws IOException, UIMAException {
775                    File textFile = new File(fileNameOrText);
776                    String text;
777                    if (textFile.exists()) {
778                            text = FileUtils.file2String(textFile);
779                    }
780                    else {
781                            text = fileNameOrText;
782                    }
783    
784                    jCas.setDocumentText(text);
785                    analysisEngine.process(jCas);
786            }
