081             * of configuration parameters.

083             * @param descriptorName
084             *            The fully qualified, Java-style, dotted name of the XML descriptor file.
085             * @param configurationData
086             *            Any additional configuration parameters to be set. These should be supplied as
087             *            (name, value) pairs, so there should always be an even number of parameters.
088             * @return The AnalysisEngine created from the XML descriptor and the configuration parameters.
089             * @throws IOException
090             * @throws UIMAException
091             */

093                            Object... configurationData) throws UIMAException, IOException {
094                    AnalysisEngineDescription aed = createAnalysisEngineDescription(descriptorName,
095                                    configurationData);
096                    return UIMAFramework.produceAnalysisEngine(aed);
097            }

100             * Provides a way to create an AnalysisEngineDescription using a descriptor file referenced by
101             * name
102             *
103             * @param descriptorName
104             * @param configurationData
105             *            should consist of name value pairs. Will override configuration parameter settings
106             *            in the descriptor file
107             * @return
108             * @throws UIMAException
109             * @throws IOException
110             */
111            public static AnalysisEngineDescription createAnalysisEngineDescription(String descriptorName,
112                            Object... configurationData) throws UIMAException, IOException {
113                    Import_impl imprt = new Import_impl();
114                    imprt.setName(descriptorName);
115                    URL url = imprt.findAbsoluteUrl(UIMAFramework.newDefaultResourceManager());
116                    ResourceSpecifier specifier = ResourceCreationSpecifierFactory
117                                    .createResourceCreationSpecifier(url, configurationData);
118                    return (AnalysisEngineDescription) specifier;
119            }

312            /**
313             * @param componentClass
314             * @param typeSystem
315             * @param typePriorities
316             * @param capabilities
317             * @param configurationParameters
318             * @param configurationValues
319             * @return
320             * @throws ResourceInitializationException
321             */
322            public static AnalysisEngineDescription createPrimitiveDescription(
323                            Class<? extends AnalysisComponent> componentClass, TypeSystemDescription typeSystem,
324                            TypePriorities typePriorities, FsIndexCollection indexes, Capability[] capabilities,
325                            ConfigurationParameter[] configurationParameters, Object[] configurationValues,
326                            Map<String, ExternalResourceDescription> externalResources)
327                            throws ResourceInitializationException {
328    
329                    // create the descriptor and set configuration parameters
330                    AnalysisEngineDescription desc = UIMAFramework.getResourceSpecifierFactory()
331                                    .createAnalysisEngineDescription();
332                    desc.setFrameworkImplementation(Constants.JAVA_FRAMEWORK_NAME);
333                    desc.setPrimitive(true);
334                    desc.setAnnotatorImplementationName(componentClass.getName());
335                    org.uimafit.descriptor.OperationalProperties componentAnno = ReflectionUtil
336                                    .getInheritableAnnotation(org.uimafit.descriptor.OperationalProperties.class,
337                                                    componentClass);
338                    if (componentAnno != null) {
339                            OperationalProperties op = desc.getAnalysisEngineMetaData().getOperationalProperties();
340                            op.setMultipleDeploymentAllowed(componentAnno.multipleDeploymentAllowed());
341                            op.setModifiesCas(componentAnno.modifiesCas());
342                            op.setOutputsNewCASes(componentAnno.outputsNewCases());
343                    }
344                    else {
345                            OperationalProperties op = desc.getAnalysisEngineMetaData().getOperationalProperties();
346                            op.setMultipleDeploymentAllowed(MULTIPLE_DEPLOYMENT_ALLOWED_DEFAULT);
347                            op.setModifiesCas(MODIFIES_CAS_DEFAULT);
348                            op.setOutputsNewCASes(OUTPUTS_NEW_CASES_DEFAULT);
349                    }
350    
351                    AnalysisEngineMetaData meta = desc.getAnalysisEngineMetaData();
352                    meta.setName(componentClass.getName());
353    
354                    if(componentClass.getPackage() != null){
355                            meta.setVendor(componentClass.getPackage().getName());
356                    }
357                    meta.setDescription("Descriptor automatically generated by uimaFIT");
358                    meta.setVersion("unknown");
359    
360                    ConfigurationData reflectedConfigurationData = createConfigurationData(componentClass);
361                    ResourceCreationSpecifierFactory.setConfigurationParameters(desc,
362                                    reflectedConfigurationData.configurationParameters,
363                                    reflectedConfigurationData.configurationValues);
364                    if (configurationParameters != null) {
365                            ResourceCreationSpecifierFactory.setConfigurationParameters(desc,
366                                            configurationParameters, configurationValues);
367                    }
368    
369                    // set the type system
370                    if (typeSystem != null) {
371                            desc.getAnalysisEngineMetaData().setTypeSystem(typeSystem);
372                    }
373    
374                    if (typePriorities != null) {
375                            desc.getAnalysisEngineMetaData().setTypePriorities(typePriorities);
376                    }
377    
378                    // set indexes from the argument to this call or from the annotation present in the
379                    // component if the argument is null
380                    if (indexes != null) {
381                            desc.getAnalysisEngineMetaData().setFsIndexCollection(indexes);
382                    }
383                    else {
384                            desc.getAnalysisEngineMetaData().setFsIndexCollection(
385                                            FsIndexFactory.createFsIndexCollection(componentClass));
386                    }
387    
388                    // set capabilities from the argument to this call or from the annotation present in the
389                    // component if the argument is null
390                    if (capabilities != null) {
391                            desc.getAnalysisEngineMetaData().setCapabilities(capabilities);
392                    }
393                    else {
394                            Capability capability = CapabilityFactory.createCapability(componentClass);
395                            if (capability != null) {
396                                    desc.getAnalysisEngineMetaData().setCapabilities(new Capability[] { capability });
397                            }
398                    }
399    
400                    // Extract external resource dependencies
401                    Collection<ExternalResourceDependency> deps = ExternalResourceInitializer
402                                    .getResourceDeclarations(componentClass).values();
403                    desc.setExternalResourceDependencies(deps.toArray(new ExternalResourceDependency[deps
404                                    .size()]));
405    
406                    // Bind External Resources
407                    if (externalResources != null) {
408                            for (Entry<String, ExternalResourceDescription> e : externalResources.entrySet()) {
409                                    bindExternalResource(desc, e.getKey(), e.getValue());
410                            }
411                    }
412    
413                    return desc;
414            }

659                            throws ResourceInitializationException {

661                    int i = 0;
662                    for (AnalysisEngineDescription aed : analysisEngineDescriptions) {
663                            names[i] = aed.getImplementationName() + "-" + i;
664                            i++;
665                    }
666    
667                    return createAggregateDescription(asList(analysisEngineDescriptions), asList(names), null,
668                                    null, null, flowControllerDescription);
669            }
670    
671            /**
672             * A factory method for creating an aggregate description.
673             *
674             * @param analysisEngineDescriptions
675             * @param componentNames
676             * @param typeSystem
677             * @param typePriorities
678             * @param sofaMappings
679             * @param flowControllerDescription
680             * @return
681             * @throws ResourceInitializationException
682             */
683            public static AnalysisEngineDescription createAggregateDescription(
684                            List<AnalysisEngineDescription> analysisEngineDescriptions,
685                            List<String> componentNames, TypeSystemDescription typeSystem,
686                            TypePriorities typePriorities, SofaMapping[] sofaMappings,
687                            FlowControllerDescription flowControllerDescription)
688                            throws ResourceInitializationException {
689    
690                    // create the descriptor and set configuration parameters
691                    AnalysisEngineDescription desc = new AnalysisEngineDescription_impl();
692                    desc.setFrameworkImplementation(Constants.JAVA_FRAMEWORK_NAME);
693                    desc.setPrimitive(false);
694    
695                    // if any of the aggregated analysis engines does not allow multiple
696                    // deployment, then the
697                    // aggregate engine may also not be multiply deployed
698                    boolean allowMultipleDeploy = true;
699                    for (AnalysisEngineDescription d : analysisEngineDescriptions) {
700                            allowMultipleDeploy &= d.getAnalysisEngineMetaData().getOperationalProperties()
701                                            .isMultipleDeploymentAllowed();
702                    }
703                    desc.getAnalysisEngineMetaData().getOperationalProperties()
704                                    .setMultipleDeploymentAllowed(allowMultipleDeploy);
705    
706                    List<String> flowNames = new ArrayList<String>();
707    
708                    for (int i = 0; i < analysisEngineDescriptions.size(); i++) {
709                            AnalysisEngineDescription aed = analysisEngineDescriptions.get(i);
710                            String componentName = componentNames.get(i);
711                            desc.getDelegateAnalysisEngineSpecifiersWithImports().put(componentName, aed);
712                            flowNames.add(componentName);
713                    }
714    
715                    if (flowControllerDescription != null) {
716                            FlowControllerDeclaration flowControllerDeclaration = new FlowControllerDeclaration_impl();
717                            flowControllerDeclaration.setSpecifier(flowControllerDescription);
718                            desc.setFlowControllerDeclaration(flowControllerDeclaration);
719                    }
720    
721                    FixedFlow fixedFlow = new FixedFlow_impl();
722                    fixedFlow.setFixedFlow(flowNames.toArray(new String[flowNames.size()]));
723                    desc.getAnalysisEngineMetaData().setFlowConstraints(fixedFlow);
724    
725                    if (typePriorities != null) {
726                            desc.getAnalysisEngineMetaData().setTypePriorities(typePriorities);
727                    }
728    
729                    if (sofaMappings != null) {
730                            desc.setSofaMappings(sofaMappings);
731                    }
732    
733                    return desc;
734            }

736            /**
737             * Creates an AnalysisEngine from the given descriptor, and uses the engine to process the file
738             * or text.
739             *
740             * @param descriptorFileName
741             *            The fully qualified, Java-style, dotted name of the XML descriptor file.
742             * @param fileNameOrText
743             *            Either the path of a file to be loaded, or a string to use as the text. If the
744             *            string given is not a valid path in the file system, it will be assumed to be
745             *            text.
746             * @return A JCas object containing the processed document.
747             * @throws IOException
748             * @throws UIMAException
749             */
750            public static JCas process(String descriptorFileName, String fileNameOrText)
751                            throws IOException, UIMAException {
752                    AnalysisEngine engine = createAnalysisEngine(descriptorFileName);
753                    JCas jCas = process(engine, fileNameOrText);
754                    engine.collectionProcessComplete();
755                    return jCas;
756            }

758            /**
759             * Processes the file or text with the given AnalysisEngine.
760             *
761             * @param analysisEngine
762             *            The AnalysisEngine object to process the text.
763             * @param fileNameOrText
764             *            Either the path of a file to be loaded, or a string to use as the text. If the
765             *            string given is not a valid path in the file system, it will be assumed to be
766             *            text.
767             * @return A JCas object containing the processed document.
768             * @throws IOException
769             * @throws UIMAException
770             */
771            public static JCas process(AnalysisEngine analysisEngine, String fileNameOrText)
772                            throws IOException, UIMAException {
773                    JCas jCas = analysisEngine.newJCas();
774                    process(jCas, analysisEngine, fileNameOrText);
775                    return jCas;
776            }

778            /**
779             * Provides a convenience method for running an AnalysisEngine over some text with a given JCas.
780             *
781             * @param jCas
782             * @param analysisEngine
783             * @param fileNameOrText
784             * @throws IOException
785             * @throws UIMAException
786             */
787            public static void process(JCas jCas, AnalysisEngine analysisEngine, String fileNameOrText)
788                            throws IOException, UIMAException {
789                    File textFile = new File(fileNameOrText);
790                    String text;
791                    if (textFile.exists()) {
792                            text = FileUtils.file2String(textFile);
793                    }
794                    else {
795                            text = fileNameOrText;
796                    }
797    
798                    jCas.setDocumentText(text);
799                    analysisEngine.process(jCas);
800            }
