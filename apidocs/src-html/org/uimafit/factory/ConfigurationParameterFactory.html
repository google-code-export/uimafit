053             * A mapping from Java class names to UIMA configuration parameter type names. Used by
054             * setConfigurationParameters().
055             */
056            public static final Map<String, String> javaUimaTypeMap = new HashMap<String, String>();
057            static {
058                    javaUimaTypeMap.put(Boolean.class.getName(), ConfigurationParameter.TYPE_BOOLEAN);
059                    javaUimaTypeMap.put(Float.class.getName(), ConfigurationParameter.TYPE_FLOAT);
060                    javaUimaTypeMap.put(Double.class.getName(), ConfigurationParameter.TYPE_FLOAT);
061                    javaUimaTypeMap.put(Integer.class.getName(), ConfigurationParameter.TYPE_INTEGER);
062                    javaUimaTypeMap.put(String.class.getName(), ConfigurationParameter.TYPE_STRING);
063                    javaUimaTypeMap.put("boolean", ConfigurationParameter.TYPE_BOOLEAN);
064                    javaUimaTypeMap.put("float", ConfigurationParameter.TYPE_FLOAT);
065                    javaUimaTypeMap.put("double", ConfigurationParameter.TYPE_FLOAT);
066                    javaUimaTypeMap.put("int", ConfigurationParameter.TYPE_INTEGER);

072             * {@link org.uimafit.descriptor.ConfigurationParameter}.

078                    return field.isAnnotationPresent(org.uimafit.descriptor.ConfigurationParameter.class);

081            /**
082             * Determines the default value of an annotated configuration parameter. The returned value is
083             * not necessarily the value that the annotated member variable will be instantiated with in
084             * ConfigurationParameterInitializer which does extra work to convert the UIMA configuration
085             * parameter value to comply with the type of the member variable.
086             *
087             * @param field
088             * @return
089             */
090            public static Object getDefaultValue(Field field) {
091                    if (isConfigurationParameterField(field)) {
092                            org.uimafit.descriptor.ConfigurationParameter annotation = field
093                                            .getAnnotation(org.uimafit.descriptor.ConfigurationParameter.class);
094    
095                            String[] stringValue = annotation.defaultValue();
096                            if (stringValue.length == 1
097                                            && stringValue[0]
098                                                            .equals(org.uimafit.descriptor.ConfigurationParameter.NO_DEFAULT_VALUE)) {
099                                    return null;
100                            }
101    
102                            String valueType = getConfigurationParameterType(field);
103                            boolean isMultiValued = isMultiValued(field);
104    
105                            if (!isMultiValued) {
106                                    if (ConfigurationParameter.TYPE_BOOLEAN.equals(valueType)) {
107                                            return Boolean.parseBoolean(stringValue[0]);
108                                    }
109                                    else if (ConfigurationParameter.TYPE_FLOAT.equals(valueType)) {
110                                            return Float.parseFloat(stringValue[0]);
111                                    }
112                                    else if (ConfigurationParameter.TYPE_INTEGER.equals(valueType)) {
113                                            return Integer.parseInt(stringValue[0]);
114                                    }
115                                    else if (ConfigurationParameter.TYPE_STRING.equals(valueType)) {
116                                            return stringValue[0];
117                                    }
118                                    throw new UIMA_IllegalArgumentException(
119                                                    UIMA_IllegalArgumentException.METADATA_ATTRIBUTE_TYPE_MISMATCH,
120                                                    new Object[] { valueType, "type" });
121                            }
122                            else {
123                                    if (ConfigurationParameter.TYPE_BOOLEAN.equals(valueType)) {
124                                            Boolean[] returnValues = new Boolean[stringValue.length];
125                                            for (int i = 0; i < stringValue.length; i++) {
126                                                    returnValues[i] = Boolean.parseBoolean(stringValue[i]);
127                                            }
128                                            return returnValues;
129                                    }
130                                    else if (ConfigurationParameter.TYPE_FLOAT.equals(valueType)) {
131                                            Float[] returnValues = new Float[stringValue.length];
132                                            for (int i = 0; i < stringValue.length; i++) {
133                                                    returnValues[i] = Float.parseFloat(stringValue[i]);
134                                            }
135                                            return returnValues;
136                                    }
137                                    else if (ConfigurationParameter.TYPE_INTEGER.equals(valueType)) {
138                                            Integer[] returnValues = new Integer[stringValue.length];
139                                            for (int i = 0; i < stringValue.length; i++) {
140                                                    returnValues[i] = Integer.parseInt(stringValue[i]);
141                                            }
142                                            return returnValues;
143                                    }
144                                    else if (ConfigurationParameter.TYPE_STRING.equals(valueType)) {
145                                            return stringValue;
146                                    }
147                                    throw new UIMA_IllegalArgumentException(
148                                                    UIMA_IllegalArgumentException.METADATA_ATTRIBUTE_TYPE_MISMATCH,
149                                                    new Object[] { valueType, "type" });
150    
151                            }
152    
153                    }
154                    else {
155                            throw new IllegalArgumentException("field is not annotated with annotation of type "
156                                            + org.uimafit.descriptor.ConfigurationParameter.class.getName());
157                    }
158            }

160            private static String getConfigurationParameterType(Field field) {
161                    Class<?> parameterClass = field.getType();
162                    String parameterClassName;
163                    if (parameterClass.isArray()) {
164                            parameterClassName = parameterClass.getComponentType().getName();
165                    }
166                    else if (Collection.class.isAssignableFrom(parameterClass)) {
167                            ParameterizedType collectionType = (ParameterizedType) field.getGenericType();
168                            parameterClassName = ((Class<?>) (collectionType.getActualTypeArguments()[0]))
169                                            .getName();
170                    }
171                    else {
172                            parameterClassName = parameterClass.getName();
173                    }
174                    String parameterType = javaUimaTypeMap.get(parameterClassName);
175                    if (parameterType == null) {
176                            return ConfigurationParameter.TYPE_STRING;
177                    }
178                    return parameterType;
179            }

181            private static boolean isMultiValued(Field field) {
182                    Class<?> parameterClass = field.getType();
183                    if (parameterClass.isArray()) {
184                            return true;
185                    }
186                    else if (Collection.class.isAssignableFrom(parameterClass)) {
187                            return true;
188                    }
189                    return false;
190            }

192            /**
193             * This method generates the default name of a configuration parameter that is defined by an
194             * {@link org.uimafit.descriptor.ConfigurationParameter} annotation when no name is given
195             *
196             * @param field
197             * @return
198             */
199            public static String getConfigurationParameterName(Field field) {
200                    if (isConfigurationParameterField(field)) {
201                            org.uimafit.descriptor.ConfigurationParameter annotation = field
202                                            .getAnnotation(org.uimafit.descriptor.ConfigurationParameter.class);
203                            String name = annotation.name();
204                            if (name.equals(org.uimafit.descriptor.ConfigurationParameter.USE_FIELD_NAME)) {
205                                    name = field.getDeclaringClass().getName() + "." + field.getName();
206                            }
207                            return name;
208                    }
209                    return null;
210            }

212            /**
213             * This method provides a convenient way to generate a configuration parameter name for a member
214             * variable that is annotated with {@link org.uimafit.descriptor.ConfigurationParameter} and no
215             * name is provided in the annotation.
216             *
217             * @param clazz
218             * @param fieldName
219             * @return
220             * @throws RuntimeException
221             */
222            public static String createConfigurationParameterName(Class<?> clazz, String fieldName)
223                            throws IllegalArgumentException {
224                    try {
225                            return ConfigurationParameterFactory.getConfigurationParameterName(clazz
226                                            .getDeclaredField(fieldName));
227                    }
228                    catch (NoSuchFieldException e) {
229                            throw new IllegalArgumentException(e);
230                    }
231            }

233            /**
234             * A factory method for creating a ConfigurationParameter from a given field definition
235             *
236             * @param field
237             * @return
238             */
239            public static ConfigurationParameter createPrimitiveParameter(Field field) {
240                    if (isConfigurationParameterField(field)) {
241                            org.uimafit.descriptor.ConfigurationParameter annotation = field
242                                            .getAnnotation(org.uimafit.descriptor.ConfigurationParameter.class);
243                            String name = getConfigurationParameterName(field);
244                            boolean multiValued = isMultiValued(field);
245                            String parameterType = getConfigurationParameterType(field);
246                            return createPrimitiveParameter(name, parameterType, annotation.description(),
247                                            multiValued, annotation.mandatory());
248                    }
249                    else {
250                            throw new IllegalArgumentException("field is not annotated with annotation of type "
251                                            + org.uimafit.descriptor.ConfigurationParameter.class.getName());
252                    }
253            }

255            /**
256             * A factory method for creating a ConfigurationParameter object.
257             *
258             * @param name
259             * @param parameterClass
260             * @param parameterDescription
261             * @param isMandatory
262             * @return
263             */
264            public static ConfigurationParameter createPrimitiveParameter(String name,
265                            Class<?> parameterClass, String parameterDescription, boolean isMandatory) {
266                    String parameterClassName;
267                    if (parameterClass.isArray()) {
268                            parameterClassName = parameterClass.getComponentType().getName();
269                    }
270                    else {
271                            parameterClassName = parameterClass.getName();
272                    }
273    
274                    String parameterType = javaUimaTypeMap.get(parameterClassName);
275                    if (parameterType == null) {
276                            throw new IllegalArgumentException("While trying to set the parameter [" + name +
277                                            "] found the type [" + parameterClassName
278                                            + "] is not a valid parameter type in UIMA."
279                                            + " uimaFIT currently does not support using such types as parameter values.");
280                    }
281                    return createPrimitiveParameter(name, parameterType, parameterDescription,
282                                    parameterClass.isArray(), isMandatory);
283            }

286             * A factory method for creating a ConfigurationParameter object.
287             *
288             * @param name
289             * @param parameterType
290             * @param parameterDescription
291             * @param isMultiValued
292             * @param isMandatory
293             * @return
294             */
295            public static ConfigurationParameter createPrimitiveParameter(String name,
296                            String parameterType, String parameterDescription, boolean isMultiValued,
297                            boolean isMandatory) {
298                    ConfigurationParameter param = new ConfigurationParameter_impl();
299                    param.setName(name);
300                    param.setType(parameterType);
301                    param.setDescription(parameterDescription);
302                    param.setMultiValued(isMultiValued);
303                    param.setMandatory(isMandatory);
304                    return param;
305            }

307            /**
308             * This method converts configuration data provided as an array of objects and returns a
309             * {@link ConfigurationData} object
310             *
311             * @param configurationData
312             * @return
313             */
314            public static ConfigurationData createConfigurationData(Object... configurationData) {
315                    if (configurationData == null) {
316                            return new ConfigurationData(new ConfigurationParameter[0], new Object[0]);
317                    }
318                    
319                    ensureParametersComeInPairs(configurationData);
320    
321                    int numberOfParameters = configurationData.length / 2;
322                    List<ConfigurationParameter> configurationParameters = new ArrayList<ConfigurationParameter>();
323                    List<Object> configurationValues = new ArrayList<Object>();
324    
325                    for (int i = 0; i < numberOfParameters; i++) {
326                            String name = (String) configurationData[i * 2];
327                            Object value = configurationData[i * 2 + 1];
328    
329                            if (value == null || value instanceof ExternalResourceDescription) {
330                                    continue;
331                            }
332                            
333                            if (value.getClass().isArray()
334                                            && value.getClass().getComponentType().getName().equals("boolean")) {
335                                    value = ArrayUtils.toObject((boolean[]) value);
336                            }
337                            else if (value.getClass().isArray()
338                                            && value.getClass().getComponentType().getName().equals("int")) {
339                                    value = ArrayUtils.toObject((int[]) value);
340                            }
341                            else if (value.getClass().isArray()
342                                            && value.getClass().getComponentType().getName().equals("float")) {
343                                    value = ArrayUtils.toObject((float[]) value);
344                            }
345    
346                            ConfigurationParameter param = ConfigurationParameterFactory.createPrimitiveParameter(
347                                            name, value.getClass(), null, false);
348                            configurationParameters.add(param);
349                            configurationValues.add(value);
350                    }
351                    return new ConfigurationData(
352                                    configurationParameters.toArray(new ConfigurationParameter[configurationParameters
353                                                    .size()]), configurationValues.toArray());
354            }

356            /**
357             * This method creates configuration data for a given class definition using reflection and the
358             * configuration parameter annotation
359             *
360             * @param componentClass
361             * @return
362             */
363            public static ConfigurationData createConfigurationData(Class<?> componentClass) {
364                    List<ConfigurationParameter> configurationParameters = new ArrayList<ConfigurationParameter>();
365                    List<Object> configurationValues = new ArrayList<Object>();
366    
367                    for (Field field : ReflectionUtil.getFields(componentClass)) {
368                            if (ConfigurationParameterFactory.isConfigurationParameterField(field)) {
369                                    configurationParameters.add(ConfigurationParameterFactory
370                                                    .createPrimitiveParameter(field));
371                                    configurationValues.add(ConfigurationParameterFactory.getDefaultValue(field));
372                            }
373                    }
374    
375                    return new ConfigurationData(
376                                    configurationParameters.toArray(new ConfigurationParameter[configurationParameters
377                                                    .size()]), configurationValues.toArray(new Object[configurationValues
378                                                    .size()]));
379            }

480            /**
481             * Helper method to make sure configuration parameter lists have always pairs of name/values.
482             * 
483             * @param configurationData the configuration parameters.
484             */
485            static void ensureParametersComeInPairs(Object[] configurationData) {
486                    if (configurationData != null && configurationData.length % 2 != 0) {
487                            throw new IllegalArgumentException("Parameter arguments have to "
488                                            + "come in key/value pairs, but found odd number of " + "arguments ["
489                                            + configurationData.length + "]");
490                    }
491            }

493            /**
494             * Fetches the parameter settings from the given resource specifier.
495             * 
496             * @param spec a resource specifier.
497             * @return the parameter settings.
498             */
499            public static Map<String, Object> getParameterSettings(ResourceSpecifier spec)
500            {
501                    Map<String, Object> settings = new HashMap<String, Object>();
502                    if (spec instanceof CustomResourceSpecifier) {
503                            for (Parameter p : ((CustomResourceSpecifier) spec).getParameters()) {
504                                    settings.put(p.getName(), p.getValue());
505                            }
506                    }
507                    else if (spec instanceof ResourceCreationSpecifier) {
508                            for (NameValuePair p : ((ResourceCreationSpecifier) spec).getMetaData()
509                                            .getConfigurationParameterSettings().getParameterSettings())
510                            {
511                                    settings.put(p.getName(), p.getValue());
512                            }
513                    }
514                    else if (spec instanceof ConfigurableDataResourceSpecifier) {
515                            for (NameValuePair p : ((ResourceCreationSpecifier) spec).getMetaData()
516                                            .getConfigurationParameterSettings().getParameterSettings())
517                            {
518                                    settings.put(p.getName(), p.getValue());
519                            }
520                    }
521                    else {
522                            throw new IllegalClassException("Unsupported resource specifier class ["
523                                            + spec.getClass() + "]");
524                    }
525                    return settings;
526            }

528            /**
529             * Sets the specified parameter in the given resource specifier. If the specified is a
530             * {@link CustomResourceSpecifier} an exception is thrown if the parameter value not a String.
531             * 
532             * @param aSpec a resource specifier.
533             * @param name the parameter name.
534             * @param value the parameter value.
535             * @throws IllegalClassException if the value is not of a supported type for the given 
536             * specifier.
537             */
538            public static void setParameter(ResourceSpecifier aSpec, String name, Object value)
539            {
540                    if (aSpec instanceof CustomResourceSpecifier) {
541                            if (!(value instanceof String || value == null)) {
542                                    throw new IllegalClassException(String.class, value);
543                            }
544                            CustomResourceSpecifier spec = (CustomResourceSpecifier) aSpec;
545                            
546                            // If the parameter is already there, update it
547                            boolean found = false;
548                            for (Parameter p : spec.getParameters()) {
549                                    if (p.getName().equals(name)) {
550                                            p.setValue((String) value);
551                                            found = true;
552                                    }
553                            }
554                            
555                            // If the parameter is not there, add it
556                            if (!found) {
557                                    Parameter[] params = new Parameter[spec.getParameters().length + 1];
558                                    System.arraycopy(spec.getParameters(), 0, params, 0, spec.getParameters().length);
559                                    params[params.length-1] = new Parameter_impl();
560                                    params[params.length-1].setName(name);
561                                    params[params.length-1].setValue((String) value);
562                                    spec.setParameters(params);
563                            }
564                    }
565                    else if (aSpec instanceof ResourceCreationSpecifier) {
566                            ResourceMetaData md = ((ResourceCreationSpecifier) aSpec).getMetaData();
567                            
568                            if (md.getConfigurationParameterDeclarations().getConfigurationParameter(null, name) == null) {
569                                    throw new IllegalArgumentException("Cannot set undeclared parameter [" + name + "]");
570                            }
571                            
572                            md.getConfigurationParameterSettings().setParameterValue(name, value);
573                    }
574                    else if (aSpec instanceof ConfigurableDataResourceSpecifier) {
575                            ResourceMetaData md = ((ConfigurableDataResourceSpecifier) aSpec).getMetaData();
576                            
577                            if (md.getConfigurationParameterDeclarations().getConfigurationParameter(null, name) == null) {
578                                    throw new IllegalArgumentException("Cannot set undeclared parameter [" + name + "]");
579                            }
580    
581                            md.getConfigurationParameterSettings().setParameterValue(name, value);
582                    }
583                    else {
584                            throw new IllegalClassException("Unsupported resource specifier class ["
585                                            + aSpec.getClass() + "]");
586                    }
587            }

589            /**
590             * Check if the given parameter can be set on the provided specifier. Some specifier types 
591             * require parameters to be declared before they can be set.
592             */
593            public static boolean canParameterBeSet(ResourceSpecifier aSpec, String name)
594            {
595                    if (aSpec instanceof CustomResourceSpecifier) {
596                            return true;
597                    }
598                    else if (aSpec instanceof ResourceCreationSpecifier) {
599                            ResourceMetaData md = ((ResourceCreationSpecifier) aSpec).getMetaData();
600                            return md.getConfigurationParameterDeclarations().getConfigurationParameter(null, name) != null;
601                    }
602                    else if (aSpec instanceof ConfigurableDataResourceSpecifier) {
603                            ResourceMetaData md = ((ConfigurableDataResourceSpecifier) aSpec).getMetaData();
604                            return md.getConfigurationParameterDeclarations().getConfigurationParameter(null, name) != null;
605                    }
606                    else {
607                            throw new IllegalClassException("Unsupported resource specifier class ["
608                                            + aSpec.getClass() + "]");
609                    }               
610            }
