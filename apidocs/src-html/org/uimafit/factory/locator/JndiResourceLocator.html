019    package org.uimafit.factory.locator;
020    
021    import java.util.Map;
022    
023    import javax.naming.InitialContext;
024    import javax.naming.NamingException;
025    
026    import org.apache.uima.resource.ResourceInitializationException;
027    import org.apache.uima.resource.ResourceSpecifier;
028    import org.uimafit.component.Resource_ImplBase;
029    import org.uimafit.descriptor.ConfigurationParameter;
030    import org.uimafit.descriptor.ExternalResourceLocator;
031    
032    /**
033     * Locate an object via JNDI.
034     *
035     * @author Richard Eckart de Castilho
036     */
037    public class JndiResourceLocator extends Resource_ImplBase implements ExternalResourceLocator {
038            public static final String PARAM_NAME = "Name";
039            @ConfigurationParameter(name = PARAM_NAME, mandatory = true)
040            private String jndiName;
041    
042            private Object resource;
043    
044            @SuppressWarnings("rawtypes")
045            @Override
046            public boolean initialize(ResourceSpecifier aSpecifier, Map aAdditionalParams)
047                            throws ResourceInitializationException {
048                    if (!super.initialize(aSpecifier, aAdditionalParams)) {
049                            return false;
050                    }
051    
052                    try {
053                            InitialContext ctx = new InitialContext();
054                            resource = ctx.lookup(jndiName);
055                    }
056                    catch (NamingException e) {
057                            throw new ResourceInitializationException(e);
058                    }
059                    return true;
060            }
061    
062            public Object getResource() {
063                    return resource;
064            }
065    }
