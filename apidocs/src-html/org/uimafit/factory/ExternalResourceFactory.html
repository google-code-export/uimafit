106            /**
107             * Create an external resource description for a custom resource.
108             *
109             * @param aName
110             *            the name of the resource (the key).
111             * @param aInterface
112             *            the interface the resource should implement.
113             * @param aParams
114             *            parameters passed to the resource when it is created.
115             * @return the description.
116             * @see CustomResourceSpecifier
117             */
118            public static ExternalResourceDescription createExternalResourceDescription(final String aName,
119                            Class<? extends Resource> aInterface, Object... aParams) {
120                    ConfigurationParameterFactory.ensureParametersComeInPairs(aParams);
121    
122                    // Extract ExternalResourceDescriptions from configurationData
123                    List<ExternalResourceBinding> bindings = new ArrayList<ExternalResourceBinding>();
124                    List<ExternalResourceDescription> descs = new ArrayList<ExternalResourceDescription>();
125                    for (Entry<String, ExternalResourceDescription> res : extractExternalResourceParameters(
126                                    aParams).entrySet()) {
127                            bindings.add(createExternalResourceBinding(res.getKey(), res.getValue()));
128                            descs.add(res.getValue());
129                    }
130                    
131                    List<Parameter> params = new ArrayList<Parameter>();
132                    if (aParams != null) {
133                            for (int i = 0; i < aParams.length / 2; i++) {
134                                    if (aParams[i * 2 + 1] instanceof ExternalResourceDescription) {
135                                            continue;
136                                    }
137                                    
138                                    Parameter param = new Parameter_impl();
139                                    param.setName((String) aParams[i * 2]);
140                                    param.setValue((String) aParams[i * 2 + 1]);
141                                    params.add(param);
142                            }
143                    }
144    
145                    CustomResourceSpecifier spec = getResourceSpecifierFactory().createCustomResourceSpecifier();
146                    spec.setResourceClassName(aInterface.getName());
147                    spec.setParameters(params.toArray(new Parameter[params.size()]));
148    
149                    ExtendedExternalResourceDescription_impl extRes = new ExtendedExternalResourceDescription_impl();
150                    extRes.setName(aName);
151                    extRes.setResourceSpecifier(spec);
152                    extRes.setExternalResourceBindings(bindings);
153                    extRes.setExternalResources(descs);
154    
155                    return extRes;
156            }

223             *

225             *            the name of the resource (the key).
226             * @param aInterface
227             *            the interface the resource should implement.

238                    ConfigurationData cfg = ConfigurationParameterFactory.createConfigurationData(aParams);
239                    ResourceMetaData_impl meta = new ResourceMetaData_impl();
240    
241                    ConfigurationData reflectedConfigurationData = createConfigurationData(aInterface);
242                    ResourceCreationSpecifierFactory.setConfigurationParameters(meta,
243                                    reflectedConfigurationData.configurationParameters,
244                                    reflectedConfigurationData.configurationValues);
245                    ResourceCreationSpecifierFactory.setConfigurationParameters(meta,
246                                    cfg.configurationParameters, cfg.configurationValues);
247    
248                    ConfigurableDataResourceSpecifier_impl spec = new ConfigurableDataResourceSpecifier_impl();
249                    spec.setUrl(aUrl);
250                    spec.setMetaData(meta);
251    
252                    ExtendedExternalResourceDescription_impl extRes = new ExtendedExternalResourceDescription_impl();
253                    extRes.setName(aName);
254                    extRes.setResourceSpecifier(spec);
255                    extRes.setImplementationName(aInterface.getName());
256                    
257                    return extRes;

621            /**
622             * Create a new dependency for the specified resource and bind it. This method is helpful for
623             * UIMA components that do not use the uimaFIT {@link ExternalResource} annotation, because no
624             * external resource dependencies can be automatically generated by uimaFIT for such components.
625             *
626             * @param aDesc
627             *            a description.
628             * @param aKey
629             *            the key to bind to.
630             * @param aApi
631             *            the resource API.
632             * @throws InvalidXMLException
633             */
634            public static void createDependency(ResourceSpecifier aDesc, String aKey, Class<?> aApi)
635                            throws InvalidXMLException {
636                    ExternalResourceDependency[] deps = getExternalResourceDependencies(aDesc);
637                    if (deps == null) {
638                            deps = new ExternalResourceDependency[] {};
639                    }
640    
641                    // Check if the resource dependency is already present
642                    boolean found = false;
643                    for (ExternalResourceDependency dep : deps) {
644                            if (dep.getKey().equals(aKey)) {
645                                    found = true;
646                                    break;
647                            }
648                    }
649    
650                    // If not, create one
651                    if (!found) {
652                            setExternalResourceDependencies(aDesc, (ExternalResourceDependency[]) ArrayUtils.add(
653                                            deps, createExternalResourceDependency(aKey, aApi, false)));            }
654            }

657             * Convenience method to set the external resource dependencies on a resource specifier. 
658             * Unfortunately different methods need to be used for different sub-classes.
659             * 
660             * @throws IllegalArgumentException if the sub-class passed is not supported.
661             */

693            /**
694             * Create a new dependency for the specified resource and bind it. This method is helpful for
695             * UIMA components that do not use the uimaFIT {@link ExternalResource} annotation, because no
696             * external resource dependencies can be automatically generated by uimaFIT for such components.
697             *
698             * @param aDesc
699             *            a description.
700             * @param aKey
701             *            the key to bind to.
702             * @param aImpl
703             *            the resource implementation.
704             * @param aUrl
705             *            the resource URL.
706             * @param aParams
707             *            additional parameters supported by the resource.
708             * @throws InvalidXMLException
709             */
710            public static void createDependencyAndBind(AnalysisEngineDescription aDesc, String aKey,
711                            Class<? extends SharedResourceObject> aImpl, String aUrl, Object... aParams)
712                            throws InvalidXMLException {
713                    if (aDesc.getExternalResourceDependency(aKey) == null) {
714                            ExternalResourceDependency[] deps = aDesc.getExternalResourceDependencies();
715                            if (deps == null) {
716                                    deps = new ExternalResourceDependency[] {};
717                            }
718                            aDesc.setExternalResourceDependencies((ExternalResourceDependency[]) ArrayUtils.add(
719                                            deps, createExternalResourceDependency(aKey, aImpl, false)));
720                    }
721                    bindResource(aDesc, aKey, aImpl, aUrl, aParams);
722            }

724            /**
725             * Scan the given resource specifier for external resource dependencies and whenever a
726             * dependency with the given key is encountered, the given resource is bound to it.
727             *
728             * @param aDesc
729             *            a description.
730             * @param aKey
731             *            the key to bind to.
732             * @param aResDesc
733             *            the resource description.
734             * @throws InvalidXMLException
735             */
736            private static void bind(AnalysisEngineDescription aDesc, String aKey,
737                            ExternalResourceDescription aResDesc) throws InvalidXMLException {
738                    // Recursively address delegates
739                    if (!aDesc.isPrimitive()) {
740                            for (Object delegate : aDesc.getDelegateAnalysisEngineSpecifiers().values()) {
741                                    bindResource((ResourceSpecifier) delegate, aKey, aResDesc);
742                            }
743                    }
744    
745                    // Bind if necessary
746                    for (ExternalResourceDependency dep : aDesc.getExternalResourceDependencies()) {
747                            if (aKey.equals(dep.getKey())) {
748                                    bindExternalResource(aDesc, aKey, aResDesc);
749                            }
750                    }
751            }

753            /**
754             * Create a new external resource binding.
755             *
756             * @param aResMgrCfg
757             *            the resource manager to create the binding in.
758             * @param aBindTo
759             *            what key to bind to.
760             * @param aRes
761             *            the resource that should be bound.
762             */
763            public static void bindExternalResource(ResourceManagerConfiguration aResMgrCfg,
764                            String aBindTo, ExternalResourceDescription aRes) {
765                    // Create a map of all bindings
766                    Map<String, ExternalResourceBinding> bindings = new HashMap<String, ExternalResourceBinding>();
767                    for (ExternalResourceBinding b : aResMgrCfg.getExternalResourceBindings()) {
768                            bindings.put(b.getKey(), b);
769                    }
770                    
771                    // Create a map of all resources
772                    Map<String, ExternalResourceDescription> resources = new HashMap<String, ExternalResourceDescription>();
773                    for (ExternalResourceDescription r : aResMgrCfg.getExternalResources()) {
774                            resources.put(r.getName(), r);
775                    }
776                    
777                    // For the current resource, add resource and binding
778                    ExternalResourceBinding extResBind = createExternalResourceBinding(aBindTo, aRes);
779                    bindings.put(extResBind.getKey(), extResBind);
780                    resources.put(aRes.getName(), aRes);
781                    
782                    // Handle nested resources
783                    bindNestedResources(aRes, bindings, resources);
784                    
785                    // Commit everything to the resource manager configuration
786                    aResMgrCfg.setExternalResourceBindings(bindings.values().toArray(
787                                    new ExternalResourceBinding[bindings.size()]));
788                    aResMgrCfg.setExternalResources(resources.values().toArray(
789                                    new ExternalResourceDescription[resources.size()]));
790            }

793             * Create a new external resource binding.
794             * 
795             * @param aRes
796             *            the resource to bind to
797             * @param aBindTo
798             *            what key to bind to.
799             * @param aNestedRes
800             *            the resource that should be bound.
801             */
802            public static void bindExternalResource(ExternalResourceDescription aRes, String aBindTo,
803                            ExternalResourceDescription aNestedRes) {
804                    if (!(aRes instanceof ExtendedExternalResourceDescription_impl)) {
805                            throw new IllegalArgumentException(
806                                            "Nested resources are only supported on instances of [" + 
807                                            ExtendedExternalResourceDescription_impl.class.getName() + "] which" + 
808                                            "can be created with uimaFIT's createExternalResourceDescription() methods.");
809                    }
810    
811                    ExtendedExternalResourceDescription_impl extRes = (ExtendedExternalResourceDescription_impl) aRes;
812    
813                    // Create a map of all bindings
814                    Map<String, ExternalResourceBinding> bindings = new HashMap<String, ExternalResourceBinding>();
815                    for (ExternalResourceBinding b : extRes.getExternalResourceBindings()) {
816                            bindings.put(b.getKey(), b);
817                    }
818                    
819                    // Create a map of all resources
820                    Map<String, ExternalResourceDescription> resources = new HashMap<String, ExternalResourceDescription>();
821                    for (ExternalResourceDescription r : extRes.getExternalResources()) {
822                            resources.put(r.getName(), r);
823                    }
824                    
825                    // For the current resource, add resource and binding
826                    ExternalResourceBinding extResBind = createExternalResourceBinding(aBindTo, aNestedRes);
827                    bindings.put(extResBind.getKey(), extResBind);
828                    resources.put(aRes.getName(), aRes);
829                    
830                    // Handle nested resources
831                    bindNestedResources(aRes, bindings, resources);
832                    
833                    // Commit everything to the resource manager configuration
834                    extRes.setExternalResourceBindings(bindings.values());
835                    extRes.setExternalResources(resources.values());
836                    
837            }

839            /**
840             * Helper method to recursively bind resources bound to resources.
841             * 
842             * @param aRes resource.
843             * @param aBindings bindings already made.
844             * @param aResources resources already bound.
845             */
846            private static void bindNestedResources(ExternalResourceDescription aRes,
847                            Map<String, ExternalResourceBinding> aBindings,
848                            Map<String, ExternalResourceDescription> aResources)      {
849                    // Handle nested resources
850                    if (aRes instanceof ExtendedExternalResourceDescription_impl) {
851                            ExtendedExternalResourceDescription_impl extRes = (ExtendedExternalResourceDescription_impl) aRes;
852                            
853                            // Tell the external resource its name. This is needed in order to find the resources
854                            // bound to this resource later on. Set only if the resource supports this parameter.
855                            // Mind that supporting this parameter is mandatory for resource implementing 
856                            // ExternalResourceAware.
857                            if (canParameterBeSet(extRes.getResourceSpecifier(), PARAM_RESOURCE_NAME)) {
858                                    ConfigurationParameterFactory.setParameter(extRes.getResourceSpecifier(),
859                                                    PARAM_RESOURCE_NAME, aRes.getName());
860                            }
861                            
862                            // Create a map of all resources
863                            Map<String, ExternalResourceDescription> res = new HashMap<String, ExternalResourceDescription>();
864                            for (ExternalResourceDescription r : extRes.getExternalResources()) {
865                                    res.put(r.getName(), r);
866                            }
867                            
868                            // Bind nested resources
869                            for (ExternalResourceBinding b : extRes.getExternalResourceBindings()) {
870                                    // Avoid re-prefixing the resource name
871                                    String key = b.getKey();
872                                    if (!key.startsWith(aRes.getName() + PREFIX_SEPARATOR)) {
873                                            key = aRes.getName() + PREFIX_SEPARATOR + b.getKey();
874                                    }
875                                    // Avoid unnecessary binding and an infinite loop when a resource binds to itself
876                                    if (!aBindings.containsKey(key)) {
877                                            // Mark the current binding as processed so we do not recurse
878                                            aBindings.put(key, b);
879                                            ExternalResourceDescription nestedRes = res.get(b.getResourceName());
880                                            aResources.put(nestedRes.getName(), nestedRes);
881                                            bindNestedResources(nestedRes, aBindings, aResources);
882                                            // Set the proper key on the binding.
883                                            b.setKey(key);
884                                    }
885                            }
886                    }
887            }

952            /**
953             * Extracts the external resource from the configuration parameters and nulls out these
954             * parameters. Mind that the array passed to this method is modified by the method.
955             * 
956             * @param configurationData the configuration parameters.
957             * @return extRes the external resource parameters.
958             */
959            protected static Map<String, ExternalResourceDescription> extractExternalResourceParameters(
960                            final Object[] configurationData) {
961                    if (configurationData == null) {
962                            return Collections.emptyMap();
963                    }
964            
965                    Map<String, ExternalResourceDescription> extRes = new HashMap<String, ExternalResourceDescription>();
966                    for (int i = 0; i < configurationData.length - 1; i += 2) {
967                            String key = (String) configurationData[i];
968                            Object value = configurationData[i + 1];
969    
970                            // Store External Resource parameters separately
971                            if (value instanceof ExternalResourceDescription) {
972                                    ExternalResourceDescription description = (ExternalResourceDescription) value;
973                                    extRes.put(key, description);
974                            }
975                    }
976                    
977                    return extRes;
978            }
