017    package org.uimafit.examples.tutorial.ex1;
018    
019    import java.io.File;
020    import java.io.FileOutputStream;
021    
022    import org.apache.uima.analysis_engine.AnalysisEngineDescription;
023    import org.apache.uima.resource.ResourceInitializationException;
024    import org.apache.uima.resource.metadata.TypeSystemDescription;
025    import org.uimafit.factory.AnalysisEngineFactory;
026    import org.uimafit.factory.TypeSystemDescriptionFactory;
027    
028    /**
029     * This class provides a main method which shows how to generate an xml descriptor file using the
030     * RoomNumberAnnotator class definition. The resulting XML descriptor file is the same as the one
031     * provided in the uimaj-examples except that instead of building the file in parallel with the
032     * class definition, it is now built completely by using the class definition.
033     * 
034     * @author Philip Ogren
035     * 
036     */
037    public class RoomNumberAnnotatorDescriptor {
038    
039            public static AnalysisEngineDescription createDescriptor()
040                            throws ResourceInitializationException {
041                    TypeSystemDescription typeSystemDescription = TypeSystemDescriptionFactory
042                                    .createTypeSystemDescription();
043                    return AnalysisEngineFactory.createPrimitiveDescription(RoomNumberAnnotator.class,
044                                    typeSystemDescription);
045            }
046    
047            public static void main(String[] args) throws Exception {
048                    File outputDirectory = new File("target/example-output/ex1/");
049                    outputDirectory.mkdirs();
050                    AnalysisEngineDescription aed = createDescriptor();
051                    aed.toXML(new FileOutputStream(new File(outputDirectory, "RoomNumberAnnotator.xml")));
052            }
053    
054    }
