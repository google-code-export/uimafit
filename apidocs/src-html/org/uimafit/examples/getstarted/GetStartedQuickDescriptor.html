017    package org.uimafit.examples.getstarted;
018    
019    import java.io.FileNotFoundException;
020    import java.io.FileOutputStream;
021    import java.io.IOException;
022    
023    import org.apache.uima.analysis_engine.AnalysisEngineDescription;
024    import org.apache.uima.resource.ResourceInitializationException;
025    import org.apache.uima.resource.metadata.TypeSystemDescription;
026    import org.uimafit.factory.AnalysisEngineFactory;
027    import org.uimafit.factory.TypeSystemDescriptionFactory;
028    import org.xml.sax.SAXException;
029    
030    /**
031     * 
032     * @author Philip Ogren
033     * 
034     */
035    public class GetStartedQuickDescriptor {
036    
037            public static void main(String[] args) throws ResourceInitializationException, FileNotFoundException, SAXException, IOException {
038                    // normally you would pass in a list of classes corresponding to type
039                    // system types here
040                    TypeSystemDescription typeSystemDescription = TypeSystemDescriptionFactory.createTypeSystemDescription(new Class[0]);
041                    // here we will instantiate the analysis engine using the value
042                    // "uimaFIT" for the parameter "stringParamName".
043                    AnalysisEngineDescription analysisEngineDescription = AnalysisEngineFactory.createPrimitiveDescription(GetStartedQuickAE.class,
044                                    typeSystemDescription, "stringParamName", "uimaFIT");
045                    analysisEngineDescription.toXML(new FileOutputStream("GetStartedQuickAE.xml"));
046            }
047    }
