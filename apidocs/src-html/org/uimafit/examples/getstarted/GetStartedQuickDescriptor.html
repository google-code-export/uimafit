017    package org.uimafit.examples.getstarted;
018    
019    import java.io.FileNotFoundException;
020    import java.io.FileOutputStream;
021    import java.io.IOException;
022    
023    import org.apache.uima.analysis_engine.AnalysisEngineDescription;
024    import org.apache.uima.resource.ResourceInitializationException;
025    import org.uimafit.factory.AnalysisEngineFactory;
026    import org.xml.sax.SAXException;
027    
028    /**
029     * 
030     * @author Philip Ogren
031     * 
032     */
033    public class GetStartedQuickDescriptor {
034    
035            public static void main(String[] args) throws ResourceInitializationException,
036                            FileNotFoundException, SAXException, IOException {
037                    // uimaFIT automatically uses all type systems listed in META-INF/org.uimafit/types.txt
038                    
039                    // Instantiate the analysis engine using the value "uimaFIT" for the parameter
040                    // PARAM_STRING ("stringParam").
041                    AnalysisEngineDescription analysisEngineDescription = AnalysisEngineFactory
042                                    .createPrimitiveDescription(GetStartedQuickAE.class,
043                                                    GetStartedQuickAE.PARAM_STRING, "uimaFIT");
044                    
045                    // Write the descriptor to an XML file
046                    analysisEngineDescription.toXML(new FileOutputStream("GetStartedQuickAE.xml"));
047            }
048    }
