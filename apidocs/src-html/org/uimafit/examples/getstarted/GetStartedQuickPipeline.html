017    package org.uimafit.examples.getstarted;
018    
019    import org.apache.uima.UIMAException;
020    import org.apache.uima.analysis_engine.AnalysisEngine;
021    import org.apache.uima.jcas.JCas;
022    import org.uimafit.factory.AnalysisEngineFactory;
023    import org.uimafit.factory.JCasFactory;
024    
025    /**
026     * 
027     * @author Philip Ogren
028     * 
029     */
030    public class GetStartedQuickPipeline {
031    
032            public static void main(String[] args) throws UIMAException {
033                    // uimaFIT automatically uses all type systems listed in META-INF/org.uimafit/types.txt
034    
035                    // uimaFIT doesn't provide any collection readers - so we will just instantiate a JCas and
036                    // run it through our AE
037                    JCas jCas = JCasFactory.createJCas();
038                    
039                    // Instantiate the analysis engine using the value "uimaFIT" for the parameter
040                    // PARAM_STRING ("stringParam").
041                    AnalysisEngine analysisEngine = AnalysisEngineFactory.createPrimitive(
042                                    GetStartedQuickAE.class, 
043                                    GetStartedQuickAE.PARAM_STRING, "uimaFIT");
044                    
045                    // run the analysis engine and look for a special greeting in your console.
046                    analysisEngine.process(jCas);
047            }
048    }
