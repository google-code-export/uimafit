017    package org.uimafit.examples.experiment.pos;
018    
019    import java.io.IOException;
020    
021    import org.apache.uima.UIMAException;
022    import org.apache.uima.analysis_engine.AnalysisEngineDescription;
023    import org.apache.uima.cas.CAS;
024    import org.apache.uima.collection.CollectionReader;
025    import org.apache.uima.resource.metadata.TypeSystemDescription;
026    import org.uimafit.component.ViewTextCopierAnnotator;
027    import org.uimafit.component.xwriter.XWriter;
028    import org.uimafit.examples.type.Sentence;
029    import org.uimafit.examples.type.Token;
030    import org.uimafit.factory.AggregateBuilder;
031    import org.uimafit.factory.AnalysisEngineFactory;
032    import org.uimafit.factory.CollectionReaderFactory;
033    import org.uimafit.factory.TypeSystemDescriptionFactory;
034    import org.uimafit.pipeline.SimplePipeline;
035    
036    /**
037     * This class demonstrates a very common (though simplified) experimental setup
038     * in which gold standard data is available for some task and you want to
039     * evaluate how well your analysis engine works against that data. Here we are
040     * evaluating "BaselineTagger" which is a (ridiculously) simple part-of-speech
041     * tagger against the part-of-speech tags found in
042     * "src/main/resources/org/uimafit/examples/pos/sample.txt.pos".
043     * <p>
044     * The basic strategy is as follows:
045     * <ul>
046     * <li>post the data "as is" into the default view</li>
047     * <li>parse the gold-standard tokens and part-of-speech tags and put the
048     * results into another view we will call GOLD_VIEW</li>
049     * <li>create another view called SYSTEM_VIEW and copy the text and Token
050     * annotations from the GOLD_VIEW into this view</li>
051     * <li>Run the BaselineTagger on the SYSTEM_VIEW over the copied Token
052     * annoations</lI>
053     * <li>Evaluate the part-of-speech tags found in the SYSTEM_VIEW with those in
054     * the GOLD_VIEW</li>
055     * </ul>
056     * 
057     * Please see comments in the code for details on how the UIMA pipeline is set
058     * up and run for this task.
059     * 
060     * @author Philip Ogren
061     * 
062     */
063    public class RunExperiment {
064    
065            public static void main(String[] args) throws UIMAException, IOException {
066                    String samplePosFileName = "src/main/resources/org/uimafit/examples/pos/sample.txt.pos";
067    
068                    TypeSystemDescription typeSystem = TypeSystemDescriptionFactory.createTypeSystemDescription(Sentence.class, Token.class);
069    
070                    // The lineReader simply copies the lines from the input file into the
071                    // default view - one line per CAS
072                    CollectionReader lineReader = CollectionReaderFactory.createCollectionReader(LineReader.class, typeSystem, LineReader.PARAM_INPUT_FILE,
073                                    samplePosFileName);
074    
075                    AggregateBuilder builder = new AggregateBuilder();
076    
077                    // The goldTagger parses the data in the default view into Token objects
078                    // along with their part-of-speech tags which will be added to the
079                    // GOLD_VIEW
080                    AnalysisEngineDescription goldTagger = AnalysisEngineFactory.createPrimitiveDescription(GoldTagger.class, typeSystem);
081                    builder.add(goldTagger);
082    
083                    // The textCopier creates the SYSTEM_VIEW and set the text of this view
084                    // to that of the text found in GOLD_VIEW
085                    AnalysisEngineDescription textCopier = AnalysisEngineFactory.createPrimitiveDescription(ViewTextCopierAnnotator.class, typeSystem,
086                                    ViewTextCopierAnnotator.PARAM_SOURCE_VIEW_NAME, ViewNames.GOLD_VIEW, ViewTextCopierAnnotator.PARAM_DESTINATION_VIEW_NAME,
087                                    ViewNames.SYSTEM_VIEW);
088                    builder.add(textCopier);
089    
090                    // The sentenceAndTokenCopier copies Token and Sentence annotations in
091                    // the GOLD_VIEW into the SYSTEM_VIEW
092                    AnalysisEngineDescription sentenceAndTokenCopier = AnalysisEngineFactory.createPrimitiveDescription(SentenceAndTokenCopier.class, typeSystem);
093                    builder.add(sentenceAndTokenCopier, ViewNames.VIEW1, ViewNames.GOLD_VIEW, ViewNames.VIEW2, ViewNames.SYSTEM_VIEW);
094    
095                    // The baselineTagger is run on the SYSTEM_VIEW
096                    AnalysisEngineDescription baselineTagger = AnalysisEngineFactory.createPrimitiveDescription(BaselineTagger.class, typeSystem);
097                    builder.add(baselineTagger, CAS.NAME_DEFAULT_SOFA, ViewNames.SYSTEM_VIEW);
098    
099                    // The evaluator will compare the part-of-speech tags in the SYSTEM_VIEW
100                    // with those in the GOLD_VIEW
101                    AnalysisEngineDescription evaluator = AnalysisEngineFactory.createPrimitiveDescription(Evaluator.class, typeSystem);
102                    builder.add(evaluator);
103    
104                    // The xWriter writes out the contents of each CAS (one per sentence) to
105                    // an XMI file. It is instructive to open one of these
106                    // xmi files in the CAS Visual Debugger and look at the contents of each
107                    // view.
108                    AnalysisEngineDescription xWriter = AnalysisEngineFactory.createPrimitiveDescription(XWriter.class, typeSystem,
109                                    XWriter.PARAM_OUTPUT_DIRECTORY_NAME, "src/main/resources/org/uimafit/examples/pos/xmi");
110                    builder.add(xWriter);
111    
112                    // runs the collection reader and the aggregate AE.
113                    SimplePipeline.runPipeline(lineReader, builder.createAggregate());
114            }
115    }
