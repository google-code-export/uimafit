018    package org.uimafit.examples.experiment.pos;
019    
020    import org.apache.uima.analysis_engine.AnalysisEngineProcessException;
021    import org.apache.uima.jcas.JCas;
022    import org.uimafit.component.JCasAnnotator_ImplBase;
023    import org.uimafit.examples.type.Token;
024    import org.uimafit.util.JCasUtil;
025    
026    /**
027     * This "baseline" part-of-speech tagger isn't very sophisticated! Notice,
028     * however, that the tagger operates on the default view. This will be mapped to
029     * the "system" view when we run our experiment.
030     * 
031     * @author Philip Ogren
032     * 
033     */
034    public class BaselineTagger extends JCasAnnotator_ImplBase {
035    
036            @Override
037            public void process(JCas jCas) throws AnalysisEngineProcessException {
038                    for (Token token : JCasUtil.iterate(jCas, Token.class)) {
039                            String word = token.getCoveredText();
040                            if (word.equals("a") || word.equals("the")) {
041                                    token.setPos("DT");
042                            }
043                            else {
044                                    token.setPos("NN");
045                            }
046                    }
047            }
048    
049    }
