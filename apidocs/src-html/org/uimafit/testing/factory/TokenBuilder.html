018    package org.uimafit.testing.factory;
019    
020    import java.util.ArrayList;
021    import java.util.List;
022    
023    import org.apache.uima.UIMAException;
024    import org.apache.uima.cas.Feature;
025    import org.apache.uima.jcas.JCas;
026    import org.apache.uima.jcas.tcas.Annotation;
027    import org.uimafit.factory.AnnotationFactory;
028    
029    /**
030     * 
031     * This class provides convenience methods for creating tokens and sentences and add them to a
032     * {@link JCas}.
033     * 
034     * @author Steven Bethard, Philip Ogren
035     * @author Richard Eckart de Castilho
036     * 
037     * @param <TOKEN_TYPE>
038     *            the type system token type (e.g. org.uimafit.examples.type.Token)
039     * @param <SENTENCE_TYPE>
040     *            the type system sentence type (e.g. org.uimafit.examples.type.Sentence)
041     */
042    
043    public class TokenBuilder<TOKEN_TYPE extends Annotation, SENTENCE_TYPE extends Annotation> {
044            private Class<TOKEN_TYPE> tokenClass;
045            private Class<SENTENCE_TYPE> sentenceClass;
046            private String posFeatureName;
047            private String stemFeatureName;
048    
049            /**
050             * Calls {@link TokenBuilder#TokenBuilder(Class, Class, String, String)} with the last two
051             * arguments as null
052             * 
053             * @param tokenClass
054             * @param sentenceClass
055             */
056            public TokenBuilder(Class<TOKEN_TYPE> tokenClass, Class<SENTENCE_TYPE> sentenceClass) {
057                    this(tokenClass, sentenceClass, null, null);
058            }
059    
060            /**
061             * Instantiates a TokenBuilder with the type system information that the builder needs to build
062             * tokens.
063             * 
064             * @param tokenClass
065             *            the class of your token type from your type system (e.g.
066             *            org.uimafit.type.Token.class)
067             * @param sentenceClass
068             *            the class of your sentence type from your type system (e.g.
069             *            org.uimafit.type.Sentence.class)
070             * @param posFeatureName
071             *            the feature name for the part-of-speech tag for your token type. This assumes that
072             *            there is a single string feature for which to put your pos tag. null is an ok
073             *            value.
074             * @param stemFeatureName
075             *            the feature name for the stem for your token type. This assumes that there is a
076             *            single string feature for which to put your stem. null is an ok value.
077             */
078            public TokenBuilder(Class<TOKEN_TYPE> tokenClass, Class<SENTENCE_TYPE> sentenceClass,
079                            String posFeatureName, String stemFeatureName) {
080                    this.tokenClass = tokenClass;
081                    this.sentenceClass = sentenceClass;
082                    setPosFeatureName(posFeatureName);
083                    setStemFeatureName(stemFeatureName);
084            }
085    
086            /**
087             * Instantiates a TokenBuilder with the type system information that the builder needs to build
088             * tokens.
089             * 
090             * @param <T>
091             *            the type system token type (e.g. org.uimafit.examples.type.Token)
092             * @param <S>
093             *            the type system sentence type (e.g. org.uimafit.examples.type.Sentence)
094             * @param tokenClass
095             *            the class of your token type from your type system (e.g.
096             *            org.uimafit.type.Token.class)
097             * @param sentenceClass
098             *            the class of your sentence type from your type system (e.g.
099             *            org.uimafit.type.Sentence.class)
100             * @return the builder.
101             */
102            public static <T extends Annotation, S extends Annotation> TokenBuilder<T, S> create(
103                            Class<T> tokenClass, Class<S> sentenceClass) {
104                    return new TokenBuilder<T, S>(tokenClass, sentenceClass);
105            }
106    
107            /**
108             * Set the feature name for the part-of-speech tag for your token type. This assumes that there
109             * is a single string feature for which to put your pos tag. null is an ok value.
110             * 
111             * @param posFeatureName
112             *            the part-of-speech feature name.
113             */
114            public void setPosFeatureName(String posFeatureName) {
115                    this.posFeatureName = posFeatureName;
116            }
117    
118            /**
119             * Set the feature name for the stem for your token type. This assumes that there is a single
120             * string feature for which to put your stem. null is an ok value.
121             * 
122             * @param stemFeatureName
123             *            the stem feature name.
124             */
125            public void setStemFeatureName(String stemFeatureName) {
126                    this.stemFeatureName = stemFeatureName;
127            }
128    
129            /**
130             * Builds white-space delimited tokens from the input text.
131             * 
132             * @param jCas
133             *            the JCas to add the tokens to
134             * @param text
135             *            the JCas will have its document text set to this.
136             * @throws UIMAException
137             */
138            public void buildTokens(JCas jCas, String text) throws UIMAException {
139                    if (text == null) {
140                            throw new IllegalArgumentException("text may not be null.");
141                    }
142                    buildTokens(jCas, text, text, null, null);
143            }
144    
145            /**
146             * see {@link #buildTokens(JCas, String, String, String, String)}
147             * 
148             * @param jCas
149             * @param text
150             * @param tokensString
151             * @throws UIMAException
152             */
153            public void buildTokens(JCas jCas, String text, String tokensString) throws UIMAException {
154                    if (tokensString == null) {
155                            throw new IllegalArgumentException("tokensText may not be null.");
156                    }
157                    buildTokens(jCas, text, tokensString, null, null);
158            }
159    
160            /**
161             * see {@link #buildTokens(JCas, String, String, String, String)}
162             * 
163             * @param jCas
164             * @param text
165             * @param tokensString
166             * @param posTagsString
167             * @throws UIMAException
168             */
169            public void buildTokens(JCas jCas, String text, String tokensString, String posTagsString)
170                            throws UIMAException {
171                    buildTokens(jCas, text, tokensString, posTagsString, null);
172            }
173    
174            /**
175             * Build tokens for the given text, tokens, part-of-speech tags, and word stems.
176             * 
177             * @param aJCas
178             *            the JCas to add the Token annotations to
179             * @param aText
180             *            this method sets the text of the JCas to this method. Therefore, it is generally a
181             *            good idea to call JCas.reset() before calling this method when passing in the
182             *            default view.
183             * @param aTokensString
184             *            the tokensString must have the same non-white space characters as the text. The
185             *            tokensString is used to identify token boundaries using white space - i.e. the
186             *            only difference between the 'text' parameter and the 'tokensString' parameter is
187             *            that the latter may have more whitespace characters. For example, if the text is
188             *            "She ran." then the tokensString might be "She ran ."
189             * @param aPosTagsString
190             *            the posTagsString should be a space delimited string of part-of-speech tags - one
191             *            for each token
192             * @param aStemsString
193             *            the stemsString should be a space delimitied string of stems - one for each token
194             * @throws UIMAException
195             */
196            public void buildTokens(JCas aJCas, String aText, String aTokensString, String aPosTagsString,
197                            String aStemsString) throws UIMAException {
198                    aJCas.setDocumentText(aText);
199    
200                    if (aPosTagsString != null && posFeatureName == null) {
201                            throw new IllegalArgumentException("posTagsString must be null if TokenBuilder is "
202                                            + "not initialized with a feature name corresponding to the part-of-speech "
203                                            + "feature of the token type (assuming your token type has such a feature).");
204                    }
205    
206                    if (aStemsString != null && stemFeatureName == null) {
207                            throw new IllegalArgumentException(
208                                            "stemsString must be null if TokenBuilder is not "
209                                                            + "initialized with a feature name corresponding to the part-of-speech feature "
210                                                            + "of the token type (assuming your token type has such a feature).");
211                    }
212    
213                    Feature posFeature = null;
214                    if (posFeatureName != null) {
215                            // String fullPosFeatureName = tokenClass.getClass().getName()+":"+posFeatureName;
216                            // posFeature = jCas.getTypeSystem().getFeatureByFullName(fullPosFeatureName);
217                            posFeature = aJCas.getTypeSystem().getType(tokenClass.getName())
218                                            .getFeatureByBaseName(posFeatureName);
219                    }
220                    Feature stemFeature = null;
221                    if (stemFeatureName != null) {
222                            stemFeature = aJCas.getTypeSystem().getType(tokenClass.getName())
223                                            .getFeatureByBaseName(stemFeatureName);
224                    }
225    
226                    String tokensString = aTokensString.replaceAll("\\s*\n\\s*", "\n");
227                    String[] sentenceStrings = tokensString.split("\n");
228                    String[] posTags = aPosTagsString != null ? aPosTagsString.split("\\s+") : null;
229                    String[] stems = aStemsString != null ? aStemsString.split("\\s+") : null;
230    
231                    int offset = 0;
232                    int tokenIndex = 0;
233    
234                    for (String sentenceString : sentenceStrings) {
235                            String[] tokenStrings = sentenceString.trim().split("\\s+");
236                            List<Annotation> tokenAnnotations = new ArrayList<Annotation>();
237                            for (String tokenString : tokenStrings) {
238                                    // move the offset up to the beginning of the token
239                                    while (!aText.startsWith(tokenString, offset)) {
240                                            offset++;
241                                            if (offset > aText.length()) {
242                                                    throw new IllegalArgumentException(String.format(
243                                                                    "unable to find string %s", tokenString));
244                                            }
245                                    }
246    
247                                    // add the Token
248                                    int start = offset;
249                                    offset = offset + tokenString.length();
250                                    Annotation token = AnnotationFactory.createAnnotation(aJCas, start, offset,
251                                                    tokenClass);
252                                    tokenAnnotations.add(token);
253    
254                                    // set the stem and part of speech if present
255                                    if (posTags != null) {
256                                            token.setStringValue(posFeature, posTags[tokenIndex]);
257                                    }
258                                    if (stems != null) {
259                                            token.setStringValue(stemFeature, stems[tokenIndex]);
260                                    }
261                                    tokenIndex++;
262                            }
263                            if (tokenAnnotations.size() > 0) {
264                                    int begin = tokenAnnotations.get(0).getBegin();
265                                    int end = tokenAnnotations.get(tokenAnnotations.size() - 1).getEnd();
266                                    AnnotationFactory.createAnnotation(aJCas, begin, end, sentenceClass);
267                            }
268                    }
269    
270            }
