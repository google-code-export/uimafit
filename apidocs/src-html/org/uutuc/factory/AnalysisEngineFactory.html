065            /**
066             * Get an AnalysisEngine from the name (Java-style, dotted) of an XML
067             * descriptor file, and a set of configuration parameters.
068             * 
069             * @param descriptorName
070             *            The fully qualified, Java-style, dotted name of the XML
071             *            descriptor file.
072             * @param configurationData
073             *            Any additional configuration parameters to be set. These
074             *            should be supplied as (name, value) pairs, so there should
075             *            always be an even number of parameters.
076             * @return The AnalysisEngine created from the XML descriptor and the
077             *         configuration parameters.
078             * @throws IOException
079             * @throws UIMAException
080             */
081            public static AnalysisEngine createAnalysisEngine(String descriptorName, Object... configurationData)
082                            throws UIMAException, IOException {
083                    Import_impl imprt = new Import_impl();
084                    imprt.setName(descriptorName);
085                    URL url = imprt.findAbsoluteUrl(UIMAFramework.newDefaultResourceManager());
086                    ResourceSpecifier specifier = ResourceCreationSpecifierFactory.createResourceCreationSpecifier(url,
087                                    configurationData);
088                    return UIMAFramework.produceAnalysisEngine(specifier);
089            }

167            public static AnalysisEngineDescription createPrimitiveDescription(
168                            Class<? extends AnalysisComponent> componentClass, TypeSystemDescription typeSystem,
169                            TypePriorities typePriorities, Capability[] capabilities, ConfigurationParameter[] configurationParameters,
170                            Object[] configurationValues) throws ResourceInitializationException {
171    
172                    // create the descriptor and set configuration parameters
173                    AnalysisEngineDescription desc = new AnalysisEngineDescription_impl();
174                    desc.setFrameworkImplementation(Constants.JAVA_FRAMEWORK_NAME);
175                    desc.setPrimitive(true);
176                    desc.setAnnotatorImplementationName(componentClass.getName());
177                    if (componentClass.isAnnotationPresent(org.uutuc.descriptor.AnalysisComponent.class)) {
178                            org.uutuc.descriptor.AnalysisComponent anno = componentClass
179                                            .getAnnotation(org.uutuc.descriptor.AnalysisComponent.class);
180                            OperationalProperties op = desc.getAnalysisEngineMetaData().getOperationalProperties();
181                            op.setMultipleDeploymentAllowed(anno.multipleDeploymentAllowed());
182                            op.setModifiesCas(anno.modifiesCas());
183                            op.setOutputsNewCASes(anno.outputsNewCases());
184                    }
185    
186                    // Extract external resource dependencies
187                    Collection<ExternalResourceDependency> deps = ExternalResourceConfigurator.getResourceDeclarations(
188                                    componentClass).values();
189                    desc.setExternalResourceDependencies(deps.toArray(new ExternalResourceDependency[deps.size()]));
190    
191                    ConfigurationData reflectedConfigurationData = ConfigurationParameterFactory
192                                    .createConfigurationData(componentClass);
193                    ResourceCreationSpecifierFactory.setConfigurationParameters(desc,
194                                    reflectedConfigurationData.configurationParameters, reflectedConfigurationData.configurationValues);
195                    if (configurationParameters != null) {
196                            ResourceCreationSpecifierFactory.setConfigurationParameters(desc,
197                                            configurationParameters, configurationValues);
198                    }
199    
200                    // set the type system
201                    if (typeSystem != null) {
202                            desc.getAnalysisEngineMetaData().setTypeSystem(typeSystem);
203                    }
204    
205                    if (typePriorities != null) {
206                            desc.getAnalysisEngineMetaData().setTypePriorities(typePriorities);
207                    }
208    
209                    if (capabilities == null) {
210                            capabilities = CapabilityFactory.createCapability(componentClass);
211                    }
212                    if (capabilities != null) {
213                            desc.getAnalysisEngineMetaData().setCapabilities(capabilities);
214                    }
215                    return desc;
216            }

237            @SuppressWarnings("unchecked")
238            public static AnalysisEngine createPrimitive(AnalysisEngineDescription desc, Object... configurationData)
239                            throws ResourceInitializationException {
240                    AnalysisEngine engine = new PrimitiveAnalysisEngine_impl();
241    
242                    if (configurationData != null) {
243                            ConfigurationData cdata = ConfigurationParameterFactory.createConfigurationData(configurationData);
244                            ConfigurationParameter[] configurationParameters = cdata.configurationParameters;
245                            Object[] configurationValues = cdata.configurationValues;
246                            ResourceCreationSpecifierFactory.setConfigurationParameters(desc, configurationParameters,
247                                            configurationValues);
248                            ResourceMetaData metaData = desc.getMetaData();
249                            ConfigurationParameterSettings paramSettings = metaData.getConfigurationParameterSettings();
250                            Map additionalParameters = new HashMap();
251                            additionalParameters.put(AnalysisEngine.PARAM_CONFIG_PARAM_SETTINGS, paramSettings);
252                            engine.initialize(desc, additionalParameters);
253                    }
254                    else {
255                            engine.initialize(desc, null);
256                    }
257                    return engine;
258            }

278            public static AnalysisEngineDescription createAggregateDescription(
279                            List<Class<? extends AnalysisComponent>> componentClasses, TypeSystemDescription typeSystem,
280                            TypePriorities typePriorities, SofaMapping[] sofaMappings, Object... configurationParameters)
281                            throws ResourceInitializationException {
282    
283                    List<AnalysisEngineDescription> primitiveEngineDescriptions = new ArrayList<AnalysisEngineDescription>();
284                    List<String> componentNames = new ArrayList<String>();
285    
286                    for (Class<? extends AnalysisComponent> componentClass : componentClasses) {
287                            AnalysisEngineDescription primitiveDescription = createPrimitiveDescription(componentClass, typeSystem,
288                                            typePriorities, configurationParameters);
289                            primitiveEngineDescriptions.add(primitiveDescription);
290                            componentNames.add(componentClass.getName());
291                    }
292                    return createAggregateDescription(primitiveEngineDescriptions, componentNames, typeSystem, typePriorities,
293                                    sofaMappings);
294            }

309            public static AnalysisEngineDescription createAggregateDescription(
310                            AnalysisEngineDescription... analysisEngineDescriptions) throws ResourceInitializationException {
311                    String[] names = new String[analysisEngineDescriptions.length];
312                    int i = 0;
313                    for (AnalysisEngineDescription aed : analysisEngineDescriptions) {
314                            names[i] = aed.getImplementationName() + "-" + i;
315                            i++;
316                    }
317    
318                    return createAggregateDescription(asList(analysisEngineDescriptions), asList(names), null, null, null);
319            }

321            @SuppressWarnings("unchecked")
322            public static AnalysisEngineDescription createAggregateDescription(
323                            List<AnalysisEngineDescription> analysisEngineDescriptions, List<String> componentNames,
324                            TypeSystemDescription typeSystem, TypePriorities typePriorities, SofaMapping[] sofaMappings)
325                            throws ResourceInitializationException {
326    
327                    // create the descriptor and set configuration parameters
328                    AnalysisEngineDescription desc = new AnalysisEngineDescription_impl();
329                    desc.setFrameworkImplementation(Constants.JAVA_FRAMEWORK_NAME);
330                    desc.setPrimitive(false);
331    
332                    List<String> flowNames = new ArrayList<String>();
333    
334                    for (int i = 0; i < analysisEngineDescriptions.size(); i++) {
335                            AnalysisEngineDescription aed = analysisEngineDescriptions.get(i);
336                            String componentName = componentNames.get(i);
337                            desc.getDelegateAnalysisEngineSpecifiersWithImports().put(componentName, aed);
338                            flowNames.add(componentName);
339                    }
340    
341                    FixedFlow fixedFlow = new FixedFlow_impl();
342                    fixedFlow.setFixedFlow(flowNames.toArray(new String[flowNames.size()]));
343                    desc.getAnalysisEngineMetaData().setFlowConstraints(fixedFlow);
344    
345                    if (typePriorities != null) {
346                            desc.getAnalysisEngineMetaData().setTypePriorities(typePriorities);
347                    }
348    
349                    if (sofaMappings != null) {
350                            desc.setSofaMappings(sofaMappings);
351                    }
352    
353                    return desc;
354            }

356            /**
357             * Creates an AnalysisEngine from the given descriptor, and uses the engine
358             * to process the file or text.
359             * 
360             * @param descriptorFileName
361             *            The fully qualified, Java-style, dotted name of the XML
362             *            descriptor file.
363             * @param fileNameOrText
364             *            Either the path of a file to be loaded, or a string to use as
365             *            the text. If the string given is not a valid path in the file
366             *            system, it will be assumed to be text.
367             * @return A JCas object containing the processed document.
368             * @throws IOException
369             * @throws UIMAException
370             */
371            public static JCas process(String descriptorFileName, String fileNameOrText) throws IOException, UIMAException {
372                    AnalysisEngine engine = createAnalysisEngine(descriptorFileName);
373                    JCas jCas = process(engine, fileNameOrText);
374                    engine.collectionProcessComplete();
375                    return jCas;
376            }
377    
378            /**
379             * Processes the file or text with the given AnalysisEngine.
380             * 
381             * @param analysisEngine
382             *            The AnalysisEngine object to process the text.
383             * @param fileNameOrText
384             *            Either the path of a file to be loaded, or a string to use as
385             *            the text. If the string given is not a valid path in the file
386             *            system, it will be assumed to be text.
387             * @return A JCas object containing the processed document.
388             * @throws IOException
389             * @throws UIMAException
390             */
391            public static JCas process(AnalysisEngine analysisEngine, String fileNameOrText) throws IOException, UIMAException {
392                    File textFile = new File(fileNameOrText);
393                    String text;
394                    if (textFile.exists()) {
395                            text = FileUtils.file2String(textFile);
396                    }
397                    else {
398                            text = fileNameOrText;
399                    }
400    
401                    JCas jCas = analysisEngine.newJCas();
402                    jCas.setDocumentText(text);
403                    analysisEngine.process(jCas);
404                    return jCas;
405            }
