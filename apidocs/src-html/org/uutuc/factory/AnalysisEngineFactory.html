044    public class AnalysisEngineFactory {
045    
046            /**
047             * Get an AnalysisEngine from the name (Java-style, dotted) of an XML
048             * descriptor file, and a set of configuration parameters.
049             * 
050             * @param descriptorName
051             *            The fully qualified, Java-style, dotted name of the XML
052             *            descriptor file.
053             * @param parameters
054             *            Any additional configuration parameters to be set. These
055             *            should be supplied as (name, value) pairs, so there should
056             *            always be an even number of parameters.
057             * @return The AnalysisEngine created from the XML descriptor and the
058             *         configuration parameters.
059             * @throws IOException
060             * @throws UIMAException
061             */
062            public static AnalysisEngine createAnalysisEngine(String descriptorName, Object... parameters)
063                            throws UIMAException, IOException {
064                    Import_impl imprt = new Import_impl();
065                    imprt.setName(descriptorName);
066                    URL url = imprt.findAbsoluteUrl(UIMAFramework.newDefaultResourceManager());
067                    ResourceSpecifier specifier = ResourceCreationSpecifierFactory.createResourceCreationSpecifier(url, parameters);
068                    return UIMAFramework.produceAnalysisEngine(specifier);
069            }
070    
071            /**
072             * Get an AnalysisEngine from an XML descriptor file and a set of
073             * configuration parameters.
074             * 
075             * @param descriptorPath
076             *            The path to the XML descriptor file.
077             * @param parameters
078             *            Any additional configuration parameters to be set. These
079             *            should be supplied as (name, value) pairs, so there should
080             *            always be an even number of parameters.
081             * @return The AnalysisEngine created from the XML descriptor and the
082             *         configuration parameters.
083             * @throws UIMAException
084             * @throws IOException
085             */
086            public static AnalysisEngine createAnalysisEngineFromPath(String descriptorPath, Object... parameters)
087                            throws UIMAException, IOException {
088                    ResourceSpecifier specifier;
089                    specifier = ResourceCreationSpecifierFactory.createResourceCreationSpecifier(descriptorPath, parameters);
090                    return UIMAFramework.produceAnalysisEngine(specifier);
091            }
092    
093            /**
094             * Get an AnalysisEngine from an AnalysisComponent class, a type system and
095             * a set of configuration parameters.
096             * 
097             * @param componentClass
098             *            The class of the AnalysisComponent to be created as an
099             *            AnalysisEngine.
100             * @param typeSystem
101             *            A description of the types used by the AnalysisComponent (may
102             *            be null).
103             * @param configurationParameters
104             *            Any additional configuration parameters to be set. These
105             *            should be supplied as (name, value) pairs, so there should
106             *            always be an even number of parameters.
107             * @return The AnalysisEngine created from the AnalysisComponent class and
108             *         initialized with the type system and the configuration
109             *         parameters.
110             * @throws ResourceInitializationException
111             */
112            public static AnalysisEngine createAnalysisEngine(Class<? extends AnalysisComponent> componentClass,
113                            TypeSystemDescription typeSystem, Object... configurationParameters) throws ResourceInitializationException {
114    
115                    // create the descriptor and set configuration parameters
116                    AnalysisEngineDescription desc = new AnalysisEngineDescription_impl();
117                    desc.setFrameworkImplementation(Constants.JAVA_FRAMEWORK_NAME);
118                    desc.setPrimitive(true);
119                    desc.setAnnotatorImplementationName(componentClass.getName());
120                    ResourceCreationSpecifierFactory.setConfigurationParameters(desc, configurationParameters);
121    
122                    // set the type system
123                    if (typeSystem != null) {
124                            desc.getAnalysisEngineMetaData().setTypeSystem(typeSystem);
125                    }
126    
127                    // create the AnalysisEngine, initialize it and return it
128                    AnalysisEngine engine = new PrimitiveAnalysisEngine_impl();
129                    engine.initialize(desc, null);
130                    return engine;
131    
132            }
133    
134            /**
135             * Creates an AnalysisEngine from the given descriptor, and uses the engine
136             * to process the file or text.
137             * 
138             * @param descriptorFileName
139             *            The fully qualified, Java-style, dotted name of the XML
140             *            descriptor file.
141             * @param fileNameOrText
142             *            Either the path of a file to be loaded, or a string to use as
143             *            the text. If the string given is not a valid path in the file
144             *            system, it will be assumed to be text.
145             * @return A JCas object containing the processed document.
146             * @throws IOException
147             * @throws UIMAException
148             */
149            public static JCas process(String descriptorFileName, String fileNameOrText) throws IOException, UIMAException {
150                    AnalysisEngine engine = createAnalysisEngine(descriptorFileName);
151                    JCas jCas = process(engine, fileNameOrText);
152                    engine.collectionProcessComplete();
153                    return jCas;
154            }
155    
156            /**
157             * Processes the file or text with the given AnalysisEngine.
158             * 
159             * @param analysisEngine
160             *            The AnalysisEngine object to process the text.
161             * @param fileNameOrText
162             *            Either the path of a file to be loaded, or a string to use as
163             *            the text. If the string given is not a valid path in the file
164             *            system, it will be assumed to be text.
165             * @return A JCas object containing the processed document.
166             * @throws IOException
167             * @throws UIMAException
168             */
169            public static JCas process(AnalysisEngine analysisEngine, String fileNameOrText) throws IOException, UIMAException {
170                    File textFile = new File(fileNameOrText);
171                    String text;
172                    if (textFile.exists()) {
173                            text = FileUtils.file2String(textFile);
174                    }
175                    else {
176                            text = fileNameOrText;
177                    }
178    
179                    JCas jCas = analysisEngine.newJCas();
180                    jCas.setDocumentText(text);
181                    analysisEngine.process(jCas);
182                    return jCas;
183            }
