040             * A mapping from Java class names to UIMA configuration parameter type
041             * names. Used by setConfigurationParameters().
042             */
043            public static final Map<String, String> javaUimaTypeMap = new HashMap<String, String>();
044            static {
045                    javaUimaTypeMap.put(Boolean.class.getName(), ConfigurationParameter.TYPE_BOOLEAN);
046                    javaUimaTypeMap.put(Float.class.getName(), ConfigurationParameter.TYPE_FLOAT);
047                    javaUimaTypeMap.put(Double.class.getName(), ConfigurationParameter.TYPE_FLOAT);
048                    javaUimaTypeMap.put(Integer.class.getName(), ConfigurationParameter.TYPE_INTEGER);
049                    javaUimaTypeMap.put(String.class.getName(), ConfigurationParameter.TYPE_STRING);
050                    javaUimaTypeMap.put("boolean", ConfigurationParameter.TYPE_BOOLEAN);
051                    javaUimaTypeMap.put("float", ConfigurationParameter.TYPE_FLOAT);
052                    javaUimaTypeMap.put("double", ConfigurationParameter.TYPE_FLOAT);
053                    javaUimaTypeMap.put("int", ConfigurationParameter.TYPE_INTEGER);
054

057            public static boolean isConfigurationParameterField(Field field) {
058                    return field.isAnnotationPresent(org.uutuc.descriptor.ConfigurationParameter.class);
059            }
060    
061            public static Object getDefaultValue(Field field) {
062                    if (isConfigurationParameterField(field)) {
063                            org.uutuc.descriptor.ConfigurationParameter annotation = field
064                                            .getAnnotation(org.uutuc.descriptor.ConfigurationParameter.class);
065    
066                            String[] stringValue = annotation.defaultValue();
067                            if (stringValue.length == 1
068                                            && stringValue[0].equals(org.uutuc.descriptor.ConfigurationParameter.NO_DEFAULT_VALUE)) return null;
069    
070                            String valueType = getConfigurationParameterType(field);
071                            boolean isMultiValued = isMultiValued(field);
072    
073                            if (!isMultiValued) {
074                                    if (ConfigurationParameter.TYPE_BOOLEAN.equals(valueType)) {
075                                            return Boolean.parseBoolean(stringValue[0]);
076                                    }
077                                    else if (ConfigurationParameter.TYPE_FLOAT.equals(valueType)) {
078                                            return Float.parseFloat(stringValue[0]);
079                                    }
080                                    else if (ConfigurationParameter.TYPE_INTEGER.equals(valueType)) {
081                                            return Integer.parseInt(stringValue[0]);
082                                    }
083                                    else if (ConfigurationParameter.TYPE_STRING.equals(valueType)) {
084                                            return stringValue[0];
085                                    }
086                                    throw new UIMA_IllegalArgumentException(UIMA_IllegalArgumentException.METADATA_ATTRIBUTE_TYPE_MISMATCH,
087                                                    new Object[] { valueType, "type" });
088                            }
089                            else {
090                                    if (ConfigurationParameter.TYPE_BOOLEAN.equals(valueType)) {
091                                            Boolean[] returnValues = new Boolean[stringValue.length];
092                                            for (int i = 0; i < stringValue.length; i++) {
093                                                    returnValues[i] = Boolean.parseBoolean(stringValue[i]);
094                                            }
095                                            return returnValues;
096                                    }
097                                    else if (ConfigurationParameter.TYPE_FLOAT.equals(valueType)) {
098                                            Float[] returnValues = new Float[stringValue.length];
099                                            for (int i = 0; i < stringValue.length; i++) {
100                                                    returnValues[i] = Float.parseFloat(stringValue[i]);
101                                            }
102                                            return returnValues;
103                                    }
104                                    else if (ConfigurationParameter.TYPE_INTEGER.equals(valueType)) {
105                                            Integer[] returnValues = new Integer[stringValue.length];
106                                            for (int i = 0; i < stringValue.length; i++) {
107                                                    returnValues[i] = Integer.parseInt(stringValue[i]);
108                                            }
109                                            return returnValues;
110                                    }
111                                    else if (ConfigurationParameter.TYPE_STRING.equals(valueType)) {
112                                            return stringValue;
113                                    }
114                                    throw new UIMA_IllegalArgumentException(UIMA_IllegalArgumentException.METADATA_ATTRIBUTE_TYPE_MISMATCH,
115                                                    new Object[] { valueType, "type" });
116    
117                            }
118    
119                    }
120                    else {
121                            throw new IllegalArgumentException("field is not annotated with annotation of type "
122                                            + org.uutuc.descriptor.ConfigurationParameter.class.getName());
123                    }
124            }

126            private static String getConfigurationParameterType(Field field) {
127                    Class<?> parameterClass = field.getType();
128                    String parameterClassName;
129                    if (parameterClass.isArray()) {
130                            parameterClassName = parameterClass.getComponentType().getName();
131                    }
132                    else if (Collection.class.isAssignableFrom(parameterClass)) {
133                            ParameterizedType collectionType = (ParameterizedType) field.getGenericType();
134                            parameterClassName = ((Class<?>) (collectionType.getActualTypeArguments()[0])).getName();
135                    }
136                    else {
137                            parameterClassName = parameterClass.getName();
138                    }
139                    String parameterType = javaUimaTypeMap.get(parameterClassName);
140                    if (parameterType == null) {
141                            return ConfigurationParameter.TYPE_STRING;
142                    }
143                    return parameterType;
144            }
145    
146            private static boolean isMultiValued(Field field) {
147                    Class<?> parameterClass = field.getType();
148                    if (parameterClass.isArray()) {
149                            return true;
150                    } else if (Collection.class.isAssignableFrom(parameterClass)) {
151                            return true;
152                    }
153                    return false;
154            }
155    
156            public static String getConfigurationParameterName(Field field) {
157                    if (isConfigurationParameterField(field)) {
158                            org.uutuc.descriptor.ConfigurationParameter annotation = field
159                            .getAnnotation(org.uutuc.descriptor.ConfigurationParameter.class);
160                            String name = annotation.name();
161                            if (name.equals(org.uutuc.descriptor.ConfigurationParameter.USE_FIELD_NAME)) {
162                                    name = field.getDeclaringClass().getName() + "." + field.getName();
163                            }
164                            return name;
165                    }
166                    return null;
167            }
168            
169            public static ConfigurationParameter createPrimitiveParameter(Field field) {
170                    if (isConfigurationParameterField(field)) {
171                            org.uutuc.descriptor.ConfigurationParameter annotation = field
172                                            .getAnnotation(org.uutuc.descriptor.ConfigurationParameter.class);
173                            String name = getConfigurationParameterName(field);
174                            boolean multiValued = isMultiValued(field);
175                            String parameterType = getConfigurationParameterType(field);
176                            return createPrimitiveParameter(name, parameterType, annotation.description(), multiValued, annotation
177                                            .mandatory());
178                    }
179                    else {
180                            throw new IllegalArgumentException("field is not annotated with annotation of type "
181                                            + org.uutuc.descriptor.ConfigurationParameter.class.getName());
182                    }
183            }

185            /*
186             * The UIMA_IllegalArgumentException statement was copied from
187             * org.apache.uima
188             * .resource.metadata.impl.ConfigurationParameter_impl.setType
189             */
190            public static ConfigurationParameter createPrimitiveParameter(String name, Class<?> parameterClass,
191                            String parameterDescription, boolean isMandatory) {
192                    String parameterClassName;
193                    if (parameterClass.isArray()) parameterClassName = parameterClass.getComponentType().getName();
194                    else parameterClassName = parameterClass.getName();
195    
196                    String parameterType = javaUimaTypeMap.get(parameterClassName);
197                    if (parameterType == null) {
198                            throw new UIMA_IllegalArgumentException(UIMA_IllegalArgumentException.METADATA_ATTRIBUTE_TYPE_MISMATCH,
199                                            new Object[] { parameterClassName, "type" });
200                    }
201                    return createPrimitiveParameter(name, parameterType, parameterDescription, parameterClass.isArray(),
202                                    isMandatory);
203            }
204    
205            public static ConfigurationParameter createPrimitiveParameter(String name, String parameterType,
206                            String parameterDescription, boolean isMultiValued, boolean isMandatory) {
207                    ConfigurationParameter param = new ConfigurationParameter_impl();
208                    param.setName(name);
209                    param.setType(parameterType);
210                    param.setDescription(parameterDescription);
211                    param.setMultiValued(isMultiValued);
212                    param.setMandatory(isMandatory);
213                    return param;
214            }

216            public static ConfigurationData createConfigurationData(Object... configurationData) {
217                    if (configurationData == null) {
218                            return new ConfigurationData(new ConfigurationParameter[0], new Object[0]);
219                    }
220                    if (configurationData.length % 2 != 0) {
221                            String message = "a value must be specified for each parameter name: an odd number of values passed in ("
222                                            + configurationData.length + ")";
223                            throw new IllegalArgumentException(message);
224                    }
225    
226                    int numberOfParameters = configurationData.length / 2;
227                    ConfigurationParameter[] configurationParameters = new ConfigurationParameter[numberOfParameters];
228                    Object[] configurationValues = new Object[numberOfParameters];
229    
230                    for (int i = 0; i < numberOfParameters; i++) {
231                            String name = (String) configurationData[i * 2];
232                            Object value = configurationData[i * 2 + 1];
233    
234                            if (value.getClass().isArray() && value.getClass().getComponentType().getName().equals("boolean")) {
235                                    value = ArrayUtils.toObject((boolean[]) value);
236                            }
237                            if (value.getClass().isArray() && value.getClass().getComponentType().getName().equals("int")) {
238                                    value = ArrayUtils.toObject((int[]) value);
239                            }
240                            if (value.getClass().isArray() && value.getClass().getComponentType().getName().equals("float")) {
241                                    value = ArrayUtils.toObject((float[]) value);
242                            }
243    
244                            ConfigurationParameter param = ConfigurationParameterFactory.createPrimitiveParameter(name, value
245                                            .getClass(), null, false);
246                            configurationParameters[i] = param;
247                            configurationValues[i] = value;
248                    }
249                    return new ConfigurationData(configurationParameters, configurationValues);
250            }
251    
252            public static ConfigurationData createConfigurationData(Class<?> componentClass) {
253                    List<ConfigurationParameter> configurationParameters = new ArrayList<ConfigurationParameter>();
254                    List<Object> configurationValues = new ArrayList<Object>();
255    
256                    for (Field field : ReflectionUtil.getFields(componentClass)) {
257                            if (ConfigurationParameterFactory.isConfigurationParameterField(field)) {
258                                    configurationParameters.add(ConfigurationParameterFactory.createPrimitiveParameter(field));
259                                    configurationValues.add(ConfigurationParameterFactory.getDefaultValue(field));
260                            }
261                    }
262    
263                    return new ConfigurationData(configurationParameters.toArray(new ConfigurationParameter[configurationParameters
264                                    .size()]), configurationValues.toArray(new Object[configurationValues.size()]));
265            }
266    
267            public static class ConfigurationData {
268                    public ConfigurationParameter[] configurationParameters;
269    
270                    public Object[] configurationValues;
271    
272                    public ConfigurationData(ConfigurationParameter[] configurationParameters, Object[] configurationValues) {
273                            this.configurationParameters = configurationParameters;
274                            this.configurationValues = configurationValues;
275                    }

277            }
