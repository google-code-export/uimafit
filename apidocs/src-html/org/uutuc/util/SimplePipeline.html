018    package org.uutuc.util;
019    
020    import java.io.IOException;
021    
022    import org.apache.uima.UIMAException;
023    import org.apache.uima.analysis_engine.AnalysisEngine;
024    import org.apache.uima.analysis_engine.AnalysisEngineDescription;
025    import org.apache.uima.collection.CollectionReader;
026    import org.apache.uima.jcas.JCas;
027    import org.uutuc.factory.AnalysisEngineFactory;
028    
029    /**
030     * @author Steven Bethard, Philip Ogren
031     *
032     */
033    public class SimplePipeline {
034    
035            /**
036             * Run the CollectionReader and AnalysisEngines as a pipeline.
037             * 
038             * @param reader
039             *            The CollectionReader that loads the documents into the CAS.
040             * @param descs
041             *            Primitive AnalysisEngineDescriptions that process the CAS, in
042             *            order. If you have a mix of primitive and aggregate engines,
043             *            then please create the AnalysisEngines yourself and call the
044             *            other runPipeline method.
045             */
046            public static void runPipeline(CollectionReader reader, AnalysisEngineDescription... descs) throws UIMAException,
047                            IOException {
048                    AnalysisEngine[] engines = new AnalysisEngine[descs.length];
049                    for (int i = 0; i < engines.length; ++i) {
050                            if (descs[i].isPrimitive()) {
051                                    engines[i] = AnalysisEngineFactory.createPrimitive(descs[i]);
052                            }
053                            else {
054                                    engines[i] = AnalysisEngineFactory.createAggregate(descs[i]);
055                            }
056                    }
057                    runPipeline(reader, engines);
058            }
059    
060            public static void runPipeline(CollectionReader reader, AnalysisEngine... engines) throws UIMAException,
061                            IOException {
062                    for (JCas jCas : new JCasIterable(reader, engines)) {
063                            assert jCas != null;
064                    }
065                    for (AnalysisEngine engine : engines) {
066                            engine.collectionProcessComplete();
067                    }
068                    reader.close();
069            }
070    
071    }
