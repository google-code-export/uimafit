069             * Get an AnalysisEngine from the name (Java-style, dotted) of an XML
070             * descriptor file, and a set of configuration parameters.
071             * 
072             * @param descriptorName
073             *            The fully qualified, Java-style, dotted name of the XML
074             *            descriptor file.
075             * @param configurationData
076             *            Any additional configuration parameters to be set. These
077             *            should be supplied as (name, value) pairs, so there should
078             *            always be an even number of parameters.
079             * @return The AnalysisEngine created from the XML descriptor and the
080             *         configuration parameters.
081             * @throws IOException
082             * @throws UIMAException
083             */
084            public static AnalysisEngine createAnalysisEngine(String descriptorName, Object... configurationData)
085                            throws UIMAException, IOException {
086                    AnalysisEngineDescription aed = createAnalysisEngineDescription(descriptorName, configurationData);
087                    return UIMAFramework.produceAnalysisEngine(aed);
088            }

090            public static AnalysisEngineDescription createAnalysisEngineDescription(String descriptorName,
091                            Object... configurationData) throws UIMAException, IOException {
092                    Import_impl imprt = new Import_impl();
093                    imprt.setName(descriptorName);
094                    URL url = imprt.findAbsoluteUrl(UIMAFramework.newDefaultResourceManager());
095                    ResourceSpecifier specifier = ResourceCreationSpecifierFactory.createResourceCreationSpecifier(url,
096                                    configurationData);
097                    return (AnalysisEngineDescription) specifier;
098            }

101             * This method provides a convenient way to instantiate an AnalysisEngine
102             * where the default view is mapped to the view name passed into the method.
103             * 
104             * @param analysisEngineDescription
105             * @param viewName
106             *            the view name to map the default view to
107             * @return an aggregate analysis engine consisting of a single component
108             *         whose default view is mapped to the the view named by viewName.
109             * @throws ResourceInitializationException
110             */
111            public static AnalysisEngine createAnalysisEngine(AnalysisEngineDescription analysisEngineDescription,
112                            String viewName) throws ResourceInitializationException {
113                    AggregateBuilder builder = new AggregateBuilder();
114                    builder.add(analysisEngineDescription, CAS.NAME_DEFAULT_SOFA, viewName);
115                    return builder.createAggregate();
116            }

194            public static AnalysisEngineDescription createPrimitiveDescription(
195                            Class<? extends AnalysisComponent> componentClass, TypeSystemDescription typeSystem,
196                            TypePriorities typePriorities, Capability[] capabilities, ConfigurationParameter[] configurationParameters,
197                            Object[] configurationValues) throws ResourceInitializationException {
198    
199                    // create the descriptor and set configuration parameters
200                    AnalysisEngineDescription desc = new AnalysisEngineDescription_impl();
201                    desc.setFrameworkImplementation(Constants.JAVA_FRAMEWORK_NAME);
202                    desc.setPrimitive(true);
203                    desc.setAnnotatorImplementationName(componentClass.getName());
204                    org.uutuc.descriptor.AnalysisComponent componentAnno = ReflectionUtil.getInheritableAnnotation(
205                                    org.uutuc.descriptor.AnalysisComponent.class, componentClass);
206                    if (componentAnno != null) {
207                            OperationalProperties op = desc.getAnalysisEngineMetaData().getOperationalProperties();
208                            op.setMultipleDeploymentAllowed(componentAnno.multipleDeploymentAllowed());
209                            op.setModifiesCas(componentAnno.modifiesCas());
210                            op.setOutputsNewCASes(componentAnno.outputsNewCases());
211                    }
212                    else {
213                            OperationalProperties op = desc.getAnalysisEngineMetaData().getOperationalProperties();
214                            op.setMultipleDeploymentAllowed(org.uutuc.descriptor.AnalysisComponent.MULTIPLE_DEPLOYMENT_ALLOWED_DEFAULT);
215                            op.setModifiesCas(org.uutuc.descriptor.AnalysisComponent.MODIFIES_CAS_DEFAULT);
216                            op.setOutputsNewCASes(org.uutuc.descriptor.AnalysisComponent.OUTPUTS_NEW_CASES_DEFAULT);
217                    }
218    
219                    AnalysisEngineMetaData meta = desc.getAnalysisEngineMetaData();
220                    meta.setName(componentClass.getName());
221                    meta.setVendor(componentClass.getPackage().getName());
222                    meta.setDescription("Descriptor automatically generated by UUTUC");
223                    meta.setVersion("unknown");
224    
225                    // Extract external resource dependencies
226                    Collection<ExternalResourceDependency> deps = ExternalResourceConfigurator.getResourceDeclarations(
227                                    componentClass).values();
228                    desc.setExternalResourceDependencies(deps.toArray(new ExternalResourceDependency[deps.size()]));
229    
230                    ConfigurationData reflectedConfigurationData = ConfigurationParameterFactory
231                                    .createConfigurationData(componentClass);
232                    ResourceCreationSpecifierFactory.setConfigurationParameters(desc,
233                                    reflectedConfigurationData.configurationParameters, reflectedConfigurationData.configurationValues);
234                    if (configurationParameters != null) {
235                            ResourceCreationSpecifierFactory.setConfigurationParameters(desc, configurationParameters,
236                                            configurationValues);
237                    }
238    
239                    // set the type system
240                    if (typeSystem != null) {
241                            desc.getAnalysisEngineMetaData().setTypeSystem(typeSystem);
242                    }
243    
244                    if (typePriorities != null) {
245                            desc.getAnalysisEngineMetaData().setTypePriorities(typePriorities);
246                    }
247    
248                    if (capabilities == null) {
249                            Capability capability = CapabilityFactory.createCapability(componentClass);
250                            if (capability != null) {
251                                    capabilities = new Capability[] {capability};
252                            }
253                    }
254                    if (capabilities != null) {
255                            desc.getAnalysisEngineMetaData().setCapabilities(capabilities);
256                    }
257                    return desc;
258            }

279            @SuppressWarnings("unchecked")
280            public static AnalysisEngine createPrimitive(AnalysisEngineDescription desc, Object... configurationData)
281                            throws ResourceInitializationException {
282                    AnalysisEngine engine = new PrimitiveAnalysisEngine_impl();
283    
284                    if (configurationData != null) {
285                            ConfigurationData cdata = ConfigurationParameterFactory.createConfigurationData(configurationData);
286                            ConfigurationParameter[] configurationParameters = cdata.configurationParameters;
287                            Object[] configurationValues = cdata.configurationValues;
288                            ResourceCreationSpecifierFactory.setConfigurationParameters(desc, configurationParameters,
289                                            configurationValues);
290                            ResourceMetaData metaData = desc.getMetaData();
291                            ConfigurationParameterSettings paramSettings = metaData.getConfigurationParameterSettings();
292                            Map additionalParameters = new HashMap();
293                            additionalParameters.put(AnalysisEngine.PARAM_CONFIG_PARAM_SETTINGS, paramSettings);
294                            engine.initialize(desc, additionalParameters);
295                    }
296                    else {
297                            engine.initialize(desc, null);
298                    }
299                    return engine;
300            }

320            public static AnalysisEngineDescription createAggregateDescription(
321                            List<Class<? extends AnalysisComponent>> componentClasses, TypeSystemDescription typeSystem,
322                            TypePriorities typePriorities, SofaMapping[] sofaMappings, Object... configurationParameters)
323                            throws ResourceInitializationException {
324    
325                    List<AnalysisEngineDescription> primitiveEngineDescriptions = new ArrayList<AnalysisEngineDescription>();
326                    List<String> componentNames = new ArrayList<String>();
327    
328                    for (Class<? extends AnalysisComponent> componentClass : componentClasses) {
329                            AnalysisEngineDescription primitiveDescription = createPrimitiveDescription(componentClass, typeSystem,
330                                            typePriorities, configurationParameters);
331                            primitiveEngineDescriptions.add(primitiveDescription);
332                            componentNames.add(componentClass.getName());
333                    }
334                    return createAggregateDescription(primitiveEngineDescriptions, componentNames, typeSystem, typePriorities,
335                                    sofaMappings);
336            }

351            public static AnalysisEngineDescription createAggregateDescription(
352                            AnalysisEngineDescription... analysisEngineDescriptions) throws ResourceInitializationException {
353                    String[] names = new String[analysisEngineDescriptions.length];
354                    int i = 0;
355                    for (AnalysisEngineDescription aed : analysisEngineDescriptions) {
356                            names[i] = aed.getImplementationName() + "-" + i;
357                            i++;
358                    }
359    
360                    return createAggregateDescription(asList(analysisEngineDescriptions), asList(names), null, null, null);
361            }

363            @SuppressWarnings("unchecked")
364            public static AnalysisEngineDescription createAggregateDescription(
365                            List<AnalysisEngineDescription> analysisEngineDescriptions, List<String> componentNames,
366                            TypeSystemDescription typeSystem, TypePriorities typePriorities, SofaMapping[] sofaMappings)
367                            throws ResourceInitializationException {
368    
369                    // create the descriptor and set configuration parameters
370                    AnalysisEngineDescription desc = new AnalysisEngineDescription_impl();
371                    desc.setFrameworkImplementation(Constants.JAVA_FRAMEWORK_NAME);
372                    desc.setPrimitive(false);
373    
374                    // if any of the aggregated analysis engines does not allow multiple
375                    // deployment, then the
376                    // aggregate engine may also not be multiply deployed
377                    boolean allowMultipleDeploy = true;
378                    for (AnalysisEngineDescription d : analysisEngineDescriptions) {
379                            allowMultipleDeploy &= d.getAnalysisEngineMetaData().getOperationalProperties()
380                                            .isMultipleDeploymentAllowed();
381                    }
382                    desc.getAnalysisEngineMetaData().getOperationalProperties().setMultipleDeploymentAllowed(allowMultipleDeploy);
383    
384                    List<String> flowNames = new ArrayList<String>();
385    
386                    for (int i = 0; i < analysisEngineDescriptions.size(); i++) {
387                            AnalysisEngineDescription aed = analysisEngineDescriptions.get(i);
388                            String componentName = componentNames.get(i);
389                            desc.getDelegateAnalysisEngineSpecifiersWithImports().put(componentName, aed);
390                            flowNames.add(componentName);
391                    }
392    
393                    FixedFlow fixedFlow = new FixedFlow_impl();
394                    fixedFlow.setFixedFlow(flowNames.toArray(new String[flowNames.size()]));
395                    desc.getAnalysisEngineMetaData().setFlowConstraints(fixedFlow);
396    
397                    if (typePriorities != null) {
398                            desc.getAnalysisEngineMetaData().setTypePriorities(typePriorities);
399                    }
400    
401                    if (sofaMappings != null) {
402                            desc.setSofaMappings(sofaMappings);
403                    }
404    
405                    return desc;
406            }

408            /**
409             * Creates an AnalysisEngine from the given descriptor, and uses the engine
410             * to process the file or text.
411             * 
412             * @param descriptorFileName
413             *            The fully qualified, Java-style, dotted name of the XML
414             *            descriptor file.
415             * @param fileNameOrText
416             *            Either the path of a file to be loaded, or a string to use as
417             *            the text. If the string given is not a valid path in the file
418             *            system, it will be assumed to be text.
419             * @return A JCas object containing the processed document.
420             * @throws IOException
421             * @throws UIMAException
422             */
423            public static JCas process(String descriptorFileName, String fileNameOrText) throws IOException, UIMAException {
424                    AnalysisEngine engine = createAnalysisEngine(descriptorFileName);
425                    JCas jCas = process(engine, fileNameOrText);
426                    engine.collectionProcessComplete();
427                    return jCas;
428            }
429    
430            /**
431             * Processes the file or text with the given AnalysisEngine.
432             * 
433             * @param analysisEngine
434             *            The AnalysisEngine object to process the text.
435             * @param fileNameOrText
436             *            Either the path of a file to be loaded, or a string to use as
437             *            the text. If the string given is not a valid path in the file
438             *            system, it will be assumed to be text.
439             * @return A JCas object containing the processed document.
440             * @throws IOException
441             * @throws UIMAException
442             */
443            public static JCas process(AnalysisEngine analysisEngine, String fileNameOrText) throws IOException, UIMAException {
444                    File textFile = new File(fileNameOrText);
445                    String text;
446                    if (textFile.exists()) {
447                            text = FileUtils.file2String(textFile);
448                    }
449                    else {
450                            text = fileNameOrText;
451                    }
452    
453                    JCas jCas = analysisEngine.newJCas();
454                    jCas.setDocumentText(text);
455                    analysisEngine.process(jCas);
456                    return jCas;
457            }
