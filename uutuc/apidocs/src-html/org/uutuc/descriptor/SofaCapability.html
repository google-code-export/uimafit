018    package org.uutuc.descriptor;
019    
020    import java.lang.annotation.ElementType;
021    import java.lang.annotation.Retention;
022    import java.lang.annotation.RetentionPolicy;
023    import java.lang.annotation.Target;
024    
025    /**
026     * @author Philip Ogren
027     * 
028     *         Adding this annotation to your analysis engine description makes your
029     *         component "sofa aware." The base CAS delivered to "sofa aware"
030     *         components has no explicit view associated with it. The logic is that
031     *         it is impossible to know the intent of a sofa aware component and it
032     *         should use getView as needed. You should therefore be aware that if
033     *         you need to work with the "_InitialView" view, then you must
034     *         explicitly request it with a call like:
035     * 
036     *         JCas initialView = jCas.getView(CAS.NAME_DEFAULT_SOFA);
037     * 
038     *         This is because the base CAS that it passed into the process method
039     *         to "sofa aware" components is not the same as the "_InitialView". See
040     *         how the following member variable is used to understand why/how:
041     * 
042     *         org.apache.uima.analysis_engine.impl.PrimitiveAnalysisEngine_impl.
043     *         mSofaAware
044     */
045    
046    @Retention(RetentionPolicy.RUNTIME)
047    @Target(ElementType.TYPE)
048    public @interface SofaCapability {
049            String[] inputSofas() default NO_DEFAULT_VALUE;
050    
051            String[] outputSofas() default NO_DEFAULT_VALUE;
052    
053            public static final String NO_DEFAULT_VALUE = "org.uutuc.descriptor.ConfigurationParameter.NO_DEFAULT_VALUE";
054    
055    }
