019    package org.uutuc.component;
020    
021    import org.apache.uima.UimaContext;
022    import org.apache.uima.resource.ResourceInitializationException;
023    import org.uutuc.factory.ExternalResourceConfigurator;
024    import org.uutuc.util.InitializeUtil;
025    
026    import edu.umd.cs.findbugs.annotations.OverrideMustInvoke;
027    
028    /**
029     * Base class for CAS annotators which initializes itself based on annotations.
030     *
031     * @author Richard Eckart de Castilho
032     */
033    public abstract class CasAnnotator_ImplBase
034            extends org.apache.uima.analysis_component.CasAnnotator_ImplBase
035    {
036            @Override
037            @OverrideMustInvoke
038            public void initialize(UimaContext aContext)
039                    throws ResourceInitializationException
040            {
041                    super.initialize(aContext);
042                    InitializeUtil.initialize(this, aContext);
043                    ExternalResourceConfigurator.configure(aContext, this);
044            }
045    }
